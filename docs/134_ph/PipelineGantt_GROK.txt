#### A. `EfficiencyKPI.tsx` — главный индекс эффективности (Overview)

```tsx
import { useMemo } from 'react';
import { ResponsiveContainer, RadialBarChart, RadialBar, PolarAngleAxis, Tooltip } from 'recharts';

const COLORS = { accent: '#4a6b8a', success: '#6a8a6a', warning: '#aa8a6a' };

interface EfficiencyKPIProps {
  successRate: number;      // 0-100
  avgDuration: number;      // секунды
  tokensPerSubtask: number;
  knowledgeLevel: number;   // 0-1
}

export default function EfficiencyKPI({ successRate, avgDuration, tokensPerSubtask, knowledgeLevel }: EfficiencyKPIProps) {
  const efficiencyScore = useMemo(() => {
    const speedScore = Math.max(0, 100 - (avgDuration / 60) * 10); // <60s = 100
    const tokenScore = Math.max(0, 100 - (tokensPerSubtask / 800) * 50);
    const klScore = knowledgeLevel * 100;
    return Math.round((successRate * 0.4) + (speedScore * 0.25) + (tokenScore * 0.2) + (klScore * 0.15));
  }, [successRate, avgDuration, tokensPerSubtask, knowledgeLevel]);

  const data = [{ name: 'Efficiency', value: efficiencyScore, fill: efficiencyScore > 85 ? COLORS.success : COLORS.accent }];

  return (
    <div style={{ background: '#111', padding: 16, borderRadius: 4, textAlign: 'center' }}>
      <div style={{ fontSize: 11, color: '#aaa', marginBottom: 8 }}>EFFICIENCY INDEX</div>
      <ResponsiveContainer width="100%" height={160}>
        <RadialBarChart cx="50%" cy="50%" innerRadius="70%" outerRadius="90%" data={data} startAngle={180} endAngle={-180}>
          <PolarAngleAxis type="number" domain={[0, 100]} tick={false} />
          <RadialBar dataKey="value" cornerRadius={10} background />
          <text x="50%" y="50%" textAnchor="middle" dominantBaseline="middle" style={{ fontSize: 32, fontWeight: 600, fill: '#e0e0e0' }}>
            {efficiencyScore}
          </text>
        </RadialBarChart>
      </ResponsiveContainer>
      <div style={{ display: 'flex', justifyContent: 'space-around', fontSize: 10, color: '#888', marginTop: 8 }}>
        <div>Success {successRate}%</div>
        <div>Speed {Math.round(100 - (avgDuration / 60) * 10)}%</div>
        <div>Tokens {Math.round(100 - (tokensPerSubtask / 800) * 50)}%</div>
      </div>
    </div>
  );
}
```

#### C. `PipelineGantt.tsx` — живой Gantt с RALF retries (Pipelines tab)

Recharts не идеален для Gantt, но можно сделать кастомный BarChart (горизонтальный) или добавить `react-flow` (рекомендую для "Arena").

Простой вариант на Recharts (быстро):

```tsx
import { BarChart, Bar, XAxis, YAxis, Tooltip, ResponsiveContainer, ReferenceLine } from 'recharts';

export default function PipelineGantt({ runs }: { runs: any[] }) {
  // runs из /api/pipeline/history
  return (
    <ResponsiveContainer width="100%" height={420}>
      <BarChart data={runs} layout="vertical" margin={{ left: 120 }}>
        <XAxis type="number" stroke="#444" />
        <YAxis type="category" dataKey="task_title" stroke="#444" width={110} fontSize={11} />
        <Bar dataKey="total_duration_s" fill="#4a6b8a" radius={[0, 4, 4, 0]} />
        {/* RALF retries как маленькие красные сегменты */}
        {runs.map((run, i) => run.ralf_retries > 0 && (
          <ReferenceLine key={i} x={run.total_duration_s} stroke="#8a6a6a" strokeDasharray="3 3" label={`RALF x${run.ralf_retries}`} />
        ))}
        <Tooltip />
      </BarChart>
    </ResponsiveContainer>
  );
}
```

Для полноценной **Pipeline Arena** (живой граф с узлами-агентами и ребрами-retries) — установи `npm install xyflow` (очень лёгкий в 2026) и сделай узлы с цветом по efficiency.
