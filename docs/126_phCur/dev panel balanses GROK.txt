### План (как будем делать)

1. Создаём сервис `BalanceTracker` — единое место для всех ключей.
2. После **каждого** вызова LLM (в `llm_call_tool.py`) обновляем расход токенов.
3. Для провайдеров, у которых есть публичный баланс API — периодически (раз в 30–60 сек) обновляем реальный остаток.
4. Выводим красивую таблицу в DevPanel (новая вкладка "Balances").

### 1. Новый файл: `src/orchestration/balance_tracker.py`

Создай этот файл:

```python
"""
Balance Tracker — единый мониторинг расхода и баланса всех API ключей
Phase 125.1: Реал-тайм трекинг токенов и денег по всем провайдерам
"""

import json
import time
import asyncio
import logging
from pathlib import Path
from datetime import datetime
from typing import Dict, Any, Optional
from dataclasses import dataclass, asdict, field

logger = logging.getLogger("BALANCE_TRACKER")

BALANCES_FILE = Path(__file__).parent.parent.parent / "data" / "balances.json"


@dataclass
class ProviderBalance:
    provider: str
    key_name: str                    # например: "openrouter_0", "xai_main"
    model: str = ""
    tokens_in: int = 0
    tokens_out: int = 0
    cost_usd: float = 0.0
    balance_usd: float = 0.0         # реальный остаток, если удалось получить
    last_updated: float = 0.0
    last_balance_check: float = 0.0


class BalanceTracker:
    _instance = None

    def __new__(cls):
        if cls._instance is None:
            cls._instance = super().__new__(cls)
            cls._instance._initialized = False
        return cls._instance

    def __init__(self):
        if self._initialized:
            return
        self._initialized = True
        self.balances: Dict[str, ProviderBalance] = {}
        self._load()
        logger.info("[BalanceTracker] Initialized")

    def _load(self):
        if BALANCES_FILE.exists():
            try:
                data = json.loads(BALANCES_FILE.read_text())
                for key, val in data.items():
                    self.balances[key] = ProviderBalance(**val)
            except Exception as e:
                logger.error(f"[BalanceTracker] Load failed: {e}")

    def _save(self):
        try:
            data = {k: asdict(v) for k, v in self.balances.items()}
            BALANCES_FILE.write_text(json.dumps(data, indent=2, default=str))
        except Exception as e:
            logger.error(f"[BalanceTracker] Save failed: {e}")

    def record_usage(self, provider: str, key_name: str, model: str, tokens_in: int, tokens_out: int, cost_usd: float = 0.0):
        """Вызывать после каждого LLM вызова"""
        key = f"{provider}_{key_name}"
        
        if key not in self.balances:
            self.balances[key] = ProviderBalance(provider=provider, key_name=key_name, model=model)

        b = self.balances[key]
        b.tokens_in += tokens_in
        b.tokens_out += tokens_out
        b.cost_usd += cost_usd
        b.last_updated = time.time()

        self._save()
        logger.debug(f"[Balance] {provider} {model}: +{tokens_in}+{tokens_out} tokens, ${cost_usd:.4f}")

    async def refresh_balance(self, provider: str, key_name: str):
        """Пытаемся получить реальный баланс через API (где возможно)"""
        # Здесь будем добавлять провайдеры по одному
        key = f"{provider}_{key_name}"
        if key not in self.balances:
            return

        b = self.balances[key]
        b.last_balance_check = time.time()

        try:
            if provider == "openrouter":
                from src.orchestration.services.api_key_service import APIKeyService
                # OpenRouter имеет /api/v1/auth/key endpoint
                # Можно реализовать позже
                pass
            elif provider == "anthropic":
                # Anthropic пока не даёт публичный баланс
                pass
            # xAI, Gemini, DeepSeek — тоже пока слабая поддержка
        except Exception as e:
            logger.debug(f"[Balance] Refresh failed for {provider}: {e}")

    def get_all_balances(self) -> list:
        """Для DevPanel"""
        return [asdict(b) for b in self.balances.values()]

    def get_total_cost_today(self) -> float:
        """Пример агрегации"""
        return sum(b.cost_usd for b in self.balances.values())


# Singleton
_balance_tracker = None

def get_balance_tracker() -> BalanceTracker:
    global _balance_tracker
    if _balance_tracker is None:
        _balance_tracker = BalanceTracker()
    return _balance_tracker
```

### 2. Новый файл: `src/components/dev/BalancesPanel.tsx`

Создай этот файл:

```tsx
/**
 * BalancesPanel — мониторинг расхода и баланса всех API ключей
 * Стиль: холодный, минималистичный, Nolan Batman
 */

import { useState, useEffect } from 'react';

interface BalanceEntry {
  provider: string;
  key_name: string;
  model: string;
  tokens_in: number;
  tokens_out: number;
  cost_usd: number;
  balance_usd: number;
  last_updated: number;
}

export function BalancesPanel() {
  const [balances, setBalances] = useState<BalanceEntry[]>([]);
  const [loading, setLoading] = useState(false);
  const [totalCost, setTotalCost] = useState(0);

  const fetchBalances = async () => {
    setLoading(true);
    try {
      const res = await fetch('http://localhost:5001/api/debug/balances');
      if (res.ok) {
        const data = await res.json();
        setBalances(data.balances || []);
        setTotalCost(data.total_cost_today || 0);
      }
    } catch (err) {
      console.error('[Balances] Fetch failed:', err);
    } finally {
      setLoading(false);
    }
  };

  useEffect(() => {
    fetchBalances();
    const interval = setInterval(fetchBalances, 15000); // обновление каждые 15 сек
    return () => clearInterval(interval);
  }, []);

  return (
    <div style={{ padding: '12px 16px', fontSize: 13, color: '#ddd' }}>
      {/* Header */}
      <div style={{ 
        display: 'flex', 
        justifyContent: 'space-between', 
        alignItems: 'center',
        marginBottom: 16,
        paddingBottom: 12,
        borderBottom: '1px solid #222'
      }}>
        <div>
          <div style={{ fontSize: 15, fontWeight: 600, color: '#ccc' }}>API Balances</div>
          <div style={{ color: '#666', fontSize: 12 }}>
            Total spent today: <span style={{ color: '#f66' }}>${totalCost.toFixed(4)}</span>
          </div>
        </div>
        <button
          onClick={fetchBalances}
          disabled={loading}
          style={{
            padding: '6px 12px',
            background: '#1a1a1a',
            border: '1px solid #333',
            borderRadius: 4,
            color: '#aaa',
            fontSize: 12,
            cursor: 'pointer'
          }}
        >
          {loading ? 'Refreshing...' : 'Refresh'}
        </button>
      </div>

      {/* Таблица */}
      <div style={{ overflowX: 'auto' }}>
        <table style={{ width: '100%', borderCollapse: 'collapse', fontSize: 12 }}>
          <thead>
            <tr style={{ borderBottom: '1px solid #222', color: '#666' }}>
              <th style={{ textAlign: 'left', padding: '8px 0' }}>Provider</th>
              <th style={{ textAlign: 'left', padding: '8px 0' }}>Key</th>
              <th style={{ textAlign: 'left', padding: '8px 0' }}>Model</th>
              <th style={{ textAlign: 'right', padding: '8px 0' }}>Tokens In</th>
              <th style={{ textAlign: 'right', padding: '8px 0' }}>Tokens Out</th>
              <th style={{ textAlign: 'right', padding: '8px 0' }}>Cost $</th>
              <th style={{ textAlign: 'right', padding: '8px 0' }}>Balance $</th>
              <th style={{ textAlign: 'right', padding: '8px 0' }}>Updated</th>
            </tr>
          </thead>
          <tbody>
            {balances.map((b, i) => (
              <tr key={i} style={{ borderBottom: '1px solid #1a1a1a' }}>
                <td style={{ padding: '9px 0', color: '#aaa' }}>{b.provider}</td>
                <td style={{ padding: '9px 0', fontFamily: 'monospace', color: '#777' }}>
                  {b.key_name}
                </td>
                <td style={{ padding: '9px 0', color: '#888' }}>{b.model || '-'}</td>
                <td style={{ padding: '9px 0', textAlign: 'right', color: '#4a6' }}>
                  {b.tokens_in.toLocaleString()}
                </td>
                <td style={{ padding: '9px 0', textAlign: 'right', color: '#6a8' }}>
                  {b.tokens_out.toLocaleString()}
                </td>
                <td style={{ padding: '9px 0', textAlign: 'right', color: '#f66' }}>
                  ${b.cost_usd.toFixed(4)}
                </td>
                <td style={{ padding: '9px 0', textAlign: 'right', color: b.balance_usd > 0 ? '#4ade80' : '#f66' }}>
                  ${b.balance_usd.toFixed(3)}
                </td>
                <td style={{ padding: '9px 0', textAlign: 'right', color: '#555', fontSize: 11 }}>
                  {new Date(b.last_updated * 1000).toLocaleTimeString([], { hour: '2-digit', minute: '2-digit' })}
                </td>
              </tr>
            ))}
          </tbody>
        </table>
      </div>

      {balances.length === 0 && (
        <div style={{ textAlign: 'center', color: '#555', padding: 40 }}>
          No balance data yet. Run some pipelines first.
        </div>
      )}
    </div>
  );
}
```
