def _extract_imports_quick(self, content: str, ext: str) -> List[str]:
    """Быстрая регулярка для базовых imports"""
    if ext == '.py':
        return re.findall(r'^(?:import|from)\s+([\w\.]+)', content, re.MULTILINE)[:10]  # max 10
    elif ext in ['.js', '.ts']:
        return re.findall(r"(?:import|require).*?['\"]([^'\"]+)['\"]", content)[:10]
    return []
```

# src/layout/dep_calculator.py - новый файл
def calculate_dep_score(source_node, target_node) -> float:
    """Ваша формула DEP(A→B) = σ(...)"""
    
    # Берём данные из Qdrant
    A_imports = source_node.get('imports', [])
    B_imports = target_node.get('imports', [])
    
    # I(A,B) = B imports A?
    I = 1.0 if source_node['path'] in B_imports else 0.0
    
    # S(A,B) = cosine similarity (уже есть в embeddings)
    S = cosine_similarity(source_node['vector'], target_node['vector'])
    
    # ΔT = временная разница
    A_time = source_node['timestamp']
    B_time = target_node['timestamp']
    delta_days = max(0, (B_time - A_time) / 86400)
    E = 0.2 + 0.8 * math.exp(-delta_days/30)
    
    # Применяем формулу
    raw_score = 0.4*I + 0.35*S*E + 0.2*0 + 0.05*1.0  # R и RRF пока константы
    final_score = sigmoid(raw_score, center=0.35)
    
    return max(0.0, min(1.0, final_score))
```

### 4. Интеграция в Layout (2 часа)
```python
# В knowledge_layout.py - заменить edge calculation
def build_prerequisite_edges_real(files_data):
    """Новые рёбра на основе DEP формулы"""
    edges = []
    
    for i, file_a in enumerate(files_data):
        for j, file_b in enumerate(files_data):
            if i == j: continue
            
            dep_score = calculate_dep_score(file_a, file_b)
            
            if dep_score > 0.6:  # threshold
                edges.append(KnowledgeEdge(
                    source=file_a['path'],      # Причина
                    target=file_b['path'],      # Следствие
                    weight=dep_score,
                    type='import_dependency'
                ))
    
    return edges
```
