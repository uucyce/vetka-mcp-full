### 1. **FileCard.tsx: Добавить Double-Click Detection** (Строки ~890+, в handleClick блок)
```typescript
// ✅ ДОБАВИТЬ В FileCard.tsx (новые импорты/хуки)
import { useState, useRef, useCallback } from 'react'; // Уже есть, но убедитесь
import type { ThreeEvent } from '@react-three/fiber';
import { useStore } from '../../store'; // Подтвердите путь к store

// В компоненте FileCard:
const [clickCount, setClickCount] = useState(0);
const clickTimer = useRef<NodeJS.Timeout>();

const handleClick = useCallback((e: ThreeEvent<MouseEvent>) => {
  // Существующий код... (Ctrl/Shift/drag/chat/artifact)
  
  // ✅ НОВОЕ: Double-click логика (после e.stopPropagation())
  const now = Date.now();
  setClickCount(prev => {
    if (now - (clickTimer.current?.['time'] || 0) < 300) { // 300ms threshold
      // DOUBLE CLICK!
      clickTimer.current = undefined;
      if (type === 'file') {
        // Открыть артефакт (как ChestIcon)
        window.dispatchEvent(new CustomEvent('vetka-open-artifact', {
          detail: { 
            artifactId: id, // или из metadata/artifactId если есть
            fileName: name, 
            filePath: path, 
            status: 'L1', // default, или из props
            artifactType: type 
          }
        }));
        return 0;
      } else if (type === 'folder' || type === 'branch') {
        // Zoom камеры к ветке
        const setCameraCommand = useStore.getState().setCameraCommand;
        setCameraCommand({ target: path, zoom: 'medium', highlight: true });
        return 0;
      }
    } else {
      // Single click timer
      if (clickTimer.current) clearTimeout(clickTimer.current);
      clickTimer.current = setTimeout(() => setClickCount(0), 300);
      clickTimer.current['time'] = now;
      return 1;
    }
    return prev;
  });
  
  if (clickCount > 0) return; // Не выполнять single-click если double
  
  // Существующий single-click: onClick?.();
}, [/* deps + */ type, id, name, path /* + useStore */]);
```
