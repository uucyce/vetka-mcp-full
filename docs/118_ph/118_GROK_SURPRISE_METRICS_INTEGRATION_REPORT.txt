```python
@router.post("/api/cam/feedback")
async def cam_feedback(data: dict):
    from src.orchestration.cam_engine import update_activation_score  # lazy import
    node_id = data['node_id']

    weight = emoji_weights.get(reaction, 0)

    return {"status": "updated"}
```

### **Fix 2: cam_engine.py — Weights + update_activation_score**

```python
# Добавить глобально (ли. ~184)
EMOJI_WEIGHTS = {'👍': 0.2, '❤️': 0.3, '🔥': 0.25, '💡': 0.15, '❌': -0.1}

# Новая func (ли. ~211 после user_reactions)
def update_activation_score(node_id: str, weight_adjustment: float):
    node = find_node(node_id)  # assume exists
    old_score = node.metadata.get('activation_score', 0.5)
    new_score = min(1.0, max(0.0, old_score + weight_adjustment))
    node.metadata['activation_score'] = new_score
    if 'user_reactions' not in node.metadata:
        node.metadata['user_reactions'] = []
    node.metadata['user_reactions'].append({
        'reaction': reaction, 'weight': weight_adjustment, 'ts': datetime.now().isoformat()
    })[-5:]  # last 5
    save_node_metadata(node)  # persist
```
