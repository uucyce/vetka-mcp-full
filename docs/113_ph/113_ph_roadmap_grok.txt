### 1. **113.1: Persistent Spatial Memory** (`useStore.ts` + Backend)
```typescript
// useStore.ts (add)
export const useStore = create<Store>((set, get) => ({
  // ... existing
  persistPositions: true,  // Toggle
  saveLayout: () => {
    const nodes = get().nodes;
    const layout = nodes.map(n => ({ id: n.id, position: n.position }));
    localStorage.setItem('vetka_layout', JSON.stringify(layout));  // Client
    fetch('/api/layout/save', { method: 'POST', body: JSON.stringify({ layout }) });  // Backend persist
  },
  loadLayout: () => {
    const saved = localStorage.getItem('vetka_layout');
    if (saved) {
      const layout = JSON.parse(saved);
      layout.forEach(({ id, position }: any) => updateNodePosition(id, position));
    }
  },
}));

// FileCard.tsx: handlePointerUp → store.saveLayout()
const handlePointerUp = useCallback((e: any) => {
  // ... existing
  useStore.getState().saveLayout();  // Auto-save on drag end
}, []);
```

**Backend** (`backend/routes/layout.py`):
```python
@app.post("/api/layout/save")

    with open("vetka_layout.json", "w") as f:  # Or DB

```

### 2. **113.2: Smooth Drag Physics** (`FileCard.tsx`)
```typescript
// Add refs
const velocity = useRef(new THREE.Vector3());
const dragSpring = { k: 0.15, friction: 0.9 };  // Tunable

// handlePointerMove
const handlePointerMove = useCallback((e: any) => {
  if (!isDragging) return;
  // ... raycast newPos
  const delta = newPos.clone().sub(meshRef.current!.position);
  velocity.current.add(delta.multiplyScalar(0.1));  // Accumulate
  velocity.current.multiplyScalar(dragSpring.friction);  // Dampen
  meshRef.current!.position.add(velocity.current);  // Spring to target

  // Update store (throttled)
  // ... moveNodeWithChildren
}, [isDragging]);

// useFrame: Auto-dampen when !dragging
useFrame(() => {
  if (!isDragging && velocity.current.length() > 0.01) {
    velocity.current.multiplyScalar(0.95);  // Settle
    meshRef.current!.position.add(velocity.current.clone().multiplyScalar(0.5));
    if (type === 'folder') moveNodeWithChildren(id, meshRef.current!.position);
  }
});
```
**Effect**: Drag "slides" to rest, feels organic.
