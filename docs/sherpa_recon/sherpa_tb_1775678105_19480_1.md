"""
STM Manager – short‑term memory cache with TTL.
Used to store frequently accessed role memories during a session.
"""
import time
import threading
from typing import Any, Optional, Dict

class STMManager:
    _instance = None
    _lock = threading.Lock()

    def __new__(cls):
        if cls._instance is None:
            with cls._lock:
                if cls._instance is None:
                    cls._instance = super().__new__(cls)
                    cls._instance._store = {}
                    cls._instance._default_ttl = 86400  # 24 hours
        return cls._instance

    def set(self, key: str, value: Any, ttl_seconds: Optional[int] = None) -> None:
        """Store a value with optional TTL (default 24h)."""
        expiry = time.time() + (ttl_seconds or self._default_ttl)
        with self._lock:
            self._store[key] = (value, expiry)

    def get(self, key: str) -> Optional[Any]:
        """Return value if exists and not expired, else None."""
        with self._lock:
            entry = self._store.get(key)
            if entry is None:
                return None
            value, expiry = entry
            if time.time() > expiry:
                del self._store[key]
                return None
            return value

    def delete(self, key: str) -> None:
        """Explicitly remove a key."""
        with self._lock:
            self._store.pop(key, None)

    def clear_session(self, session_id: str) -> None:
        """Remove all keys prefixed with a given session ID."""
        prefix = f"session:{session_id}:"
        with self._lock:
            to_delete = [k for k in self._store if k.startswith(prefix)]
            for k in to_delete:
                del self._store[k]

    def cleanup_expired(self) -> int:
        """Remove all expired entries. Returns number removed."""
        now = time.time()
        removed = 0
        with self._lock:
            expired = [k for k, (_, exp) in self._store.items() if now > exp]
            for k in expired:
                del self._store[k]
                removed += 1
        return removed

# Singleton instance
stm = STMManager()

from qdrant_client import models
from typing import List, Dict, Any, Optional
from .embedding_utils import get_embedding   # assume exists

COLLECTION_NAME = "role_memory"

def ingest_role_memory_entry(
    entry_id: str,          # e.g. f"{callsign}_{task_id}_{timestamp}"
    callsign: str,
    task_id: str,
    task_title: str,
    q1: str,
    q2: str,
    q3: str,
    full_text: str,
    project: Optional[str] = None,
    date: Optional[str] = None,
) -> bool:
    """
    Index one role memory entry into Qdrant.
    Non‑blocking – errors are logged but do not raise.
    """
    try:
        client = get_qdrant_client()   # your existing singleton
        # Build combined text for embedding (prioritise Q1–Q3)
        searchable_text = f"{task_title}\n{q1}\n{q2}\n{q3}"
        vector = get_embedding(searchable_text)

        payload = {
            "callsign": callsign,
            "task_id": task_id,
            "task_title": task_title,
            "q1": q1,
            "q2": q2,
            "q3": q3,
            "full_text": full_text[:5000],  # trim for storage
            "project": project or "",
            "date": date or "",
            "timestamp": date,   # for sorting
        }

        client.upsert(
            collection_name=COLLECTION_NAME,
            points=[
                models.PointStruct(
                    id=entry_id,
                    vector=vector,
                    payload=payload,
                )
            ],
        )
        return True
    except Exception as e:
        logger.debug("[Qdrant] Failed to index role memory: %s", e)
        return False

def query_role_memory(
    callsign: str,
    query_text: str,
    limit: int = 3,
    project: Optional[str] = None,
) -> List[Dict[str, Any]]:
    """
    Semantic search over role memory for a given callsign.
    Returns list of payload dicts, most relevant first.
    """
    try:
        client = get_qdrant_client()
        query_vector = get_embedding(query_text)

        filter_conditions = [models.FieldCondition(
            key="callsign",
            match=models.MatchValue(value=callsign),
        )]
        if project:
            filter_conditions.append(models.FieldCondition(
                key="project",
                match=models.MatchValue(value=project),
            ))

        results = client.search(
            collection_name=COLLECTION_NAME,
            query_vector=query_vector,
            query_filter=models.Filter(must=filter_conditions),
            limit=limit,
            with_payload=True,
        )
        return [hit.payload for hit in results]
    except Exception as e:
        logger.debug("[Qdrant] Query failed: %s", e)
        return []

from src.memory.qdrant_client import query_role_memory
from src.memory.stm_manager import stm

def _retrieve_role_memories(callsign: str, query_context: str, limit: int = 3) -> list[dict]:
    """
    Semantic retrieval for role memory. Falls back to Qdrant query.
    Results are cached in STM for the duration of the session.
    """
    stm_key = f"session:{callsign}:role_memories"
    cached = stm.get(stm_key)
    if cached is not None:
        return cached

    # Query Qdrant
    results = query_role_memory(callsign, query_context, limit=limit)
    if not results:
        # Fallback: return empty (no blind last‑N)
        return []

    # Transform to format expected by the session response
    formatted = []
    for r in results:
        formatted.append({
            "task_id": r.get("task_id"),
            "title": r.get("task_title"),
            "q1": r.get("q1"),
            "q2": r.get("q2"),
            "q3": r.get("q3"),
            "date": r.get("date"),
        })
    # Store in STM with TTL = session lifetime (we'll clear on session close)
    stm.set(stm_key, formatted, ttl_seconds=86400)   # 24h
    return formatted
