def is_nil_bug_report(text: str) -> bool:
    """
    Return True if the text expresses that no bugs were found.
    """
    if not text or not isinstance(text, str):
        return True

    text_lower = text.lower().strip()

    # Explicit no-bug phrases
    no_bug_patterns = [
        r"no bugs",
        r"нет багов",
        r"none found",
        r"0 bugs",
        r"not found",
        r"no issues? found",
        r"no defects",
        r"ни одного бага",
        r"багов не найдено",
        r"no problems? found",
        r"nothing to report",
    ]

    for pattern in no_bug_patterns:
        if re.search(pattern, text_lower):
            return True

    # If the text is very short and consists only of negations like "none", "нет", "no"
    if len(text.split()) <= 2 and text_lower in {"none", "нет", "no", "0"}:
        return True

    return False

Why this works: It covers all the patterns mentioned in the AC (no bugs / нет багов / none found / 0 bugs / not found variants) and also handles degenerate cases like a single‑word "none" or very short negations.

def close_nil_bug_tasks(board: "TaskBoard") -> int:
    """Close all existing [DEBRIEF-BUG] tasks that represent nil‑bug reports."""
    closed_count = 0
    for task in board.tasks.values():
        if not task.get("title", "").startswith("[DEBRIEF-BUG] "):
            continue

        # Check both title and description for nil‑bug patterns
        haystack = f"{task.get('title', '')} {task.get('description', '')}"
        if is_nil_bug_report(haystack):
            task["status"] = "closed"
            task["closed_reason"] = "debrief_spam"
            task["closed_at"] = datetime.utcnow().isoformat()
            closed_count += 1

    if closed_count:
        board._save()          # or board.save()
        logger.info("Closed %d spam debrief‑bug tasks", closed_count)
    return closed_count

Call this function once after deploying the code change (e.g., via a one‑off script or an admin endpoint).
