class LLMModelRegistry:
    def __init__(self):
        self._profiles = {}
        self._load_profiles()  # This is synchronous

    # --- NEW METHOD ---
    def get_profile_sync(self, model_id: str) -> dict:
        """
        Synchronous getter for model profiles. Enables use in non-async contexts.

        Args:
            model_id: The model identifier (e.g., "gpt-4").

        Returns:
            A copy of the model profile dictionary.

        Raises:
            KeyError: If the model_id is not found.
        """
        if model_id not in self._profiles:
            raise KeyError(f"Model '{model_id}' not found in registry")
        # Return a copy to prevent external mutation of the internal cache
        return self._profiles[model_id].copy()

    # --- MODIFIED METHOD ---
    async def get_profile(self, model_id: str) -> dict:
        """Async getter. Delegates to the synchronous implementation."""
        # Maintains backward compatibility for existing async callers
        return self.get_profile_sync(model_id)
