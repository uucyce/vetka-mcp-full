# src/llm/model_registry.py

import json
from pathlib import Path
from typing import Dict, Any, Optional
import logging

logger = logging.getLogger(__name__)

class LLMModelRegistry:
    """
    Registry for LLM model profiles.
    
    Manages model configurations including:
    - Context window sizes
    - Token limits
    - Pricing information
    - Capabilities (function calling, vision, etc.)
    """
    
    def __init__(self, profiles_path: Optional[Path] = None):
        self._profiles: Dict[str, Dict[str, Any]] = {}
        self._profiles_path = profiles_path or Path("data/model_profiles.json")
        self._load_profiles()
    
    def _load_profiles(self) -> None:
        """Load model profiles from JSON file (synchronous)."""
        if self._profiles_path.exists():
            try:
                with open(self._profiles_path, 'r') as f:
                    data = json.load(f)
                    self._profiles = data.get('models', {})
                logger.info(f"Loaded {len(self._profiles)} model profiles")
            except Exception as e:
                logger.error(f"Failed to load model profiles: {e}")
                self._profiles = self._get_default_profiles()
        else:
            logger.warning(f"Model profiles not found at {self._profiles_path}")
            self._profiles = self._get_default_profiles()
    
    def _get_default_profiles(self) -> Dict[str, Dict[str, Any]]:
        """Fallback default profiles."""
        return {
            "gpt-4": {
                "context_window": 8192,
                "max_tokens": 4096,
                "supports_functions": True,
                "supports_vision": False,
                "input_cost_per_1k": 0.03,
                "output_cost_per_1k": 0.06,
            },
            "gpt-3.5-turbo": {
                "context_window": 4096,
                "max_tokens": 4096,
                "supports_functions": True,
                "supports_vision": False,
                "input_cost_per_1k": 0.001,
                "output_cost_per_1k": 0.002,
            },
            "claude-3-opus": {
                "context_window": 200000,
                "max_tokens": 4096,
                "supports_functions": True,
                "supports_vision": True,
                "input_cost_per_1k": 0.015,
                "output_cost_per_1k": 0.075,
            }
        }
    
    # ============================================================
    # NEW METHOD: Synchronous getter (A1.2)
    # ============================================================
    
    def get_profile_sync(self, model_id: str) -> Dict[str, Any]:
        """
        Synchronous getter for model profiles.
        
        Enables use in non-async contexts like ModelPolicy.
        
        Args:
            model_id: The model identifier (e.g., "gpt-4", "claude-3-opus")
        
        Returns:
            A COPY of the model profile dictionary to prevent external mutation.
        
        Raises:
            KeyError: If the model_id is not found in the registry.
        
        Example:
            >>> registry = LLMModelRegistry()
            >>> profile = registry.get_profile_sync("gpt-4")
            >>> print(profile["context_window"])
            8192
        """
        if model_id not in self._profiles:
            raise KeyError(f"Model '{model_id}' not found in registry. "
                          f"Available models: {list(self._profiles.keys())}")
        
        # Return a copy to prevent external mutation of internal cache
        return self._profiles[model_id].copy()
    
    # ============================================================
    # MODIFIED METHOD: Now delegates to sync version
    # ============================================================
    
    async def get_profile(self, model_id: str) -> Dict[str, Any]:
        """
        Async getter for model profiles.
        
        Delegates to synchronous implementation for consistency.
        Maintains backward compatibility for existing async callers.
        
        Args:
            model_id: The model identifier
        
        Returns:
            A copy of the model profile dictionary.
        
        Example:
            >>> profile = await registry.get_profile("gpt-4")
        """
        # Delegate to sync implementation
        # No await needed - pure delegation
        return self.get_profile_sync(model_id)
    
    # Additional utility methods
    
    def list_models(self) -> list[str]:
        """List all available model IDs."""
        return list(self._profiles.keys())
    
    def model_exists(self, model_id: str) -> bool:
        """Check if a model exists in the registry."""
        return model_id in self._profiles
    
    def get_profile_or_default(self, model_id: str) -> Optional[Dict[str, Any]]:
        """Get profile or return None if not found (no exception)."""
        try:
            return self.get_profile_sync(model_id)
        except KeyError:
            return None

from typing import Optional, Dict, Any
from .model_registry import LLMModelRegistry

class ModelPolicy:
    """
    Policy for model selection and capability checking.
    
    Now uses synchronous registry access per A1.2.
    """
    
    def __init__(self, registry: LLMModelRegistry):
        self.registry = registry
        self._cache: Dict[str, Dict[str, Any]] = {}
    
    def get_model_capabilities(self, model_id: str) -> Dict[str, Any]:
        """
        Get model capabilities synchronously.
        
        This is the primary consumer of get_profile_sync().
        Previously required async/await, now works in sync contexts.
        """
        # Check cache first
        if model_id in self._cache:
            return self._cache[model_id].copy()
        
        # Synchronous registry access - NO AWAIT NEEDED
        try:
            profile = self.registry.get_profile_sync(model_id)
            capabilities = {
                "context_window": profile.get("context_window", 4096),
                "max_tokens": profile.get("max_tokens", 4096),
                "supports_functions": profile.get("supports_functions", False),
                "supports_vision": profile.get("supports_vision", False),
                "cost_per_1k_input": profile.get("input_cost_per_1k", 0.0),
                "cost_per_1k_output": profile.get("output_cost_per_1k", 0.0),
            }
            self._cache[model_id] = capabilities
            return capabilities.copy()
        except KeyError as e:
            raise ValueError(f"Cannot create policy for unknown model: {model_id}") from e
    
    def can_use_functions(self, model_id: str) -> bool:
        """Check if model supports function calling."""
        caps = self.get_model_capabilities(model_id)
        return caps.get("supports_functions", False)
    
    def can_use_vision(self, model_id: str) -> bool:
        """Check if model supports vision."""
        caps = self.get_model_capabilities(model_id)
        return caps.get("supports_vision", False)
    
    def get_context_window(self, model_id: str) -> int:
        """Get model's context window size."""
        caps = self.get_model_capabilities(model_id)
        return caps.get("context_window", 4096)

import pytest
from src.llm.model_registry import LLMModelRegistry

def test_get_profile_sync_returns_copy():
    registry = LLMModelRegistry()
    profile = registry.get_profile_sync("gpt-4")
    original_context = profile["context_window"]
    
    # Modify the returned dict
    profile["context_window"] = 99999
    
    # Registry should be unchanged
    new_profile = registry.get_profile_sync("gpt-4")
    assert new_profile["context_window"] == original_context
