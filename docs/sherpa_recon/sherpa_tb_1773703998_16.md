class LLMModelRegistry:
    def __init__(self):
        self._profiles = {}
        self._load_profiles()

    def get_profile_sync(self, model_id: str) -> Dict:
        """
        Synchronous getter for model profiles.
        
        Args:
            model_id: Identifier of the model (e.g., "gpt-4", "claude-3")
            
        Returns:
            Dict containing model profile with keys like:
            - context_window: int
            - capabilities: List[str]
            - cost_per_token: float
            - provider: str
            
        Raises:
            KeyError: If model_id not found in registry
        """
        if model_id not in self._profiles:
            raise KeyError(f"Model '{model_id}' not found in registry")
        return self._profiles[model_id].copy()  # Return copy to prevent mutation

    async def get_profile(self, model_id: str) -> Dict:
        """Async getter maintains backward compatibility."""
        return self.get_profile_sync(model_id)

    async def list_models(self) -> List[str]:
        return list(self._profiles.keys())

    def get_model_capabilities(self, model_id: str) -> Dict:

        Synchronous capability check for routing decisions.
        Called from hot paths where async is undesirable.
        """
        try:
            profile = self.registry.get_profile_sync(model_id)
            return {
                "context_window": profile.get("context_window", 4096),
                "supports_streaming": "streaming" in profile.get("capabilities", []),
                "supports_functions": "functions" in profile.get("capabilities", []),
                "provider": profile.get("provider", "unknown"),
                "cost_per_1k_tokens": profile.get("cost_per_1k_tokens", {}),
