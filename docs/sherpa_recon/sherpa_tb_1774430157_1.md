from pathlib import Path
import subprocess

def get_main_repo_root() -> Path:
    """Return absolute path to main repo root, bypassing worktrees."""
    candidate = Path(__file__).resolve().parent.parent.parent
    parts = candidate.parts
    try:
        wt_idx = parts.index(".claude")
        return Path(*parts[:wt_idx])
    except ValueError:
        return candidate

def git_grep_across_branches(pattern: str, branches: list[str], file_patterns: list[str]) -> dict[str, list[str]]:
    """Run git grep on multiple branches, return {branch: [matching lines]}."""
    repo_root = get_main_repo_root()
    results = {}
    for branch in branches:
        cmd = ["git", "grep", "-n", pattern] + [f"--:{fp}" for fp in file_patterns] + [branch]
        try:
            out = subprocess.check_output(cmd, cwd=repo_root, text=True, stderr=subprocess.DEVNULL)
            results[branch] = out.strip().split("\n")
        except subprocess.CalledProcessError:
            results[branch] = []
    return results
