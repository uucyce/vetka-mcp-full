#!/usr/bin/env python3
"""
Task Board Export System for VETKA
Converts SQLite task_board.db to git-readable formats for external collaboration.

Task: 199.SHARED - Shared task board export
"""

import sqlite3
import json
import argparse
import sys
from pathlib import Path
from datetime import datetime
from dataclasses import dataclass, asdict
from typing import List, Optional, Dict, Any
from enum import Enum


class Priority(Enum):
    P0 = "P0"  # Critical
    P1 = "P1"  # High
    P2 = "P2"  # Medium
    P3 = "P3"  # Low
    P4 = "P4"  # Backlog


class Status(Enum):
    BACKLOG = "backlog"
    READY = "ready"
    IN_PROGRESS = "in_progress"
    REVIEW = "review"
    DONE = "done"
    BLOCKED = "blocked"


@dataclass
class Task:
    id: str
    title: str
    description: str
    priority: Priority
    status: Status
    project: str  # CUT, PARALLAX, VETKA, SHARED
    assignee: Optional[str]
    created_at: str
    updated_at: str
    labels: List[str]
    parent_id: Optional[str] = None
    estimated_hours: Optional[float] = None
    claimed_by: Optional[str] = None  # For external contributors
    
    def to_dict(self) -> Dict[str, Any]:
        return {
            **asdict(self),
            'priority': self.priority.value,
            'status': self.status.value
        }


class TaskBoardExporter:
    """
    Export task board to multiple formats for external collaboration.
    """
    
    def __init__(self, db_path: Path):
        self.db_path = db_path
        self.conn = sqlite3.connect(db_path)
        self.conn.row_factory = sqlite3.Row
    
    def _fetch_tasks(self, project_filter: Optional[str] = None) -> List[Task]:
        """Fetch tasks from SQLite with optional project filter."""
        query = """
            SELECT t.*, GROUP_CONCAT(l.name) as labels
            FROM tasks t
            LEFT JOIN task_labels tl ON t.id = tl.task_id
            LEFT JOIN labels l ON tl.label_id = l.id
            WHERE 1=1
        """
        params = []
        
        if project_filter:
            query += " AND t.project = ?"
            params.append(project_filter)
        
        query += " GROUP BY t.id ORDER BY t.priority, t.created_at"
        
        cursor = self.conn.execute(query, params)
        rows = cursor.fetchall()
        
        tasks = []
        for row in rows:
            tasks.append(Task(
                id=row['id'],
                title=row['title'],
                description=row['description'],
                priority=Priority(row['priority']),
                status=Status(row['status']),
                project=row['project'],
                assignee=row['assignee'],
                created_at=row['created_at'],
                updated_at=row['updated_at'],
                labels=row['labels'].split(',') if row['labels'] else [],
                parent_id=row.get('parent_id'),
                estimated_hours=row.get('estimated_hours'),
                claimed_by=row.get('claimed_by')
            ))
        
        return tasks
    
    def export_markdown(self, output_path: Path, project: Optional[str] = None) -> None:
        """
        Export tasks to Markdown format for human readability.
        
        Generates a structured markdown file with:
        - Priority groupings
        - Status columns
        - Claim instructions for external contributors
        """
        tasks = self._fetch_tasks(project)
        
        # Group by priority
        by_priority: Dict[Priority, List[Task]] = {p: [] for p in Priority}
        for task in tasks:
            by_priority[task.priority].append(task)
        
        lines = [
            "# VETKA Task Board",
            "",
            f"> Auto-generated from task_board.db | {datetime.now().isoformat()}",
            "> ",
            "> **For External Contributors:** See [How to Claim a Task](#claiming-tasks) below.",
            "",
            "## 📊 Board Overview",
            "",
            f"- **Total Tasks:** {len(tasks)}",
            f"- **Ready for Pickup:** {len([t for t in tasks if t.status == Status.READY])}",
            f"- **In Progress:** {len([t for t in tasks if t.status == Status.IN_PROGRESS])}",
            "",
        ]
        
        # Add project-specific header if filtering
        if project:
            lines.insert(0, f"# {project} Task Board")
            lines.insert(1, "")
        
        # Priority sections
        priority_emojis = {
            Priority.P0: "🔴",
            Priority.P1: "🟠", 
            Priority.P2: "🟡",
            Priority.P3: "🟢",
            Priority.P4: "⚪"
        }
        
        for priority in [Priority.P0, Priority.P1, Priority.P2, Priority.P3, Priority.P4]:
            priority_tasks = by_priority[priority]
            if not priority_tasks:
                continue
            
            emoji = priority_emojis[priority]
            lines.extend([
                f"## {emoji} {priority.value} Priority",
                "",
                "| ID | Task | Status | Assignee | Claim |",
                "|----|------|--------|----------|-------|"
            ])
            
            for task in priority_tasks:
                status_emoji = self._status_emoji(task.status)
                claim_link = f"[Claim](docs/CONTRIBUTING.md#claiming-tasks)" if task.status == Status.READY else "-"
                assignee = task.assignee or task.claimed_by or "Unassigned"
                
                lines.append(
                    f"| `{task.id}` | **{task.title}**<br>{task.description[:60]}... "
                    f"| {status_emoji} {task.status.value} | {assignee} | {claim_link} |"
                )
            
            lines.append("")
        
        # Contributing guide section
        lines.extend([
            "---",
            "",
            "## 🎯 Claiming Tasks",
            "",
            "### For External Contributors",
            "",
            "1. **Find a READY task** in the table above (status = 🟢 ready)",
            "2. **Open an issue** titled `Claim: {task-id}` describing your approach",
            "3. **Wait for assignment** - maintainers will assign the task to you",
            "4. **Fork and create a worktree**:",
            "   ```bash",
            "   # Main repo setup",
            "   git clone https://github.com/vetka-org/vetka.git",
            "   cd vetka",
            "   ",
            "   # Create worktree for your task",
            "   python scripts/create_worktree.py --task {task-id}",
            "   cd ../vetka-{task-id}",
            "   ```",
            "5. **Submit PR** when complete - link to the task ID in description",
            "",
            "### Task Status Legend",
            "",
            "- 🔴 **P0**: Critical - system down or data loss risk",
            "- 🟠 **P1**: High - major feature or blocking next release", 
            "- 🟡 **P2**: Medium - important but not blocking",
            "- 🟢 **P3**: Low - nice to have",
            "- ⚪ **P4**: Backlog - future consideration",
            "",
            "- ⬜ **backlog**: Not yet ready for work",
            "- 🟢 **ready**: Ready for pickup by contributor",
            "- 🟡 **in_progress**: Actively being worked",
            "- 🔵 **review**: PR submitted, under review",
            "- ✅ **done**: Completed and merged",
            "- 🚫 **blocked**: Blocked by dependency",
        ])
        
        output_path.write_text('\n'.join(lines), encoding='utf-8')
        print(f"Exported {len(tasks)} tasks to {output_path}")
    
    def export_json(self, output_path: Path, include_hierarchy: bool = True) -> None:
        """
        Export tasks to JSON format for API consumption.
        
        Includes project hierarchy: main board → sub-boards (CUT/PARALLAX/VETKA)
        """
        if include_hierarchy:
            data = self._build_hierarchy()
        else:
            tasks = self._fetch_tasks()
            data = {
                "generated_at": datetime.now().isoformat(),
                "tasks": [t.to_dict() for t in tasks]
            }
        
        output_path.write_text(
            json.dumps(data, indent=2, default=str), 
            encoding='utf-8'
        )
        print(f"Exported to {output_path}")
    
    def _build_hierarchy(self) -> Dict[str, Any]:
        """Build project hierarchy: main → sub-projects."""
        projects = ["CUT", "PARALLAX", "VETKA", "SHARED"]
        
        hierarchy = {
            "generated_at": datetime.now().isoformat(),
            "main_board": {
                "name": "VETKA Master Board",
                "description": "Cross-project coordination and P0-P1 critical tasks",
                "tasks": []
            },
            "sub_boards": {}
        }
        
        for project in projects:
            tasks = self._fetch_tasks(project)
            board_data = {
                "name": f"{project} Board",
                "description": f"{project}-specific tasks and implementation details",
                "tasks": [t.to_dict() for t in tasks],
                "stats": {
                    "total": len(tasks),
                    "by_status": {},
                    "by_priority": {}
                }
            }
            
            # Calculate stats
            for task in tasks:
                status = task.status.value
                priority = task.priority.value
                board_data["stats"]["by_status"][status] = \
                    board_data["stats"]["by_status"].get(status, 0) + 1
                board_data["stats"]["by_priority"][priority] = \
                    board_data["stats"]["by_priority"].get(priority, 0) + 1
            
            if project == "SHARED":
                hierarchy["main_board"]["tasks"].extend([t.to_dict() for t in tasks])
            else:
                hierarchy["sub_boards"][project] = board_data
        
        return hierarchy
    
    def export_github_issues(self, output_dir: Path) -> None:
        """
        Generate GitHub issue templates for READY tasks.
        
        Creates structured issue files that can be imported to GitHub.
        """
        output_dir.mkdir(parents=True, exist_ok=True)
        
        ready_tasks = [
            t for t in self._fetch_tasks() 
            if t.status == Status.READY
        ]
        
        for task in ready_tasks:
            issue_path = output_dir / f"issue_{task.id}.md"
            
            content = f"""---
name: Task Claim
about: Claim task {task.id}
title: 'Claim: {task.id} - {task.title}'
labels: ['claim', '{task.project.lower()}', '{task.priority.value.lower()}']
assignees: ''
---

## Task Claim

**Task ID:** {task.id}
**Project:** {task.project}
**Priority:** {task.priority.value}
**Estimated Hours:** {task.estimated_hours or 'TBD'}

### Description
{task.description}

### Acceptance Criteria
- [ ] Task completed according to description
- [ ] Tests pass (if applicable)
- [ ] Documentation updated (if applicable)

### Claim Process
1. Comment below with your approach/plan
2. Wait for maintainer assignment
3. Create worktree: `python scripts/create_worktree.py --task {task.id}`
4. Submit PR linking this issue

### Labels
{', '.join(task.labels)}

---
*This issue was auto-generated from task_board.db. Do not edit manually.*
"""
            issue_path.write_text(content, encoding='utf-8')
        
        print(f"Generated {len(ready_tasks)} GitHub issue templates in {output_dir}")
    
    def _status_emoji(self, status: Status) -> str:
        """Get emoji for status."""
        return {
            Status.BACKLOG: "⬜",
            Status.READY: "🟢",
            Status.IN_PROGRESS: "🟡",
            Status.REVIEW: "🔵",
            Status.DONE: "✅",
            Status.BLOCKED: "🚫"
        }.get(status, "⬜")
    
    def close(self):
        self.conn.close()


def main():
    parser = argparse.ArgumentParser(
        description="Export VETKA task board to git-readable formats"
    )
    parser.add_argument(
        "--db", 
        default="task_board.db",
        help="Path to task_board.db"
    )
    parser.add_argument(
        "--format",
        choices=["markdown", "json", "github", "all"],
        default="all",
        help="Export format"
    )
    parser.add_argument(
        "--output",
        default="TASKS.md",
        help="Output file (for markdown/json) or directory (for github)"
    )
    parser.add_argument(
        "--project",
        choices=["CUT", "PARALLAX", "VETKA", "SHARED"],
        help="Filter to specific project (for markdown export)"
    )
    
    args = parser.parse_args()
    
    db_path = Path(args.db)
    if not db_path.exists():
        print(f"Error: Database not found at {db_path}")
        sys.exit(1)
    
    exporter = TaskBoardExporter(db_path)
    
    try:
        if args.format in ("markdown", "all"):
            output = Path(args.output if args.format == "markdown" else "TASKS.md")
            exporter.export_markdown(output, args.project)
        
        if args.format in ("json", "all"):
            output = Path("tasks.json")
            exporter.export_json(output, include_hierarchy=True)
        
        if args.format in ("github", "all"):
            output_dir = Path(".github/ready_tasks")
            exporter.export_github_issues(output_dir)
    
    finally:
        exporter.close()


if __name__ == "__main__":
    main()

            exported['hierarchy_json'] = json_path

            github_dir = self.repo_root / ".github" / "ready_tasks"
            exporter.export_github_issues(github_dir)
            exported['github_templates'] = github_dir

                worktree_path = self.repo_root.parent / f"vetka-{project}"
                if worktree_path.exists():
                    sub_board_path = worktree_path / "TASKS.md"
                    exporter.export_markdown(sub_board_path, project=project)
                    exported[f'{project.lower()}_md'] = sub_board_path
            
        finally:
            exporter.close()
        
        return exported
    
    def claim_task(self, task_id: str, contributor: str) -> bool:
        """
        External contributor claims a task.
        Updates task status and creates tracking issue.
        """
        task = self.get_task(task_id)
        if not task or task.status != Status.READY:
            return False
        
        # Update task
        self.update_task(task_id, {
            'claimed_by': contributor,
            'status': Status.IN_PROGRESS,
            'updated_at': datetime.now().isoformat()
        })
        
        # Trigger export to update TASKS.md
        self.export_for_collaboration()
        
        return True

# Pre-commit hook to sync task_board.db → TASKS.md
# Install: ln -s ../../scripts/sync_task_board.sh .git/hooks/pre-commit

set -e

echo "🔄 Syncing task board to git-readable format..."

# Export to markdown
python -m src.tools.task_board_export --format=all

# Stage the exported files
git add TASKS.md tasks.json .github/ready_tasks/ 2>/dev/null || true

# Check for worktrees and update sub-boards
for project in CUT PARALLAX VETKA; do
    worktree="../vetka-${project}"
    if [ -d "$worktree" ]; then

        python -m src.tools.task_board_export --format=markdown --project=$project --output="${worktree}/TASKS.md"
        git add "${worktree}/TASKS.md" 2>/dev/null || true
    fi
done

echo "✅ Task board sync complete"
