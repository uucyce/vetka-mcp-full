# VETKA Memory Stack

Vector memory management and semantic search for the VETKA video NLE system.

## External Dependencies

### Required Infrastructure

| Component | Required For | Type | Install |
|-----------|-------------|------|---------|

#!/bin/bash
# VETKA Memory Stack Dependencies Setup

echo "🔧 Setting up VETKA Memory Stack dependencies..."

# Check Docker
if ! command -v docker &> /dev/null; then
    echo "❌ Docker not found. Please install Docker first."
    exit 1
fi

# Start Qdrant
if ! docker ps | grep -q vetka-qdrant; then
    echo "🚀 Starting Qdrant..."
    docker run -d \
        --name vetka-qdrant \
        -p 6333:6333 \
        -p 6334:6334 \
        -v qdrant_storage:/qdrant/storage \
        qdrant/qdrant
    
    echo "⏳ Waiting for Qdrant to be ready..."
    sleep 5
    
    # Health check
    until curl -s http://localhost:6333/health | grep -q "ok"; do
        echo "   Waiting..."
        sleep 2
    done
    
    echo "✅ Qdrant is ready!"
else
    echo "✅ Qdrant already running"
fi

echo ""
echo "📊 Status:"
echo "  Qdrant REST API: http://localhost:6333"
echo "  Qdrant gRPC:     http://localhost:6334"
echo ""
echo "🎯 Next steps:"
echo "  1. Set VETKA_QDRANT_URL=http://localhost:6333"
echo "  2. Install Python deps: pip install -r requirements.txt"
echo "  3. Run memory stack initialization"
