<div className="workflow-card extract-card davinci-controls">
  <h3>Extract</h3>
  <div className="control-group">
    <label>Isolation</label>
    <input
      type="range"
      min="0"
      max="100"
      value={isolationThreshold}
      onChange={(e) => {
        const val = Number(e.target.value);
        setIsolationThreshold(val);
        // Adjust focus/motion or matte sensitivity
        window.vetkaParallaxLab?.setFocus(0.5, 0.5, val / 100, 0.2);
      }}
    />
    <span className="value">{isolationThreshold}%</span>
  </div>

  <div className="control-group">
    <label>Preview Split</label>
    <div className="toggle-switch">
      <button
        className={splitMode === 'off' ? 'active' : ''}
        onClick={() => setSplitMode('off')}
      >
        Off
      </button>
      <button
        className={splitMode === 'matte' ? 'active' : ''}
        onClick={() => {
          setSplitMode('matte');
          window.vetkaParallaxLab?.toggleMatteOverlay();
        }}
      >
        Matte
      </button>
      <button
        className={splitMode === 'sidebyside' ? 'active' : ''}
        onClick={() => setSplitMode('sidebyside')}
      >
        Split
      </button>
    </div>
  </div>

  <div className="control-group">
    <label>Cleanup</label>
    <button
      className="cleanup-button"
      onClick={() => {
        window.vetkaParallaxLab?.applyAiAssistSuggestion();
        window.vetkaParallaxLab?.clearManualHints();
      }}
    >
      ✨ Auto Refine
    </button>
  </div>
