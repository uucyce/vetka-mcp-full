import hashlib
import json
from datetime import datetime
from pathlib import Path

def record_source_provenance(export_root: Path) -> dict:
    """Content-addressed provenance per PARALLAX_HOVER_POLITSIA_ARTIFACT_PROVENANCE_RECON."""
    manifest = {
        "version": "1.0",
        "recorded_at": datetime.now().isoformat(),
        "historical_reference": "camera_contract_fresh_20260327",
        "source_files": {}
    }
    
    # Critical JSON metadata files
    for fname in ["plate_layout.json", "plate_export_manifest.json", 
                  "plate_stack.json", "plate_export_job_state.json"]:
        fpath = export_root / fname
        if fpath.exists():
            content = fpath.read_bytes()
            manifest["source_files"][fname] = {
                "sha256": hashlib.sha256(content).hexdigest(),
                "size_bytes": len(content),
                "mtime_iso": datetime.fromtimestamp(fpath.stat().st_mtime).isoformat()
            }
    
    # Plate assets
    for png in sorted(export_root.glob("*.png")):
        manifest["source_files"][png.name] = {
            "sha256": hashlib.sha256(png.read_bytes()).hexdigest()
        }
    
    return manifest
