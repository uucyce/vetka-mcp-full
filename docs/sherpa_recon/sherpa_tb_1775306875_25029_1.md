import ast
import subprocess
from dataclasses import dataclass
from pathlib import Path
from typing import Dict, List, Optional, Set

# Reuse hallucination detection from sherpa
from sherpa import detect_hallucinated_paths


@dataclass
class VerifyResult:
    accepted: bool
    rollback_to: Optional[str]  # Status to rollback to on failure
    reason: str
    confidence: float  # 0.0-1.0


class TaskVerifier:
    """
    Quality gate middleware for TaskBoard.
    Validates every status transition and can rollback invalid tasks.
    """

    TRANSITION_CHECKS = {
        ("pending", "scout_recon"): ["check_scout_markers_valid"],
        ("scout_recon", "recon_done"): ["check_hallucination", "check_term_overlap", "check_min_length"],
        ("recon_done", "claimed"): [],
        ("claimed", "done_worktree"): ["check_files_in_allowed_paths", "check_syntax_valid"],
        ("done_worktree", "verified"): [],
        ("done_worktree", "needs_fix"): [],
    }

    def __init__(self, project_root: Path):
        """
        Args:
            project_root: Root directory of the project (where allowed_paths are resolved)
        """
        self.project_root = Path(project_root).resolve()

    def check_transition(self, task: Dict, old_status: str, new_status: str) -> VerifyResult:
        """
        Validate a status transition for a given task.

        Args:
            task: Task dictionary (must contain all required fields for the checks)
            old_status: Current status before transition
            new_status: Requested new status

        Returns:
            VerifyResult with acceptance decision and rollback info
        """
        checks = self.TRANSITION_CHECKS.get((old_status, new_status), [])
        for check_name in checks:
            check_fn = getattr(self, check_name)
            result = check_fn(task)
            if not result.accepted:
                return VerifyResult(
                    accepted=False,
                    rollback_to=old_status,
                    reason=f"{check_name}: {result.reason}",
                    confidence=result.confidence,
                )
        return VerifyResult(accepted=True, rollback_to=None, reason="ok", confidence=1.0)

    def check_scout_markers_valid(self, task: Dict) -> VerifyResult:
        """Verify scout markers point to real files and valid line ranges."""
        markers = task.get("scout_context", [])
        if not markers:
            return VerifyResult(False, None, "no scout markers", 0.0)

        for marker in markers:
            file_path = self.project_root / marker["file"]
            if not file_path.is_file():
                return VerifyResult(False, None, f"file not found: {marker['file']}", 0.0)

            try:
                lines = file_path.read_text(encoding="utf-8").splitlines()
                line_count = len(lines)
            except Exception as e:
                return VerifyResult(False, None, f"cannot read {marker['file']}: {e}", 0.0)

            start = marker.get("start_line", 1)
            end = marker.get("end_line", start)
            if start < 1 or end > line_count or start > end:
                return VerifyResult(
                    False, None,
                    f"invalid lines {start}-{end} in {marker['file']} (total {line_count})",
                    0.0
                )

        return VerifyResult(True, None, "ok", 1.0)

    def check_hallucination(self, task: Dict) -> VerifyResult:
        """
        Check recon response for hallucinated file paths.
        Uses detect_hallucinated_paths from sherpa.py.
        Accepts if hallucination score > 0.5 (higher = more realistic).
        """
        recon_docs = task.get("recon_docs", [])
        if not recon_docs:
            return VerifyResult(False, None, "no recon_docs to verify", 0.0)

        allowed_paths = task.get("allowed_paths", [])
        # Concatenate all recon markdown files
        full_response = []
        for doc_path in recon_docs:
            resolved = self.project_root / doc_path
            if not resolved.is_file():
                return VerifyResult(False, None, f"recon doc missing: {doc_path}", 0.0)
            try:
                full_response.append(resolved.read_text(encoding="utf-8"))
            except Exception as e:
                return VerifyResult(False, None, f"cannot read {doc_path}: {e}", 0.0)

        response_text = "\n\n".join(full_response)
        try:
            result = detect_hallucinated_paths(response_text, allowed_paths)
            score = result.get("score", 0.0)
            if score > 0.5:
                return VerifyResult(True, None, f"hallucination score {score:.2f} > 0.5", score)
            else:
                return VerifyResult(False, None, f"hallucination score {score:.2f} <= 0.5", score)
        except Exception as e:
            return VerifyResult(False, None, f"hallucination check failed: {e}", 0.0)

    def check_term_overlap(self, task: Dict) -> VerifyResult:
        """
        Verify that >15% of the task's key terms appear in the recon response.
        Terms are extracted from title and description.
        """
        title = task.get("title", "")
        description = task.get("description", "")
        text = f"{title} {description}".lower()
        # Simple tokenization (split on non-alphanumeric)
        import re
        tokens = re.findall(r"[a-z0-9_]+", text)
        stopwords = {"the", "a", "an", "and", "or", "but", "to", "of", "for", "in", "on", "at", "with", "by", "is", "are", "was", "were", "be", "been", "being", "have", "has", "had", "having", "do", "does", "did", "doing", "will", "would", "could", "should", "this", "that", "these", "those", "it", "they", "we", "you", "he", "she", "it", "them"}
        terms = {t for t in tokens if t not in stopwords and len(t) > 2}
        if not terms:
            # No meaningful terms - skip check (treat as passed)
            return VerifyResult(True, None, "no meaningful terms to check", 1.0)

        recon_docs = task.get("recon_docs", [])
        recon_text = ""
        for doc_path in recon_docs:
            resolved = self.project_root / doc_path
            if resolved.is_file():
                try:
                    recon_text += resolved.read_text(encoding="utf-8").lower()
                except Exception:
                    continue
        if not recon_text:
            return VerifyResult(False, None, "no recon content to verify", 0.0)

        matched = sum(1 for term in terms if term in recon_text)
        overlap = matched / len(terms)
        if overlap > 0.15:
            return VerifyResult(True, None, f"term overlap {overlap:.2%} > 15%", overlap)
        else:
            return VerifyResult(False, None, f"term overlap {overlap:.2%} <= 15%", overlap)

    def check_min_length(self, task: Dict) -> VerifyResult:
        """Verify recon response total length > 500 characters."""
        recon_docs = task.get("recon_docs", [])
        total_chars = 0
        for doc_path in recon_docs:
            resolved = self.project_root / doc_path
            if resolved.is_file():
                try:
                    total_chars += len(resolved.read_text(encoding="utf-8"))
                except Exception:
                    continue
        if total_chars > 500:
            return VerifyResult(True, None, f"recon length {total_chars} > 500", min(1.0, total_chars / 500))
        else:
            return VerifyResult(False, None, f"recon length {total_chars} <= 500", total_chars / 500 if total_chars > 0 else 0.0)

    def check_files_in_allowed_paths(self, task: Dict) -> VerifyResult:
        """Verify that all changed files (from agent work) are within allowed_paths."""
        changed_files = task.get("changed_files", [])
        if not changed_files:
            # No changes means nothing to validate
            return VerifyResult(True, None, "no changed files to check", 1.0)

        allowed_paths = task.get("allowed_paths", [])
        if not allowed_paths:
            return VerifyResult(False, None, "no allowed_paths defined", 0.0)

        # Resolve allowed directories relative to project root
        allowed_dirs = []
        for path in allowed_paths:
            resolved = (self.project_root / path).resolve()
            if resolved.is_dir():
                allowed_dirs.append(resolved)
            else:
                # Try parent directory if path is a file
                parent = resolved.parent
                if parent.exists():
                    allowed_dirs.append(parent)

        if not allowed_dirs:
            return VerifyResult(False, None, "none of allowed_paths resolve to existing directories", 0.0)

        for file_path in changed_files:
            abs_path = (self.project_root / file_path).resolve()
            if not any(abs_path.is_relative_to(allowed_dir) for allowed_dir in allowed_dirs):
                return VerifyResult(
                    False, None,
                    f"file {file_path} is outside all allowed paths",
                    0.0
                )
        return VerifyResult(True, None, "all changed files within allowed paths", 1.0)

    def check_syntax_valid(self, task: Dict) -> VerifyResult:
        """
        Basic syntax validation:
        - .py files: use ast.parse
        - .ts files: simple check (balanced braces, etc.) or try tsc --noEmit if available
        """
        changed_files = task.get("changed_files", [])
        if not changed_files:
            return VerifyResult(True, None, "no files to validate", 1.0)

        for file_path in changed_files:
            abs_path = self.project_root / file_path
            if not abs_path.is_file():
                continue  # file might be deleted, skip

            ext = abs_path.suffix.lower()
            if ext == ".py":
                try:
                    ast.parse(abs_path.read_text(encoding="utf-8"))
                except SyntaxError as e:
                    return VerifyResult(False, None, f"syntax error in {file_path}: {e}", 0.0)
            elif ext == ".ts":
                # Basic TypeScript check: try to run tsc --noEmit if available
                try:
                    result = subprocess.run(
                        ["tsc", "--noEmit", "--skipLibCheck", str(abs_path)],
                        capture_output=True,
                        text=True,
                        timeout=5,
                        cwd=self.project_root
                    )
                    if result.returncode != 0:
                        # Extract first line of error
                        error_line = result.stderr.splitlines()[0] if result.stderr else "unknown error"
                        return VerifyResult(False, None, f"TypeScript error in {file_path}: {error_line}", 0.0)
                except FileNotFoundError:
                    # tsc not installed – fallback to basic brace/parentheses check
                    content = abs_path.read_text(encoding="utf-8")
                    # Very basic: check for balanced braces, parentheses, brackets
                    stack = []
                    pairs = {"{": "}", "(": ")", "[": "]"}
                    for ch in content:
                        if ch in pairs:
                            stack.append(ch)
                        elif ch in pairs.values():
                            if not stack or pairs[stack.pop()] != ch:
                                return VerifyResult(False, None, f"unbalanced brace/parenthesis in {file_path}", 0.0)
                    if stack:
                        return VerifyResult(False, None, f"unclosed brace/parenthesis in {file_path}", 0.0)
                except subprocess.TimeoutExpired:
                    return VerifyResult(False, None, f"tsc timeout for {file_path}", 0.0)
                except Exception as e:
                    return VerifyResult(False, None, f"cannot check {file_path}: {e}", 0.0)
            # Other extensions are not checked
        return VerifyResult(True, None, "all files passed syntax checks", 1.0)
