from collections import defaultdict
import asyncio

class RateLimiter:
    def __init__(self, rpm: int, accounts: int):
        self.requests_per_minute = rpm * accounts
        self.last_request = defaultdict(float)
    
    async def wait_if_needed(self, service: str):
        now = asyncio.get_event_loop().time()
        elapsed = now - self.last_request[service]
        if elapsed < 60.0 / self.requests_per_minute:
            await asyncio.sleep(60.0 / self.requests_per_minute - elapsed)
        self.last_request[service] = now

async def wait_for_complete_response(page, timeout=60):
    """Wait for streaming response to finish."""
    start = time.time()
    last_length = 0
    stable_count = 0
    
    while time.time() - start < timeout:
        # Check for "thinking" indicator
        if await page.locator('.thinking, .loading, .streaming').is_visible():
            await asyncio.sleep(0.5)
            continue
        
        # Get current text length
        text = await page.locator('.response-content').inner_text()
        current_length = len(text)
        
        if current_length == last_length:
            stable_count += 1
            if stable_count >= 3:  # 1.5 seconds stable
                return text
        else:
            stable_count = 0
            last_length = current_length
        
        await asyncio.sleep(0.5)
    
    return await page.locator('.response-content').inner_text()

from playwright.async_api import async_playwright
from playwright_stealth import Stealth  # Python port

async def launch_stealth_browser(account_id: str):
    async with Stealth().use_async(async_playwright()) as p:
        browser = await p.chromium.launch(
            headless=False,  # Headed mode reduces detection
            args=[
                '--disable-blink-features=AutomationControlled',
                '--no-sandbox',
            ]
        )
        context = await browser.new_context(
            user_agent='Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36',
            viewport={'width': 1280, 'height': 720},
            locale='en-US',
            timezone_id='America/New_York',
            storage_state=f'sessions/{account_id}.json'
        )
        return context

async def handle_captcha(page, account_id):
