import subprocess
from pathlib import Path

def generate_test_clip(duration: float, bpm: int, output_path: Path):
    """Generate synthetic test video with audio beat markers."""
    # Video: color bars with duration
    # Audio: click track at specified BPM for beat detection validation
    cmd = [
        "ffmpeg",
        "-f", "lavfi", "-i", f"testsrc=duration={duration}:size=1920x1080:rate=30",
        "-f", "lavfi", "-i", f"sine=frequency=1000:duration={duration}",  # beep track
        "-pix_fmt", "yuv420p",
        "-c:v", "libx264", "-preset", "fast",
        "-c:a", "aac",
        "-shortest",
        str(output_path)
    ]
    subprocess.run(cmd, check=True)
