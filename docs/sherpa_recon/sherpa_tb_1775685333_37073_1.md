#!/usr/bin/env python3
"""
Pre-commit hook: Check task board compliance before allowing commit.
MARKER_210.TASK_BOARD_GUARDRAIL - Extended for hotfix discipline (P0/P1)
"""

import os
import sys
import subprocess
import json
import re
from pathlib import Path
from datetime import datetime

REPO_ROOT = Path(__file__).parent.parent.resolve()
TASK_BOARD_SCRIPT = REPO_ROOT / "scripts" / "vetka_task_board.py"
AUDIT_LOG = REPO_ROOT / "logs" / "bypass_audit.log"

def get_current_branch():
    result = subprocess.run(["git", "branch", "--show-current"], capture_output=True, text=True)
    return result.stdout.strip()

def is_merge_commit(commit_hash):
    """Check if a commit is a merge commit (has 2+ parents)."""
    result = subprocess.run(["git", "rev-list", "--parents", "-n", "1", commit_hash],
                            capture_output=True, text=True)
    parents = result.stdout.strip().split()
    return len(parents) > 2  # First is commit, rest are parents

def get_claimed_task(role):
    """Query task board for claimed/running tasks for the role."""
    if not TASK_BOARD_SCRIPT.exists():
        return None
    result = subprocess.run(
        [sys.executable, str(TASK_BOARD_SCRIPT), "action=list", f"assigned_to={role}", "filter_status=claimed,running"],
        capture_output=True, text=True
    )
    if result.returncode != 0:
        return None
    try:
        data = json.loads(result.stdout)
        tasks = data.get("tasks", [])
        if not tasks:
            return None
        # Return the most recent claimed task
        return tasks[0]
    except json.JSONDecodeError:
        return None

def is_hotfix_commit(branch, commit_msg, task):
    """Determine if this is a hotfix (priority 1 task or branch/msg pattern)."""
    if task and task.get("priority") == 1:
        return True
    if branch.startswith("hotfix/") or branch.startswith("fix/"):
        return True
    if re.search(r'\[hotfix\]|hotfix:|P0|P1', commit_msg, re.IGNORECASE):
        return True
    return False

def log_bypass(commit_msg):
    """Log --no-verify usage to audit file."""
    AUDIT_LOG.parent.mkdir(exist_ok=True)
    user = os.environ.get("USER", "unknown")
    branch = get_current_branch()
    timestamp = datetime.now().isoformat()
    with open(AUDIT_LOG, "a") as f:
        f.write(f"[{timestamp}] USER={user} BRANCH={branch} MSG={commit_msg}\n")

#!/usr/bin/env python3
"""
Audit script to detect commits that were made directly to main/master
(bypassing the required merge request flow).
MARKER_210.AUDIT_MAIN_COMMITS
"""

import subprocess
import sys
from datetime import datetime

def get_direct_main_commits(since="1 month ago"):
    """Return list of non-merge commits on main that are not merges from PRs."""
    # Get all commits on main
    cmd = ["git", "log", "main", f"--since={since}", "--pretty=format:%H|%an|%ad|%s", "--date=short"]
    result = subprocess.run(cmd, capture_output=True, text=True)
    if result.returncode != 0:
        print("Error: Not a git repository or main branch missing", file=sys.stderr)
        return []
    
    direct_commits = []
    for line in result.stdout.strip().split("\n"):
        if not line:
            continue
        parts = line.split("|", 3)
        if len(parts) < 4:
            continue
        commit_hash, author, date, subject = parts
        # Check if it's a merge commit (has 2 parents)
        parents = subprocess.run(["git", "rev-list", "--parents", "-n", "1", commit_hash],
                                 capture_output=True, text=True).stdout.strip().split()
        if len(parents) > 2:  # First is commit hash, rest are parents
            continue  # merge commit, allowed
        # Also skip commits that are from a PR merge (GitHub/GitLab typically have "Merge pull request" in subject)
        if subject.startswith("Merge pull request") or subject.startswith("Merge branch"):
            continue
        direct_commits.append((commit_hash, author, date, subject))
    return direct_commits

def main():
    print("=== Direct-to-main Commit Audit ===")
    print(f"Run at {datetime.now().isoformat()}\n")
    commits = get_direct_main_commits()
    if not commits:
        print("✅ No direct commits to main found (only merge commits).")
        return 0
    print(f"❌ Found {len(commits)} non-merge commit(s) directly on main:\n")
    for h, a, d, s in commits:
        print(f"  {h[:8]} | {d} | {a} | {s}")
    print("\nThese commits bypassed the required task board + merge request flow.")
    print("Action required: Revert each and redo via hotfix task + merge request.")
    return 1

if __name__ == "__main__":
    sys.exit(main())
