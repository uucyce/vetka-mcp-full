def test_task_wake_parse_and_get(task_board, sample_task):
    """Имитация wake: из сообщения извлекается task_id → action=get → возвращается задача."""
    # Генерируем уведомление как в предыдущем тесте
    task_board.notify(
        recipient_role="Delta",
        message="Task completed",
        ntype="task_done_worktree",
        source_role="Gamma",
        task_id=sample_task
    )
    notifications = task_board.get_notifications(role="Delta", unread_only=True)
    wake_message = notifications[-1]["message"]

    # Парсим task_id из сообщения (ожидаем формат "TASK-WAKE tb_xxx: ...")
    import re
    match = re.search(r"TASK-WAKE\s+(tb_\S+)", wake_message)
    assert match is not None, f"Не удалось извлечь task_id из сообщения: {wake_message}"
