# src/services/youtube_service.py
import asyncio
import aiohttp
from typing import Optional, Dict, Any

class AsyncYouTubeFetcher:
    """Async YouTube API client – single source of truth."""
    
    def __init__(self, api_key: str):
        self.api_key = api_key
        self._session: Optional[aiohttp.ClientSession] = None

    async def _get_session(self) -> aiohttp.ClientSession:
        if self._session is None or self._session.closed:
            self._session = aiohttp.ClientSession()
        return self._session

    async def fetch_video_info(self, video_id: str) -> Dict[str, Any]:
        """Async fetch – used directly by FastAPI routes."""
        session = await self._get_session()
        params = {"part": "snippet,statistics", "id": video_id, "key": self.api_key}
        async with session.get("https://www.googleapis.com/youtube/v3/videos", params=params) as resp:
            resp.raise_for_status()
            data = await resp.json()
            return data.get("items", [{}])[0]

    async def close(self):
        if self._session and not self._session.closed:
            await self._session.close()

class YouTubeService:
    """Synchronous wrapper for CLI / test contexts."""
    
    def __init__(self, api_key: str):
        self._async_fetcher = AsyncYouTubeFetcher(api_key)

    def fetch_video_info(self, video_id: str) -> Dict[str, Any]:
        """Sync method – runs the async fetcher in a temporary event loop."""
        try:
            loop = asyncio.get_running_loop()
        except RuntimeError:  # no running loop
            return asyncio.run(self._async_fetcher.fetch_video_info(video_id))
        else:
            # Already in async context – create a new loop in a thread
            import concurrent.futures
            with concurrent.futures.ThreadPoolExecutor() as executor:
                future = executor.submit(asyncio.run, self._async_fetcher.fetch_video_info(video_id))
                return future.result()

Note on sync adapter: The above handles both no‑loop (CLI) and existing‑loop (rare for sync code) cases. Simpler: just use asyncio.run() for CLI, since tests can be marked @pytest.mark.asyncio.
