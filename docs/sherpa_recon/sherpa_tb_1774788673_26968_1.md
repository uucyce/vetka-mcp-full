3.2 Promotion handler
tsx
Copy
Download
const promoteDraftToLayer = () => {
  if (!selectedDraftId) return;

  const draft = draftPlates.find(d => d.id === selectedDraftId);
  if (!draft) return;

  const newLayer: AuthoritativeLayer = {
    ...draft,
    id: `auth_${Date.now()}_${Math.random().toString(36).substr(2, 8)}`,
    isAuthoritative: true,
    promotedAt: Date.now(),
  };

  // Remove from drafts
  setDraftPlates(prev => prev.filter(d => d.id !== selectedDraftId));
  // Add to authoritative layers
  setAuthoritativeLayers(prev => [...prev, newLayer]);
  // Select the new layer
  setSelectedAuthoritativeId(newLayer.id);
  setSelectedDraftId(null); // clear draft selection
};
