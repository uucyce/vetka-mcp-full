def _validate_fail_notes(self, notes: str) -> tuple[bool, str]:
    """
    Validate that fail verdict notes contain structured root cause info.
    
    Returns: (is_valid, error_message)
    """
    if not notes:
        return False, "QA FAIL requires structured notes: root_cause + failing_tests (min 50 chars). Example: Root cause: X returns Y instead of Z. Reproduce: pytest tests/test_foo.py::test_bar"
    
    stripped = notes.strip()
    
    # Check minimum length
    if len(stripped) < _QA_FAIL_MIN_LENGTH:
        return False, f"QA FAIL notes too short ({len(stripped)} chars, min {_QA_FAIL_MIN_LENGTH}). Required structure: root_cause (what's broken) + failing_tests (how to reproduce). Example: 'Root cause: API returns 500 on null input. Reproduce: curl -X POST http://localhost/api/test -d \\\"{{}}\\\"'"
    
    # Check for root cause indicators (case-insensitive)
    notes_lower = stripped.lower()
    has_indicator = any(indicator.lower() in notes_lower for indicator in _ROOT_CAUSE_INDICATORS)
    
    if not has_indicator:
        indicators_str = ", ".join(_ROOT_CAUSE_INDICATORS[:8]) + "..."
        return False, f"QA FAIL notes missing root cause indicators. Must contain one of: {indicators_str}. Required: root_cause description + failing_tests commands. Current notes: '{stripped[:100]}...'"
    
    return True, ""
