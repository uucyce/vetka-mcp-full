#!/usr/bin/env python3
"""
generate_claude_md.py — VETKA CLAUDE.md Generator
Phase 203.3: Role Memory section for all agents

Generates CLAUDE.md files with Role Memory integration.
"""

import argparse
import os
import sys
import yaml
from pathlib import Path
from jinja2 import Environment, FileSystemLoader, select_autoescape

# === Configuration ===
PROJECT_ROOT = Path(__file__).parent.parent.parent
REGISTRY_FILE = PROJECT_ROOT / "data/templates/agent_registry.yaml"
TEMPLATE_FILE = PROJECT_ROOT / "data/templates/claude_md_template.j2"
WORKTREE_BASE = PROJECT_ROOT / ".claude/worktrees"
MEMORY_BASE = PROJECT_ROOT / "memory/roles"


def load_registry() -> dict:
    """Load agent registry from YAML."""
    with open(REGISTRY_FILE, 'r') as f:
        return yaml.safe_load(f)


def get_role_config(registry: dict, callsign: str) -> dict | None:
    """Get configuration for a specific role."""
    return registry.get('roles', {}).get(callsign)


def load_role_memory(memory_path: str) -> str | None:
    """Load role memory content if it exists."""
    full_path = PROJECT_ROOT / memory_path
    if full_path.exists():
        return full_path.read_text()
    return None


def generate_claude_md(role_config: dict, registry: dict) -> str:
    """
    Generate CLAUDE.md content using Jinja2 template.
    Includes Role Memory section.
    """
    env = Environment(
        loader=FileSystemLoader(TEMPLATE_FILE.parent),
        autoescape=select_autoescape(['html', 'xml']),
        trim_blocks=True,
        lstrip_blocks=True
    )
    
    template = env.get_template(TEMPLATE_FILE.name)
    
    # Prepare context for template
    context = {
        'role': role_config,
        'registry': registry,
        'role_memory': load_role_memory(role_config.get('memory_path', '')),
        'memory_base': str(MEMORY_BASE.relative_to(PROJECT_ROOT)),
    }
    
    return template.render(**context)


def write_claude_md(callsign: str, content: str) -> Path:
    """Write CLAUDE.md to role's worktree."""
    worktree_name = get_role_config(load_registry(), callsign).get('worktree')
    worktree_path = WORKTREE_BASE / worktree_name
    claude_md_path = worktree_path / "CLAUDE.md"
    
    # Ensure directory exists
    worktree_path.mkdir(parents=True, exist_ok=True)
    
    # Write file
    claude_md_path.write_text(content)
    return claude_md_path


def generate_all() -> list[tuple[str, Path]]:
    """Generate CLAUDE.md for all roles in registry."""
    registry = load_registry()
    results = []
    
    for callsign, config in registry.get('roles', {}).items():
