Core AppleScript Approach
The solution uses osascript to query System Events for the frontmost window's position and size, then passes these coordinates to screencapture -R .
AppleScript to get window bounds:
applescript

tell application "System Events"
    tell (first process where it is frontmost)
        tell front window
            set winPos to position
            set winSize to size
            return (item 1 of winPos) & "," & (item 2 of winPos) & "," & (item 1 of winSize) & "," & (item 2 of winSize)
        end tell
    end tell
end tell
This returns: x,y,width,height format needed for screencapture -R x,y,w,h .

"""
Active Window Screenshot Tool for VETKA MCP.

Captures the frontmost window using osascript bounds + screencapture -R.
Fallback to full screen if window covers >95% of display or osascript fails.

@status: active
@phase: 96
@depends: base_tool
@used_by: mcp_server
"""

import subprocess
import re
import os
from typing import Any, Dict, Optional, Tuple
from .base_tool import BaseMCPTool


class ActiveWindowScreenshotTool(BaseMCPTool):
    """Capture screenshot of the active/frontmost window"""
    
    # Threshold for treating window as full-screen (95% of display)
    FULLSCREEN_THRESHOLD = 0.95
    
    @property
    def name(self) -> str:
        return "vetka_screenshot_active_window"
    
    @property
    def description(self) -> str:
        return (
            "Capture screenshot of the active/frontmost window. "
            "Uses osascript to get window bounds, then screencapture -R. "
            "Falls back to full screen if window covers >95% of display "
            "or if accessibility permissions prevent window detection."
        )
    
    def _get_screen_resolution(self) -> Tuple[int, int]:
        """Get main display resolution using system_profiler"""
        try:
            result = subprocess.run(
                ["system_profiler", "SPDisplaysDataType", "-json"],
                capture_output=True,
                text=True,
                timeout=5
            )
            import json
            data = json.loads(result.stdout)
            displays = data.get("SPDisplaysDataType", [])
            for display in displays:
                if "spdisplays_ndrvs" in display:
                    for monitor in display["spdisplays_ndrvs"]:
                        if "_spdisplays_resolution" in monitor:
                            res_str = monitor["_spdisplays_resolution"]
                            match = re.match(r"(\d+)\s*x\s*(\d+)", res_str)
                            if match:
                                return int(match.group(1)), int(match.group(2))
            # Fallback to common resolution
            return 1920, 1080
        except Exception:
            return 1920, 1080
    
    def _get_active_window_bounds(self) -> Optional[Tuple[int, int, int, int]]:
        """
        Get bounds of frontmost window using osascript.
        Returns (x, y, width, height) or None if failed.
        """
        applescript = '''
        tell application "System Events"
            tell (first process where it is frontmost)
                if (count of windows) > 0 then
                    tell front window
                        set winPos to position
                        set winSize to size
                        return (item 1 of winPos) & "," & (item 2 of winPos) & "," & (item 1 of winSize) & "," & (item 2 of winSize)
                    end tell
                else
                    return "NO_WINDOW"
                end if
            end tell
        end tell
        '''
        
        try:
            result = subprocess.run(
                ["osascript", "-e", applescript],
                capture_output=True,
                text=True,
                timeout=10
            )
            
            if result.returncode != 0:
                self.logger.warning(f"osascript error: {result.stderr}")
                return None
            
            output = result.stdout.strip()
            
            if output == "NO_WINDOW":
                self.logger.info("No window found for frontmost process")
                return None
            
            # Parse x,y,w,h
            parts = output.split(",")
            if len(parts) != 4:
                self.logger.warning(f"Unexpected osascript output format: {output}")
                return None
            
            x, y, w, h = map(int, parts)
            return (x, y, w, h)
            
        except subprocess.TimeoutExpired:
            self.logger.error("osascript timed out")
            return None
        except Exception as e:
            self.logger.error(f"Error getting window bounds: {e}")
            return None
    
    def _is_near_fullscreen(self, x: int, y: int, w: int, h: int) -> bool:
        """Check if window covers >95% of screen"""
        screen_w, screen_h = self._get_screen_resolution()
        screen_area = screen_w * screen_h
        window_area = w * h
        
        if screen_area == 0:
            return False
            
        ratio = window_area / screen_area
        return ratio > self.FULLSCREEN_THRESHOLD
    
    def _capture_region(self, x: int, y: int, w: int, h: int, output_path: str) -> bool:
        """Capture specific region using screencapture -R"""
        try:
            # screencapture -R x,y,w,h
            region_arg = f"{x},{y},{w},{h}"
            result = subprocess.run(
                ["screencapture", "-R", region_arg, output_path],
                capture_output=True,
                text=True,
                timeout=15
            )
            return result.returncode == 0
        except Exception as e:
            self.logger.error(f"screencapture error: {e}")
            return False
    
    def _capture_fullscreen(self, output_path: str) -> bool:
        """Fallback to full screen capture"""
        try:
            result = subprocess.run(
                ["screencapture", output_path],
                capture_output=True,
                text=True,
                timeout=15
            )
            return result.returncode == 0
        except Exception as e:
            self.logger.error(f"Fullscreen capture error: {e}")
            return False
    
    async def execute(self, params: Dict[str, Any]) -> Dict[str, Any]:
        """
        Execute active window screenshot capture.
        
        Parameters:
            output_path: Path to save screenshot (optional, defaults to temp)
            format: Image format - png, jpg, pdf, tiff (default: png)
            
        Returns:
            Dict with success status, file path, and metadata
        """
        output_path = params.get("output_path")
        fmt = params.get("format", "png").lower()
        
        # Generate temp path if not provided
        if not output_path:
            import tempfile
            import uuid
            temp_dir = tempfile.gettempdir()
            output_path = os.path.join(temp_dir, f"vetka_screenshot_{uuid.uuid4().hex}.{fmt}")
        
        # Ensure directory exists
        os.makedirs(os.path.dirname(os.path.abspath(output_path)), exist_ok=True)
        
        # Get active window bounds
        bounds = self._get_active_window_bounds()
        
        if bounds is None:
            self.logger.info("Falling back to fullscreen (no window bounds)")
            success = self._capture_fullscreen(output_path)
            capture_mode = "fullscreen_fallback"
            bounds_info = None
        else:
            x, y, w, h = bounds
            
            # Check if effectively fullscreen
            if self._is_near_fullscreen(x, y, w, h):
                self.logger.info(f"Window covers >95% of screen, using fullscreen capture")
                success = self._capture_fullscreen(output_path)
                capture_mode = "fullscreen_optimized"
            else:
                self.logger.info(f"Capturing region: {x},{y},{w},{h}")
                success = self._capture_region(x, y, w, h, output_path)
                capture_mode = "window_region"
            
            bounds_info = {"x": x, "y": y, "width": w, "height": h}
        
        if not success:
            return {
                "success": False,
                "error": "Screenshot capture failed",
                "capture_mode": "failed"
            }
        
        # Get file info
        file_size = os.path.getsize(output_path) if os.path.exists(output_path) else 0
        
        return {
            "success": True,
            "file_path": output_path,
            "capture_mode": capture_mode,
            "window_bounds": bounds_info,
            "file_size_bytes": file_size,
            "format": fmt
        }

Key Implementation Details
osascript reliability: The script queries System Events which requires Accessibility permissions . If denied, it gracefully falls back to fullscreen.
screencapture -R format: The -R flag expects x,y,width,height coordinates .
Fullscreen detection: Windows covering >95% of screen area trigger fullscreen capture to avoid redundant region calculations.
Error handling: Covers cases where:
No window exists (background apps)
Accessibility permissions denied
osascript timeout
screencapture failure
Output formats: Supports PNG (default), JPG, PDF, TIFF via file extension .
