class CleanPlateValidator:
    """
    Validates clean plate quality before shipping.
    Detects common failure modes:
    - Foreground residue (incomplete mask dilation)
    - Blurry artifacts (over-aggressive inpainting)
    - Color inconsistency (poor boundary blending)
    - Texture repetition (LaMa pattern artifacts)
    """
    
    VALIDATION_CHECKS = [
        "residue_detection",
        "blur_metrics", 
        "color_consistency",
        "texture_naturalness",
        "edge_continuity"
    ]
    
    def validate(self, clean_plate: np.ndarray, 
                 original: np.ndarray, 
                 mask: np.ndarray) -> Tuple[bool, Dict]:
        """
        Returns (is_valid, diagnostics).
        """
        results = {}
        
        # Check 1: Foreground residue detection
        # Compare gradient magnitude in hole region vs surrounding
        results["residue"] = self._check_residue(clean_plate, mask)
        
        # Check 2: Blur detection
        # Laplacian variance should be similar to surrounding texture
        results["blur"] = self._check_blur_metrics(clean_plate, mask)
        
        # Check 3: Color consistency at boundaries
        results["color"] = self._check_color_consistency(
            clean_plate, original, mask
        )
        
        overall_score = np.mean([
            results[check]["score"] for check in self.VALIDATION_CHECKS
        ])
        
        is_valid = overall_score >= self.quality_threshold
        
        return is_valid, results
