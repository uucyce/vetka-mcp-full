import asyncio
import json
import logging
from pathlib import Path
from watchdog.observers import Observer
from watchdog.events import FileSystemEventHandler

from src.orchestration.agent_pipeline import run_pipeline_for_task  # hypothetical

logger = logging.getLogger(__name__)

class ScoutTaskHandler(FileSystemEventHandler):
    def __init__(self, loop: asyncio.AbstractEventLoop):
        self.loop = loop

    def on_created(self, event):
        if not event.is_directory and event.src_path.endswith(".json"):
            logger.info(f"New scout task detected: {event.src_path}")
            # Schedule async pipeline run
            asyncio.run_coroutine_threadsafe(
                self._process_task(event.src_path), self.loop
            )

    async def _process_task(self, task_path: str):
        try:
            with open(task_path, "r") as f:
                task = json.load(f)
            # Call SCOUT pipeline
            result = await run_pipeline_for_task(
                task=task,
                role="scout",
                model="gemma4:e2b",
                headless=True
            )
            logger.info(f"SCOUT result: {result}")
            # Optionally move processed file to .done/
            Path(task_path).rename(task_path + ".done")
        except Exception as e:
            logger.exception(f"Auto‑trigger failed for {task_path}: {e}")

def start_auto_trigger(watch_dir: str = "data/scout_tasks/"):
    """Start watchdog observer for SCOUT auto‑trigger."""
    loop = asyncio.get_event_loop()
    event_handler = ScoutTaskHandler(loop)
    observer = Observer()
    observer.schedule(event_handler, watch_dir, recursive=False)
    observer.start()
    logger.info(f"Auto‑trigger watching {watch_dir}")
    return observer
