import glob
import os
import logging
from pathlib import Path

logger = logging.getLogger(__name__)

def cleanup_screenshot_temps(max_age_seconds: int = 3600) -> dict:
    """
    Delete /tmp/vetka_screenshot_*.jpg files older than max_age_seconds.
    
    Returns:
        dict: {"deleted": [filenames], "errors": [error_strings], "count": int}
    """
    result = {"deleted": [], "errors": [], "count": 0}
    pattern = "/tmp/vetka_screenshot_*.jpg"
    
    try:
        for filepath in glob.glob(pattern):
            try:
                mtime = os.path.getmtime(filepath)
                age = time.time() - mtime
                if age > max_age_seconds:
                    os.remove(filepath)
                    result["deleted"].append(filepath)
                    result["count"] += 1
                    logger.debug(f"Deleted stale screenshot: {filepath} (age={age:.0f}s)")
            except Exception as e:
                err_msg = f"Failed to process {filepath}: {e}"
                result["errors"].append(err_msg)
                logger.warning(err_msg)
    except Exception as e:
        logger.error(f"cleanup_screenshot_temps failed: {e}")
        result["errors"].append(str(e))
    
    if result["count"]:
        logger.info(f"Cleaned up {result['count']} stale screenshot(s) from /tmp")
    return result
