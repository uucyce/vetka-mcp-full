"""
MCP Browser Profile Manager - isolates Playwright browser sessions.

@status: active
@phase: 147
"""

import os
import tempfile
import shutil
import uuid
from pathlib import Path
from typing import Optional, Dict, Any
from dataclasses import dataclass
from enum import Enum

from playwright.async_api import async_playwright, BrowserContext, Browser


class IsolationStrategy(Enum):
    """Browser isolation strategies."""
    EPHEMERAL = "ephemeral"          # Fresh profile per session (most isolated)
    DEDICATED = "dedicated"          # Fixed MCP-only profile (medium isolation)
    SHARED_FALLBACK = "shared"       # Shared profile as fallback (least isolated)


@dataclass
class BrowserSession:
    """Represents an active browser session."""
    context: BrowserContext
    browser: Browser
    profile_path: Optional[Path]
    strategy: IsolationStrategy
    session_id: str


class BrowserProfileManager:
    """
    Manages isolated browser profiles for MCP automation.
    
    Key features:
    - No collision with user Chrome sessions
    - Automatic fallback if primary strategy fails
    - Cleanup of ephemeral profiles after use
    """
    
    def __init__(self, base_profile_dir: Optional[Path] = None):
        self.base_dir = base_profile_dir or Path.home() / ".cache" / "vetka" / "mcp-browser-profiles"
        self.base_dir.mkdir(parents=True, exist_ok=True)
        
        self.dedicated_profile_dir = self.base_dir / "dedicated-profile"
        self.active_sessions: Dict[str, BrowserSession] = {}
        
    def _get_ephemeral_profile(self) -> Path:
        """Create a temporary profile for ephemeral session."""
        temp_dir = Path(tempfile.mkdtemp(prefix="vetka-mcp-"))
        return temp_dir
    
    def _get_dedicated_profile(self) -> Path:
        """Get or create the dedicated MCP profile."""
        self.dedicated_profile_dir.mkdir(parents=True, exist_ok=True)
        return self.dedicated_profile_dir
    
    async def launch_isolated_browser(
        self,
        strategy: IsolationStrategy = IsolationStrategy.DEDICATED,
        headless: bool = False,
        **launch_kwargs
    ) -> BrowserSession:
        """
        Launch an isolated browser context.
        
        Strategy fallback chain:
        1. DEDICATED: Fixed MCP profile (persistent cookies/session)
        2. EPHEMERAL: Temporary profile (fresh each time)
        3. SHARED_FALLBACK: Last resort, may collide but logs warning
        """
        
        strategies_to_try = []
        if strategy == IsolationStrategy.DEDICATED:
            strategies_to_try = [IsolationStrategy.DEDICATED, IsolationStrategy.EPHEMERAL]
        elif strategy == IsolationStrategy.EPHEMERAL:
            strategies_to_try = [IsolationStrategy.EPHEMERAL]
        else:
            strategies_to_try = [IsolationStrategy.SHARED_FALLBACK, IsolationStrategy.EPHEMERAL]
        
        last_error = None
        
        for current_strategy in strategies_to_try:
            try:
                session_id = str(uuid.uuid4())
                
                if current_strategy == IsolationStrategy.DEDICATED:
                    profile_path = self._get_dedicated_profile()
                    # Critical: Use fresh user-data-dir with no remote debugging port
                    # This ensures Playwright doesn't try to attach to existing Chrome
                    context = await self._launch_persistent_context(
                        profile_path, headless=headless, **launch_kwargs
                    )
                    browser = context.browser
                    
                elif current_strategy == IsolationStrategy.EPHEMERAL:
                    profile_path = self._get_ephemeral_profile()
                    context = await self._launch_persistent_context(
                        profile_path, headless=headless, **launch_kwargs
                    )
                    browser = context.browser
                    
                else:  # SHARED_FALLBACK
                    profile_path = None
                    playwright = await async_playwright().start()
                    browser = await playwright.chromium.launch(
                        headless=headless,
                        **launch_kwargs
                    )
                    context = await browser.new_context()
                
                session = BrowserSession(
                    context=context,
                    browser=browser,
                    profile_path=profile_path,
                    strategy=current_strategy,
                    session_id=session_id
                )
                
                self.active_sessions[session_id] = session
                
                # Log success with strategy info
                import logging
                logger = logging.getLogger(__name__)
                logger.info(f"Browser launched with strategy: {current_strategy.value}")
                
                return session
                
            except Exception as e:
                last_error = e
                import logging
                logger = logging.getLogger(__name__)
                logger.warning(f"Strategy {current_strategy.value} failed: {e}")
                continue
        
        raise RuntimeError(f"All browser launch strategies failed. Last error: {last_error}")
    
    async def _launch_persistent_context(self, profile_path: Path, headless: bool, **kwargs):
        """Launch persistent context with profile isolation."""
        playwright = await async_playwright().start()
        
        # Critical: Force new session by clearing any existing lock files
        lock_file = profile_path / "SingletonLock"
        if lock_file.exists():
            lock_file.unlink(missing_ok=True)
        
        socket_file = profile_path / "SingletonSocket"
        if socket_file.exists():
            socket_file.unlink(missing_ok=True)
        
        # Launch with explicit flags to prevent attaching to existing Chrome
        context = await playwright.chromium.launch_persistent_context(
            str(profile_path),
            headless=headless,
            args=[
                '--no-first-run',
                '--no-default-browser-check',
                '--disable-default-apps',
                '--disable-sync',
                '--disable-background-networking',
                '--disable-client-side-phishing-detection',
                '--disable-component-update',
                '--disable-domain-reliability',
                '--disable-features=ChromeWhatsNewUI,LegacyTLS,OptimizationHints',
                '--disable-hang-monitor',
                '--disable-popup-blocking',
                '--disable-prompt-on-repost',
                '--disable-web-resources',
                '--enable-automation',
                '--force-fieldtrials=*ChromeWhatsNewUI/Disabled/',
                '--force-fieldtrial-params=*ChromeWhatsNewUI.Disabled/',
                '--metrics-recording-only',
                '--no-pings',
                '--no-service-autorun',
                '--password-store=basic',
                '--use-mock-keychain',
            ],
            **kwargs
        )
        
        return context
    
    async def close_session(self, session_id: str):
        """Close a browser session and cleanup profile if ephemeral."""
        session = self.active_sessions.pop(session_id, None)
        if session:
            await session.context.close()
            await session.browser.close()
            
            # Cleanup ephemeral profiles
            if session.strategy == IsolationStrategy.EPHEMERAL and session.profile_path:
                shutil.rmtree(session.profile_path, ignore_errors=True)
    
    async def close_all(self):
        """Close all active sessions."""
        for session_id in list(self.active_sessions.keys()):
            await self.close_session(session_id)


# Global singleton
_default_manager: Optional[BrowserProfileManager] = None


def get_browser_manager() -> BrowserProfileManager:
    """Get or create the global browser profile manager."""
    global _default_manager
    if _default_manager is None:
        _default_manager = BrowserProfileManager()
    return _default_manager

import asyncio
import sys
from typing import Any, Dict, Optional

from src.mcp.browser_profile_manager import get_browser_manager, IsolationStrategy


class IsolatedPlaywrightMCP:
    """
    Wrapper around Playwright MCP that uses isolated browser profiles.
    
    Usage:
        mcp = IsolatedPlaywrightMCP()
        await mcp.start()
        # Use browser_navigate, browser_tools, etc.
    """
    
    def __init__(self, strategy: IsolationStrategy = IsolationStrategy.DEDICATED):
        self.strategy = strategy
        self.browser_manager = get_browser_manager()
        self._active_session = None
        self._original_launch = None
        
    async def start(self):
        """Start MCP with isolated browser."""
        # Launch isolated browser session
        self._active_session = await self.browser_manager.launch_isolated_browser(
            strategy=self.strategy,
            headless=False  # Visible for debugging
        )
        
        # Patch Playwright tools to use this context
        await self._patch_mcp_tools()
        
        return self._active_session
    
    async def _patch_mcp_tools(self):
        """Patch existing MCP tools to use our isolated context."""
        try:
            # Import the MCP server module
            from src.mcp.mycelium_mcp_server import server
            
            # Override browser context creation
            original_setup = getattr(server, '_setup_browser', None)
            
            async def patched_browser_setup(*args, **kwargs):
                """Use our isolated browser instead."""
                return self._active_session.context
            
            server._setup_browser = patched_browser_setup
            
        except ImportError:
            # Fallback: monkeypatch playwright.launch directly
            import playwright.async_api
            
            original_launch = playwright.async_api._impl._BrowserType.launch
            
            async def isolated_launch(self, **kwargs):
                """Replace launch with our isolated context."""
                if 'persistent' in kwargs:
                    # Use our manager for persistent contexts
                    session = await get_browser_manager().launch_isolated_browser(
                        strategy=self.strategy
                    )
                    return session.context
                return await original_launch(self, **kwargs)
            
            playwright.async_api._impl._BrowserType.launch = isolated_launch
    
    async def stop(self):
        """Clean up browser session."""
        if self._active_session:
            await self.browser_manager.close_session(self._active_session.session_id)
            self._active_session = None

## Browser Profile Isolation for MCP Playwright

### Problem
Playwright MCP previously failed with `Opening in existing browser session` when user Chrome was already running. This occurred because Playwright tried to attach to the existing Chrome instance with remote debugging.

### Solution
Implemented three-layer isolation strategy:

1. **Dedicated Profile (Default)**
   - Fixed MCP-only Chrome profile at `~/.cache/vetka/mcp-browser-profiles/dedicated-profile`
   - Preserves cookies/session across MCP restarts
   - Uses unique flags to prevent attaching to user Chrome
   - **No collision with user Chrome**

2. **Ephemeral Profile (Fallback)**
   - Temporary profile per session
   - Automatically cleaned up on exit
   - 100% isolated, but loses session state

3. **Shared Fallback (Last Resort)**
   - Standard Chrome launch without persistent context
   - Logs warning but may still collide
   - Only used if both above fail
