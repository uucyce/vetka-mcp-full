import inspect
import textwrap
from typing import Type, get_type_hints
from dataclasses import dataclass, fields


def generate_tool_wrapper(tool_class: Type) -> str:
    """
    Auto-generate MCP tool wrapper from SessionInitTool (or any Tool class).
    
    Extracts _execute_async signature and generates matching wrapper function.
    """
    # Get the _execute_async method
    execute_method = getattr(tool_class, '_execute_async', None)
    if not execute_method:
        raise ValueError(f"{tool_class.__name__} missing _execute_async method")
    
    # Get signature
    sig = inspect.signature(execute_method)
    type_hints = get_type_hints(execute_method)
    
    # Build parameter list (skip 'self')
    params = []
    for name, param in sig.parameters.items():
        if name == 'self':
            continue
        
        # Build param with type annotation
        param_type = type_hints.get(name, 'Any')
        default = ''
        if param.default is not inspect.Parameter.empty:
            default = f' = {repr(param.default)}'
        
        params.append(f'{name}: {param_type.__name__ if hasattr(param_type, "__name__") else str(param_type)}{default}')
    
    # Build function call arguments
    call_args = [name for name in sig.parameters.keys() if name != 'self']
    
    # Generate wrapper code
    tool_name = tool_class.__name__.replace('Tool', '').lower()
    wrapper_code = f'''\
# MARKER_AUTO_GENERATED: Wrapper for {tool_class.__name__}
# Generated from _execute_async signature — DO NOT EDIT MANUALLY

async def vetka_{tool_name}(
    {',\\n    '.join(params)}
) -> dict:
    """
    {tool_class.__doc__ or f"Auto-generated wrapper for {tool_class.__name__}"}
    
    MARKER_AUTO: This wrapper is auto-generated from {tool_class.__name__}._execute_async
    """
    tool = {tool_class.__name__}()
    return await tool._execute_async(
        {',\\n        '.join(call_args)}
    )
'''
    return wrapper_code

# src/mcp/tools/regenerate_wrappers.py
# MARKER_198.DEBRIEF: Fix 4/4 — build-time wrapper generation

import re
from pathlib import Path
from wrapper_generator import generate_tool_wrapper

# Registry of all tool classes that need wrappers
TOOL_REGISTRY = [
    "src.mcp.tools.session_init.SessionInitTool",
    "src.mcp.tools.task_board_tools.TaskBoardTool", 
    "src.mcp.tools.experience_report.ExperienceReportTool",
    # ... etc
]

def regenerate_all_wrappers():
    """Regenerate all tool wrappers from their source classes."""
    for tool_path in TOOL_REGISTRY:
        module_path, class_name = tool_path.rsplit('.', 1)
        module = __import__(module_path, fromlist=[class_name])
        tool_class = getattr(module, class_name)
        
        wrapper_code = generate_tool_wrapper(tool_class)
        
        # Find the wrapper file and replace MARKER_AUTO_GENERATED section
        wrapper_file = Path(f"src/mcp/tools/wrappers/{class_name.lower().replace('tool', '')}_wrapper.py")
        if wrapper_file.exists():
            content = wrapper_file.read_text()
            # Replace between markers
            pattern = r'(MARKER_AUTO_GENERATED_START.*?MARKER_AUTO_GENERATED_END)'
            new_content = re.sub(
                pattern, 
                f'MARKER_AUTO_GENERATED_START\\n{wrapper_code}\\n# MARKER_AUTO_GENERATED_END',
                content,
                flags=re.DOTALL
            )
            wrapper_file.write_text(new_content)
            print(f"✓ Regenerated {wrapper_file}")
        else:
            # Create new wrapper file
            wrapper_file.parent.mkdir(parents=True, exist_ok=True)
            wrapper_file.write_text(f"# Auto-generated wrapper\\n{wrapper_code}")
            print(f"✓ Created {wrapper_file}")


if __name__ == "__main__":
    regenerate_all_wrappers()
