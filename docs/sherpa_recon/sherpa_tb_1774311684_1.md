@router.post("/detect-scenes")
async def detect_scenes(video_path: str, method: str = "histogram"):
    from src.cutting.scene_graph.detector import detect
    scenes = detect(video_path, method)
    return {"scenes": scenes}

@router.post("/graph")
async def build_graph(cuts: list[dict]) -> TimelineGraph:
    graph = TimelineGraph()
    for cut in cuts:
        graph.add_clip(ClipNode(**cut))
    return graph

@router.post("/render")
async def start_render(graph: TimelineGraph, background_tasks: BackgroundTasks):
    job_id = str(uuid.uuid4())
    background_tasks.add_task(render_cut, graph.dict(), f"/tmp/{job_id}.mp4")
    return {"job_id": job_id}

@router.get("/status/{job_id}")
