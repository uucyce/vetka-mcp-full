Analysis: [DEBRIEF-BUG] Pre-transcode blocks scan-matrix job completion
The Problem
Currently, when a user imports 50 ProRes files, the scan-matrix job stays in 'running' state for minutes while transcoding happens synchronously. This violates the principle of job decoupling — metadata scanning should complete immediately, while transcoding should be a separate background job with its own progress tracking.
Architecture Pattern: Async Job Queue with Decoupled Stages
Based on production patterns from FFmpeg-based video processing systems , the solution involves:
Scan Job → Completes immediately after metadata extraction
Transcode Job → Spawned separately with independent progress tracking
Job State Machine → Clear separation between scanning → scan_complete → transcoding → ready

from enum import Enum, auto

class MediaJobStage(Enum):
    """Decoupled job stages for scan/transcode separation."""
    SCAN_QUEUED = auto()
    SCANNING = auto()
    SCAN_COMPLETE = auto()      # ← NEW: Scan done, transcode not started
    TRANSCODE_QUEUED = auto()   # ← NEW: Waiting for transcode worker
    TRANSCODING = auto()
    TRANSCODE_COMPLETE = auto() # ← NEW: Proxy ready
    READY = auto()
    FAILED = auto()

class MediaJob:
    """Split job tracking for scan vs transcode phases."""
    job_id: str
    file_path: str
    
    # Scan phase (fast, metadata-only)
    scan_stage: MediaJobStage
    scan_progress: float  # 0-100, metadata extraction
    scan_completed_at: Optional[datetime]
    metadata: Optional[MediaMetadata]
    
    # Transcode phase (slow, background)
    transcode_job_id: Optional[str]  # ← Separate job ID for transcode
    transcode_stage: Optional[MediaJobStage]
    transcode_progress: float  # 0-100, FFmpeg progress
    proxy_path: Optional[str]
    
    # Aggregate status for UI
    @property
    def overall_status(self) -> str:
        if self.scan_stage == MediaJobStage.SCAN_COMPLETE:
            if self.transcode_stage == MediaJobStage.TRANSCODE_COMPLETE:
                return "ready"
            elif self.transcode_stage:
                return f"transcoding_{int(self.transcode_progress)}%"
            return "scan_complete_pending_transcode"
        return self.scan_stage.name.lower()

class MediaScanner:
    """Fast metadata scanner that spawns transcode separately."""
    
    async def scan_and_spawn_transcode(self, job_id: str, file_path: str):
        """MARKER_DEBRIEF_BUG_002: Complete scan fast, spawn transcode async."""
        queue = get_job_queue()
        
        try:
            # Phase 1: Fast metadata scan (seconds)
            await queue.update_job(job_id, stage=MediaJobStage.SCANNING)
            metadata = await self._extract_metadata(file_path)
            
            # Mark scan complete immediately
            await queue.update_job(
                job_id, 
                stage=MediaJobStage.SCAN_COMPLETE,
                metadata=metadata,
                scan_completed_at=utc_now()
            )
            
            # Phase 2: Spawn separate transcode job (minutes, background)
            # This does NOT block the scan job completion
            transcode_job = await queue.create_job(
                type="media_transcode",
                payload={
                    "source_job_id": job_id,
                    "file_path": file_path,
                    "target_codec": "h264",
                    "proxy_settings": {...}
                },
                priority=JobPriority.LOW,  # Background priority
                parent_job_id=job_id  # Link for UI aggregation
            )
            
            # Update scan job with transcode reference
            await queue.update_job(
                job_id,
                transcode_job_id=transcode_job.id,
                transcode_stage=MediaJobStage.TRANSCODE_QUEUED
            )
            
            # Hand off to transcode worker (non-blocking)
            await self._enqueue_transcode(transcode_job.id)
            
        except Exception as e:
            await queue.update_job(job_id, stage=MediaJobStage.FAILED, error=str(e))

"""
MARKER_DEBRIEF_BUG_003: Separate transcode worker pool.
Runs as independent process, not blocking API or scan jobs.
"""

import asyncio
from typing import Optional
import ffmpeg

class TranscodeWorker:
    """Background worker for FFmpeg transcoding with progress reporting."""
    
    def __init__(self, worker_id: str, queue: JobQueue):
        self.worker_id = worker_id
        self.queue = queue
        self.current_job: Optional[str] = None
        
    async def run(self):
        """Main worker loop - picks up transcode jobs independently."""
        while True:
            job = await self.queue.dequeue(
                job_types=["media_transcode"],
                timeout=30
            )
            if job:
                await self._process_transcode(job)
    
    async def _process_transcode(self, job: Job):
        """Execute FFmpeg transcode with progress tracking."""
        self.current_job = job.id
        
        try:
            await self.queue.update_job(
                job.id, 
                stage=MediaJobStage.TRANSCODING,
                worker_id=self.worker_id
            )
            
            # FFmpeg async execution with progress parsing
            process = await asyncio.create_subprocess_exec(
                "ffmpeg",
                "-i", job.payload["file_path"],
                "-c:v", "libx264",
                "-preset", "fast",
                "-progress", "pipe:1",  # Enable progress output
                job.payload["output_path"],
                stdout=asyncio.subprocess.PIPE,
                stderr=asyncio.subprocess.PIPE
            )
            
            # Parse progress without blocking
            progress = 0.0
            while True:
                line = await process.stdout.readline()
                if not line:
                    break
                    
                # Parse FFmpeg progress (out_time_ms, progress, etc.)
                parsed = self._parse_progress_line(line.decode())
                if parsed:
                    progress = parsed.get("progress_percent", progress)
                    await self.queue.update_job(
                        job.id,
                        transcode_progress=progress,
                        # Also update parent scan job for UI aggregation
                        parent_update={"transcode_progress": progress}
                    )
            
            await process.wait()
            
            if process.returncode == 0:
                await self._mark_transcode_complete(job)
            else:
                await self._mark_transcode_failed(job, "FFmpeg error")
                
        except Exception as e:
            await self._mark_transcode_failed(job, str(e))
        finally:
            self.current_job = None
    
    def _parse_progress_line(self, line: str) -> Optional[dict]:
        """Parse FFmpeg progress output (out_time_ms, progress, etc.)."""
        if line.startswith("out_time_ms="):
            # Convert to percentage if duration known
            return {"progress_percent": self._calculate_percent(line)}
        return None

interface DecoupledJobStatus {
  scanStatus: 'queued' | 'scanning' | 'complete' | 'failed';
  scanProgress: number;  // Fast: 0-100 in seconds
  
  transcodeStatus?: 'queued' | 'transcoding' | 'complete' | 'failed';
  transcodeProgress?: number;  // Slow: 0-100 in minutes
}

const JobStatusBadge: React.FC<{ job: DecoupledJobStatus }> = ({ job }) => {
  // Show scan completion immediately
  if (job.scanStatus === 'scanning') {
    return <Badge color="blue">Scanning {job.scanProgress}%</Badge>;
  }
  
  if (job.scanStatus === 'complete') {
    // Scan done! Show transcode progress separately
    if (job.transcodeStatus === 'transcoding') {
      return (
        <div>
          <Badge color="green">✓ Scanned</Badge>
          <ProgressBar value={job.transcodeProgress} label="Transcoding..." />
        </div>
      );
    }
    if (job.transcodeStatus === 'queued') {
      return <Badge color="yellow">Scanned • Transcode queued</Badge>;
    }
    return <Badge color="green">Ready</Badge>;
  }
  
  return <Badge color="red">Failed</Badge>;
};
