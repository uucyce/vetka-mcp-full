"""
MARKER_198.PATH_RESOLVER: Unified project root resolution for git worktrees.

Handles:
- Claude Code worktrees (.claude/worktrees/<name>/)
- Standard git worktrees
- Normal repo (non-worktree) execution
- MCP server context
"""

from __future__ import annotations

import os
from pathlib import Path
from typing import Optional

# Environment override for MCP servers or explicit configuration
ENV_MAIN_REPO = "VETKA_MAIN_REPO"
ENV_WORKTREE_ROOT = "VETKA_WORKTREE_ROOT"


def resolve_main_repo_root() -> Path:
    """
    Return the main repository root (where .git/ lives).
    
    Priority:
    1. VETKA_MAIN_REPO env var (explicit override)
    2. Detect .claude/worktrees/ pattern (Claude Code)
    3. Detect standard git worktree (via .git file)
    4. Fallback to __file__ relative
    """
    # Priority 1: Explicit env var
    if env_root := os.environ.get(ENV_MAIN_REPO):
        path = Path(env_root)
        if path.is_dir():
            return path.resolve()
    
    # Priority 2: Claude Code worktree detection
    candidate = Path(__file__).resolve()
    parts = candidate.parts
    if ".claude" in parts:
        try:
            wt_idx = parts.index(".claude")
            main_root = Path(*parts[:wt_idx])
            if (main_root / ".git").exists() or (main_root / "data").exists():
                return main_root
        except (ValueError, IndexError):
            pass
    
    # Priority 3: Standard git worktree detection (.git is a file pointing to main)
    git_dir = candidate
    while len(git_dir.parts) > 1:
        git_file = git_dir / ".git"
        if git_file.exists():
            if git_file.is_file():
                # Git worktree: .git file contains "gitdir: <path>"
                try:
                    content = git_file.read_text().strip()
                    if content.startswith("gitdir:"):
                        gitdir = Path(content.replace("gitdir:", "").strip())
                        # Main repo is parent of .git directory
                        main_root = gitdir.parent.parent if gitdir.name == ".git" else gitdir.parent
                        if main_root.exists():
                            return main_root.resolve()
                except Exception:
                    pass
            elif git_file.is_dir():
                # Main repo: .git is a directory
                return git_dir.resolve()
        git_dir = git_dir.parent
    
    # Priority 4: Fallback
    return candidate.parent.parent.parent


def resolve_worktree_root() -> Optional[Path]:
    """Return the current worktree root if in a worktree, else None."""
    if env_wt := os.environ.get(ENV_WORKTREE_ROOT):
        path = Path(env_wt)
        if path.is_dir():
            return path.resolve()
    
    # Detect if we're in a worktree (not main repo)
    main = resolve_main_repo_root()
    current = Path(__file__).resolve()
    
    # If current path is under .claude/worktrees/, we're in a worktree
    if ".claude" in current.parts:
        return current.parent.parent.parent  # Up to worktree root
    
    # Check if .git is a file (worktree) vs directory (main repo)
    git_file = current.parent.parent.parent / ".git"
    if git_file.exists() and git_file.is_file():
        return current.parent.parent.parent
    
    return None


def get_data_dir() -> Path:
    """Return data directory (always in main repo)."""
    return resolve_main_repo_root() / "data"


def get_memory_dir() -> Path:
    """Return memory directory (always in main repo)."""
    return resolve_main_repo_root() / "memory"
