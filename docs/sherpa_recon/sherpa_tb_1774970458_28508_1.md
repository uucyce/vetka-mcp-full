import subprocess
from pathlib import Path

async def _auto_commit_files(
    self, 
    task_id: str, 
    files: List[str], 
    message: str
) -> Optional[str]:
    """
    Stage files, create commit, and trigger VETKA hooks.
    
    Returns:
        Commit hash if successful, None otherwise
    """
    try:
        repo_root = self._get_repo_root()
        
        # 1. Stage all written files
        for file_path in files:
            rel_path = Path(file_path).relative_to(repo_root)
            subprocess.run(
                ["git", "add", str(rel_path)],
                cwd=repo_root,
                check=True,
                capture_output=True
            )
        
        # 2. Check if there are staged changes
        status_result = subprocess.run(
            ["git", "diff", "--cached", "--name-only"],
            cwd=repo_root,
            capture_output=True,
            text=True
        )
        
        if not status_result.stdout.strip():
            logger.info(f"[CORE-GAP-1] No changes to commit for task {task_id}")
            return None
        
        # 3. Create commit with task reference
        # Message format: "[task:{id}] {description}"
        commit_msg = f"{message}\n\nAuto-generated by VETKA Pipeline\nTask-Id: {task_id}"
        
        commit_result = subprocess.run(
            ["git", "commit", "-m", commit_msg],
            cwd=repo_root,
            check=True,
            capture_output=True,
            text=True
        )
        
        # 4. Extract commit hash
        hash_result = subprocess.run(
            ["git", "rev-parse", "HEAD"],
            cwd=repo_root,
            capture_output=True,
            text=True,
            check=True
        )
        commit_hash = hash_result.stdout.strip()
        
        # 5. Trigger VETKA hooks if vetka_git_commit exists
        await self._trigger_vetka_hooks(task_id, commit_hash, files)
        
        logger.info(f"[CORE-GAP-1] Auto-committed {len(files)} files for task {task_id}: {commit_hash[:8]}")
        return commit_hash
        
    except subprocess.CalledProcessError as e:
        logger.error(f"[CORE-GAP-1] Git commit failed: {e.stderr}")
        return None
    except Exception as e:
        logger.error(f"[CORE-GAP-1] Unexpected error during commit: {e}")
        return None
