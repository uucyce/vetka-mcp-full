interface MonitorTransportProps {
  feed: 'source' | 'timeline';
  // ... other props
}

export const MonitorTransport: React.FC<MonitorTransportProps> = ({ feed }) => {
  const { focusedPanel, markInPoint, markOutPoint, goToInPoint, goToOutPoint } = useCutEditorStore();
  
  const handleMarkIn = () => {
    // FOCUS GUARD: Only execute if the correct panel is focused
    if (feed === 'source' && focusedPanel !== 'source') {
      console.warn('[MonitorTransport] Source action blocked: timeline has focus');
      return;
    }
    if (feed === 'timeline' && focusedPanel !== 'timeline') {
      console.warn('[MonitorTransport] Timeline action blocked: source has focus');
      return;
    }
    
    // Execute on correct target
    if (feed === 'source') {
      markInPoint({ target: 'source' });
    } else {
      markInPoint({ target: 'timeline' });
    }
  };
  
  // Similar guards for markOut, goToIn, goToOut
};
