PARALLAX-B: Scene Assembly Architecture
Core Concept: The Parallax Scene Contract
The goal is to create a deterministic, headless assembly pipeline that produces an ordered parallax scene data structure—not just a folder of plates. This contract must encode:
Layer ordering (z-depth hierarchy)
Relative object distances (depth-normalized positioning)
Lens/focal behavior hints (camera-ready metadata)
Clean plate relationships (background repair linkage)

interface ParallaxSceneContract {
  version: "1.0.0";
  sourceImage: string;           // Original input reference
  depthMap: string;              // Path to depth map
  metadata: SceneMetadata;
  layers: OrderedLayer[];
  camera: CameraHints;
  exportReady: boolean;
}

interface SceneMetadata {
  generatedAt: string;
  depthRange: { min: number; max: number };  // Raw depth values
  normalizedRange: { near: number; far: number };  // 0-1 or lens units
  units: "normalized" | "millimeters" | "pixels";
}

interface OrderedLayer {
  id: string;
  plateId: string;               // Links to layer_space/manifest
  type: "foreground" | "midground" | "background" | "clean_plate";
  
  // Depth positioning
  depth: {
    rawValue: number;            // From depth map (median or representative)
    normalizedZ: number;         // 0 (far) to 1 (near) or -1 to +1
    worldZ?: number;             // Optional: lens-aware distance
  };
  
  // Visual assets
  assets: {
    rgbaPlate: string;           // Extracted layer with alpha
    cleanPlate?: string;         // Background repair behind this layer
    matte?: string;              // Optional refined matte
  };
  
  // Ordering (deterministic, not just array index)
  sortOrder: number;             // Explicit z-order (0 = far background)
  parallaxFactor: number;        // Derived from depth (0.1 = far, 1.0 = near)
  
  // Spatial bounds for camera safety checks
  bounds: {
    width: number;
    height: number;
    originalPosition: { x: number; y: number };
  };
}

interface CameraHints {
  focalLength?: number;          // 35mm equivalent hint
  aperture?: number;             // For depth of field simulation
  nearPlane: number;             // Camera clipping
  farPlane: number;
  safeMargins: {
    horizontal: number;          // Pixels safe to pan
    vertical: number;
  };
  recommendedMoves: CameraMovePreset[];
}

interface CameraMovePreset {
  name: "push_in" | "push_out" | "pan_left" | "pan_right" | "orbit_cw" | "orbit_ccw";
  intensity: "subtle" | "moderate" | "dramatic";
  duration: number;              // Seconds
  keyframes: KeyframeHint[];
}

def assign_layer_depths(
    depth_map: np.ndarray,
    layer_manifest: List[LayerManifestEntry],
    strategy: str = "centroid_median"
) -> List[LayerDepthAssignment]:
    """
    Deterministic depth assignment for each extracted layer.
    Avoids randomness—same inputs always produce same scene.
    """
    assignments = []
    
    for layer in layer_manifest:
        # Sample depth at layer's non-transparent pixels
        mask = load_alpha_mask(layer.mask_path)
        masked_depth = depth_map[mask > 0.5]
        
        if strategy == "centroid_median":
            # Use median of depth values at layer's visible pixels
            # More robust than mean against edges
            depth_value = float(np.median(masked_depth))
        elif strategy == "centroid_mean":
            depth_value = float(np.mean(masked_depth))
        elif strategy == "contour_sample":
            # Sample along contour for edge objects
            depth_value = sample_contour_depth(depth_map, mask)
        
        assignments.append(LayerDepthAssignment(
            layer_id=layer.id,
            plate_id=layer.plate_id,
            raw_depth=depth_value,
            sample_strategy=strategy
        ))
    
    return assignments

def compute_layer_ordering(
    assignments: List[LayerDepthAssignment],
    depth_range: Tuple[float, float]
) -> List[OrderedLayer]:
    """
    Convert raw depths to explicit scene ordering.
    Background (far) = 0, Foreground (near) = N
    """
    min_d, max_d = depth_range
    
    # Sort by depth (far to near)
    sorted_layers = sorted(assignments, key=lambda x: x.raw_depth, reverse=True)
    
    ordered = []
    for idx, layer in enumerate(sorted_layers):
        normalized = (layer.raw_depth - min_d) / (max_d - min_d)
        
        # Parallax factor: how much layer moves relative to camera
        # Near objects (normalized ~1) move faster (factor ~1.0)
        # Far objects (normalized ~0) move slower (factor ~0.1)
        parallax_factor = 0.1 + (normalized * 0.9)
        
        # Classify layer type based on depth percentiles
        if normalized < 0.2:
            layer_type = "background"
        elif normalized < 0.6:
            layer_type = "midground"
        else:
            layer_type = "foreground"
        
        ordered.append(OrderedLayer(
            ...  # populate from schema
            sort_order=idx,
            depth=DepthInfo(
                raw_value=layer.raw_depth,
                normalized_z=normalized,
                world_z=None  # Computed later if lens hints provided
            ),
            parallax_factor=parallax_factor,
            type=layer_type
        ))
    
    return ordered

def link_clean_plates(
    ordered_layers: List[OrderedLayer],
    clean_plate_manifest: Dict[str, str]
) -> List[OrderedLayer]:
    """
    Associate each layer with the clean plate that sits BEHIND it.
    A layer's clean plate fills the hole when this layer is displaced.
    """
    for i, layer in enumerate(ordered_layers):
        # Find the next layer behind (i-1 in far-to-near sorted list)
        if i > 0:
            behind_layer = ordered_layers[i-1]
            # If there's a clean plate specifically for this layer
            if layer.plate_id in clean_plate_manifest:
                layer.assets.clean_plate = clean_plate_manifest[layer.plate_id]
            else:
                # Use the layer behind as implicit clean plate reference
                layer.assets.clean_plate = behind_layer.assets.rgbaPlate
    
    return ordered_layers

def compute_camera_hints(
    ordered_layers: List[OrderedLayer],
    source_dimensions: Tuple[int, int],
    lens_preset: Optional[str] = None
) -> CameraHints:
    """
    Calculate safe camera movement margins based on layer bounds.
    Prevents revealing empty edges during parallax moves.
    """
    max_parallax = max(l.parallax_factor for l in ordered_layers)
    min_parallax = min(l.parallax_factor for l in ordered_layers)
    
    # Conservative safe margins based on max expected displacement
    # Assuming 10% camera movement range
    movement_range = 0.1
    max_displacement = (max_parallax - min_parallax) * movement_range
    
    width, height = source_dimensions
    safe_h = int(width * max_displacement * 0.5)  # Half for each direction
    safe_v = int(height * max_displacement * 0.5)
    
    # Lens presets for AE compatibility
    presets = {
        "standard": {"focal": 50, "aperture": 2.8},
        "telephoto": {"focal": 85, "aperture": 1.8},
        "wide": {"focal": 24, "aperture": 4.0}
    }
    lens = presets.get(lens_preset, presets["standard"])
    
    return CameraHints(
        focalLength=lens["focal"],
        aperture=lens["aperture"],
        nearPlane=0.1,
        farPlane=1000.0,
        safeMargins=SafeMargins(horizontal=safe_h, vertical=safe_v),
        recommendedMoves=generate_preset_moves(ordered_layers)
    )

# src/parallax/cli/assemble_scene.py

@click.command()
@click.option('--depth-map', required=True, help='Path to depth map PNG')
@click.option('--layer-manifest', required=True, help='layer_space.json from extraction')
@click.option('--clean-plate-dir', help='Directory with clean plate outputs')
@click.option('--output', required=True, help='Output scene contract JSON path')
@click.option('--lens-preset', default='standard', 
              type=click.Choice(['standard', 'telephoto', 'wide']))
@click.option('--strategy', default='centroid_median',
              type=click.Choice(['centroid_median', 'centroid_mean', 'contour_sample']))
def assemble_scene(
    depth_map: str,
    layer_manifest: str,
    clean_plate_dir: Optional[str],
    output: str,
    lens_preset: str,
    strategy: str
):
    """
    Deterministic parallax scene assembly from depth + layers + clean plates.
    
    Produces a coherent scene contract ready for motion generation or AE import.
    """
    # Load inputs
    depth = cv2.imread(depth_map, cv2.IMREAD_GRAYSCALE)
    manifest = LayerManifest.load(layer_manifest)
    clean_plates = load_clean_plates(clean_plate_dir) if clean_plate_dir else {}
    
    # Assembly pipeline
    assembler = SceneAssembler(strategy=strategy, lens_preset=lens_preset)
    contract = assembler.assemble(
        depth_map=depth,
        layer_manifest=manifest,
        clean_plates=clean_plates,
        source_image=manifest.source_image
    )
    
    # Validate and write
    validate_contract(contract)
    contract.save(output)
    
    click.echo(f"Scene contract written to {output}")
    click.echo(f"Layers: {len(contract.layers)}")
    click.echo(f"Depth range: {contract.metadata.depth_range.min:.2f} - "
               f"{contract.metadata.depth_range.max:.2f}")
    click.echo(f"Safe camera margins: ±{contract.camera.safeMargins.horizontal}px "
               f"horizontal, ±{contract.camera.safeMargins.vertical}px vertical")
Integration with Existing Layer Space
The assembly pipeline reuses existing contracts:
layer_space.json (from extraction): Provides plateId, bounds, alpha paths
manifest.json (from cleanup): Links clean plates to target plates
Depth service: Already produces depth_map.png
The scene contract is a new aggregation layer that doesn't replace these but references them, maintaining the established object-centric authority pattern .
