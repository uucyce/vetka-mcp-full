export const CorrectionTools = ({ selectedObject, brushMode, setBrushMode, brushSize, setBrushSize }) => {
  if (!selectedObject) return null;
  return (
    <div className="correction-tools panel">
      <div className="eyebrow">CORRECT: {selectedObject.name}</div>
      <div className="tool-row">
        <button className={brushMode === 'add' ? 'active' : ''} onClick={() => setBrushMode('add')}>➕ Add</button>
        <button className={brushMode === 'remove' ? 'active' : ''} onClick={() => setBrushMode('remove')}>✖ Remove</button>
        <button className={brushMode === 'refine' ? 'active' : ''} onClick={() => setBrushMode('refine')}>✨ Refine edge</button>
      </div>
      <label>Brush size <input type="range" min="5" max="100" value={brushSize} onChange={e => setBrushSize(Number(e.target.value))} /></label>
      <div className="hint">Click + drag on the monitor to correct this object</div>
    </div>
  );
};

4. Risks & Edge Cases
Risk	Mitigation
No plate/object selected when brush active	Disable brush strokes and show tooltip: “Select an object first”
Overlapping plates cause ambiguous selection	Use Z-order + modifier key (Ctrl+click cycles through plates)
Brush strokes affect wrong object	Store selectedObjectId at stroke start; ignore strokes if selection changes mid-stroke
Performance of outline drawing on high-res images	Use requestAnimationFrame throttling; draw only on selection change or canvas resize
Portrait/square images break outline positioning	Use relative bounds (0–1 normalized) and scale to actual canvas display size
Existing appendMatteSeed expects global seeds, not per-object	Extend API: appendMatteSeed(x, y, mode, targetPlateId) – update backend contract (out of scope for this task but note as dependency)
