const handleExtractChange = (updater: (prev: ExtractParams) => ExtractParams) => {
  setExtractParams(updater);
  // Auto-show mask preview if not already showing
  if (!extractPreviewActive && previewMode !== 'selection') {
    setPreviousPreviewMode(previewMode);
    setPreviewMode('selection');
    setExtractPreviewActive(true);
    // Auto-revert after 2 seconds of inactivity (debounced)
    if (extractTimeoutRef.current) clearTimeout(extractTimeoutRef.current);

      setExtractPreviewActive(false);

  }
};

Missing TypeScript definitions for new ExtractParams.	Define interface in lib/metrics.ts and update FocusSettings or create ExtractSettings.
