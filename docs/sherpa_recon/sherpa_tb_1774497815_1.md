const drawSplitFrame = (timeA: number, timeB: number) => {
  const video = videoRef.current;
  const canvas = canvasRef.current;

  const w = canvas.width / 2;
  const h = canvas.height;

  video.currentTime = timeA;

  ctx?.drawImage(video, w, 0, w, h);
};
