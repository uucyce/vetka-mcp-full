from abc import ABC, abstractmethod
from typing import Any, Dict

class BaseRuntime(ABC):
    @abstractmethod
    async def initialize(self) -> None:
        """Prepare runtime (load models, start subprocesses, etc.)."""
        pass

    @abstractmethod
    async def run_task(self, task_id: str, workflow_family: str, params: Dict[str, Any]) -> Dict[str, Any]:
        """
        Execute a task and return result metadata including:
        - runtime_ms: int
        - success: bool
        - artifacts: List[str] (paths of produced artifacts)
        """
        pass

    @abstractmethod
    async def cold_start_measure(self, workflow_family: str) -> float:
        """Measure time (ms) from runtime init to first task completion."""
        pass

    @abstractmethod
    async def shutdown(self) -> None:
        """Clean up resources."""
        pass

from src.dependencies import get_orchestrator
from src.benchmark.runtime_interface import BaseRuntime

class LocalRuntime(BaseRuntime):
    async def initialize(self):
        # Use existing orchestrator from app state (or initialize fresh)
        self.orchestrator = get_orchestrator(request=None)  # adapt for standalone use
        await self.orchestrator.ready()

    async def run_task(self, task_id, workflow_family, params):
        start = time.perf_counter()
        try:
            result = await self.orchestrator.execute_workflow(workflow_family, params)
            runtime_ms = (time.perf_counter() - start) * 1000
            success = result.get("status") == "success"
            artifacts = result.get("output_files", [])
            return {"runtime_ms": runtime_ms, "success": success, "artifacts": artifacts}
        except Exception as e:
            return {"runtime_ms": (time.perf_counter() - start) * 1000, "success": False, "artifacts": [], "error": str(e)}

async def run_benchmark(runtime: BaseRuntime, runtime_name: str, iterations: int = 10):
    store = MCCBenchmarkStore()
    await runtime.initialize()

    for workflow in BENCHMARK_WORKLOADS:
        # Cold start
        cold_start_ms = await runtime.cold_start_measure(workflow.name)
        store.add_record({
            "runtime_name": runtime_name,
            "workflow_family": workflow.name,
            "runtime_ms": 0,  # not applicable
            "cold_start_ms": cold_start_ms,
            "run_status": "cold_start"
        })

        # Warm iterations
        for i in range(iterations):
            result = await runtime.run_task(f"{workflow.name}_{i}", workflow.name, workflow.params)
            artifact_status = validate_artifacts(result.get("artifacts", []), workflow.expected_artifacts)
            store.add_record({
                "runtime_name": runtime_name,
                "workflow_family": workflow.name,
                "runtime_ms": result["runtime_ms"],
                "cold_start_ms": 0,
                "artifact_present_count": artifact_status["present"],
                "artifact_missing_count": artifact_status["missing"],
                "required_artifact_count": len(workflow.expected_artifacts),
                "success_rate": 1.0 if result["success"] else 0.0,
                "run_status": "measured"
            })

    await runtime.shutdown()

4. Risks & Edge Cases
Risk	Mitigation
LiteRT interface unknown	Implement a mock LiteRTRuntime that simulates expected performance (or use a simple subprocess wrapper). In final integration, replace with real adapter once LiteRT spec is available.
Environment interference (caching, GPU state)	Run each runtime in isolation (separate processes or fresh Docker containers). Clear system caches between runs where possible.
Non‑deterministic FFmpeg timings	Use identical input files and fixed hardware (disable turbo boost, pin CPU cores). Run many iterations and report confidence intervals.
Cold start definition ambiguous	Define as: time from calling initialize() until first run_task() completes. Store separately from warm runtime_ms.
Concurrent writes to benchmark store	Benchmark runner runs sequentially, no concurrency. For safety, add file lock or use SQLite instead of JSON if parallel execution needed.
Missing LiteRT implementation blocks testing	Provide a fallback “simulated” runtime that reads pre‑recorded performance data – allows validation of benchmark harness before LiteRT is ready.
Artifact validation failure due to path differences	Normalize artifact paths to relative paths from project root; compare using checksums (MD5) rather than simple existence.
Long benchmark duration	Make benchmark asynchronous, return a task ID, and allow polling. Provide CLI flag to limit iterations.
