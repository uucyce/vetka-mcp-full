from fastapi import BackgroundTasks, HTTPException

class ProxyStatus(str, Enum):
    MISSING = "missing"
    GENERATING = "generating"
    READY = "ready"
    FAILED = "failed"

# In-memory store of generation tasks (replace with Redis/DB in production)
_generation_tasks: dict[str, str] = {}  # video_id -> status

@router.get("/video/{video_id}/proxy-status")
async def get_proxy_status(video_id: str):
    """Return current proxy status."""
    proxy_path = get_proxy_path(video_id)
    if proxy_path.exists():
        return {"status": ProxyStatus.READY}
    status = _generation_tasks.get(video_id, ProxyStatus.MISSING)
    return {"status": status}

@router.post("/video/{video_id}/generate-proxy")
async def generate_proxy(video_id: str, background_tasks: BackgroundTasks):
    """Start proxy generation in background."""
    proxy_path = get_proxy_path(video_id)
    if proxy_path.exists():
        return {"status": ProxyStatus.READY}
    if _generation_tasks.get(video_id) == ProxyStatus.GENERATING:
        return {"status": ProxyStatus.GENERATING}
    
    _generation_tasks[video_id] = ProxyStatus.GENERATING
    background_tasks.add_task(_run_ffmpeg_proxy_gen, video_id)
    return {"status": ProxyStatus.GENERATING}

async def _run_ffmpeg_proxy_gen(video_id: str):
    """Background task: generate low-res proxy using FFmpeg."""
    try:
        input_path = get_original_path(video_id)
        output_path = get_proxy_path(video_id)
        # Example FFmpeg command: scale to 480p, low bitrate
        cmd = [
            "ffmpeg", "-i", str(input_path),
            "-vf", "scale=854:480", "-c:v", "libx264", "-preset", "fast",
            "-crf", "28", "-c:a", "aac", "-b:a", "96k",
            "-y", str(output_path)
        ]
        await asyncio.create_subprocess_exec(*cmd, stdout=asyncio.PIPE, stderr=asyncio.PIPE)
        # Mark as ready
        _generation_tasks[video_id] = ProxyStatus.READY
    except Exception as e:
        logger.error(f"Proxy generation failed for {video_id}: {e}")
        _generation_tasks[video_id] = ProxyStatus.FAILED

import { useEffect, useState } from 'react';
import { api } from '../services/api';

type ProxyStatus = 'missing' | 'generating' | 'ready' | 'failed';

export const VideoPreview: React.FC<{ videoId: string }> = ({ videoId }) => {
  const [proxyStatus, setProxyStatus] = useState<ProxyStatus>('missing');
  const [pollInterval, setPollInterval] = useState<number | null>(null);

  // Check initial status and trigger generation if missing
  useEffect(() => {
    const checkAndGenerate = async () => {
      const status = await api.getProxyStatus(videoId);
      if (status === 'missing') {
        // Trigger generation
        await api.generateProxy(videoId);
        setProxyStatus('generating');
        // Start polling every 2 seconds
        const interval = window.setInterval(async () => {
          const newStatus = await api.getProxyStatus(videoId);
          setProxyStatus(newStatus);
          if (newStatus === 'ready' || newStatus === 'failed') {
            clearInterval(interval);
            setPollInterval(null);
          }
        }, 2000);
        setPollInterval(interval);
      } else {
        setProxyStatus(status);
      }
    };
    checkAndGenerate();

    return () => {
      if (pollInterval) clearInterval(pollInterval);
    };
  }, [videoId]);

  // Badge rendering logic
  const badgeText = {
    missing: 'Proxy unavailable → generating...',
    generating: 'Generating proxy...',
    ready: 'Proxy playback',
    failed: 'Proxy generation failed',
  }[proxyStatus];

  return (
    <div className="video-preview">
      <div className="source-hint-badge">{badgeText}</div>
      {/* Video player uses original or proxy based on ready status */}
      <video src={proxyStatus === 'ready' ? getProxyUrl(videoId) : getOriginalUrl(videoId)} />
    </div>
  );
};
