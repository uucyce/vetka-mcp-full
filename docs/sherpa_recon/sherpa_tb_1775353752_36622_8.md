#!/bin/bash
# scripts/synapse_voice.sh — Qwen TTS voice notification generator
# Phase SYNAPSE-VOICE | MARKER_VOICE.1
#
# Usage: synapse_voice.sh "phrase text" [VOICE]
#   VOICE options: Vivian, Serena, Dylan, Eric, Ryan, etc. (default: Dylan)
#
# Generates WAV via Qwen TTS, caches in data/synapse_audio/, plays via afplay.

set -euo pipefail

PHRASE="${1:?Usage: synapse_voice.sh \"phrase text\" [VOICE]}"
VOICE="${2:-Dylan}"  # Default male voice, good for Russian/English mix
LOG_PREFIX="[SYNAPSE-VOICE]"

# ── Paths ────────────────────────────────────────────────────────────
SCRIPT_DIR="$(cd "$(dirname "${BASH_SOURCE[0]}")" && pwd)"
REPO_ROOT="$(cd "$SCRIPT_DIR/.." && pwd)"
AUDIO_DIR="$REPO_ROOT/data/synapse_audio"
CACHE_DIR="$AUDIO_DIR/cache"
mkdir -p "$CACHE_DIR"

# ── Cache key: MD5 of phrase + voice ─────────────────────────────────
CACHE_KEY=$(echo -n "${PHRASE}:${VOICE}" | md5 -q 2>/dev/null || echo -n "${PHRASE}:${VOICE}" | md5sum | cut -d' ' -f1)
CACHE_FILE="$CACHE_DIR/${CACHE_KEY}.wav"

# ── Check cache ──────────────────────────────────────────────────────
if [ -f "$CACHE_FILE" ]; then
    echo "$LOG_PREFIX Cache hit: $CACHE_FILE"
    afplay "$CACHE_FILE" &
    exit 0
fi

# ── Generate via Qwen TTS ───────────────────────────────────────────
echo "$LOG_PREFIX Generating TTS for: '$PHRASE' (voice: $VOICE)"

# Ensure qwen-tts Python environment exists
QWEN_ENV="${QWEN_TTS_ENV:-$REPO_ROOT/.venvs/qwen-tts}"
if [ ! -d "$QWEN_ENV" ]; then
    echo "$LOG_PREFIX Creating qwen-tts environment..."
    python3 -m venv "$QWEN_ENV"
    source "$QWEN_ENV/bin/activate"
    pip install -q qwen-tts soundfile
else
    source "$QWEN_ENV/bin/activate"
fi

# Generate TTS using Python inline script
python3 << EOF
import sys
import torch
import soundfile as sf
from pathlib import Path

try:
    from qwen_tts import Qwen3TTSModel
    
    # Use smaller 0.6B model for speed (or 1.7B for quality)
    model = Qwen3TTSModel.from_pretrained(
        "Qwen/Qwen3-TTS-12Hz-0.6B-CustomVoice",
        device_map="auto",
        torch_dtype=torch.bfloat16 if torch.cuda.is_available() else torch.float32,
    )
    
    # Auto-detect language (Russian phrases work fine)
    wavs, sr = model.generate_custom_voice(
        text="$PHRASE",
        language="auto",
        speaker="$VOICE",
    )
    
    output_path = Path("$CACHE_FILE")
    output_path.parent.mkdir(parents=True, exist_ok=True)
    sf.write(str(output_path), wavs[0], sr)
    print(f"TTS generated: {output_path}")
    
except Exception as e:
    print(f"TTS generation failed: {e}", file=sys.stderr)
    sys.exit(1)
EOF

if [ -f "$CACHE_FILE" ]; then
    echo "$LOG_PREFIX Playing: $CACHE_FILE"
    afplay "$CACHE_FILE" &
else
    echo "$LOG_PREFIX ERROR: Failed to generate audio" >&2
    # Fallback to ping if TTS fails
    afplay "$REPO_ROOT/data/Ping.aiff" 2>/dev/null || true
    exit 1
fi

# MARKER_VOICE.3: SYNAPSE-VOICE integration — TTS notifications
import hashlib
import subprocess
from pathlib import Path

# Pre-defined voice phrases map (Russian + English)
_VOICE_PHRASES = {
    # Task lifecycle
    "task_completed": {
        "ru": "Задача {task_id} завершена, требуется проверка",
        "en": "Task {task_id} completed, QA needed",
    },
    "task_verified": {
        "ru": "Задача {task_id} проверена — готова к мержу",
        "en": "Task {task_id} verified — ready to merge!",
    },
    "task_failed_qa": {
        "ru": "Внимание: задача {task_id} не прошла проверку",
        "en": "Alert: task {task_id} failed QA",
    },
    "task_claimed": {
        "ru": "Агент {agent} взял задачу {task_id}",
        "en": "Agent {agent} claimed task {task_id}",
    },
    # Agent lifecycle
    "agent_spawned": {
        "ru": "Агент {agent} активирован",
        "en": "Agent {agent} is online",
    },
    "agent_wake": {
        "ru": "Агент {agent} пробуждён",
        "en": "Agent {agent} awakened",
    },
}

def _synapse_voice(
    self,
    event_type: str,
    task_id: Optional[str] = None,
    agent: Optional[str] = None,
    language: str = "ru",  # Default Russian per user preference
    voice: str = "Dylan",
    async_play: bool = True,
) -> None:
    """
    Trigger voice notification via Qwen TTS.
    
    MARKER_VOICE.4: Async TTS playback — never blocks task completion
    """
    # Get phrase template
    templates = _VOICE_PHRASES.get(event_type)
    if not templates:
        return
    
    phrase_template = templates.get(language, templates.get("en", ""))
    phrase = phrase_template.format(task_id=task_id or "unknown", agent=agent or "unknown")
    
    # Build cache key
    cache_key = hashlib.md5(f"{phrase}:{voice}".encode()).hexdigest()
    
    # Paths
    repo_root = Path(__file__).resolve().parent.parent.parent
    audio_dir = repo_root / "data" / "synapse_audio"
    cache_file = audio_dir / "cache" / f"{cache_key}.wav"
    
    # Ensure audio directory exists
    cache_file.parent.mkdir(parents=True, exist_ok=True)
    
    # Check cache or generate
    if not cache_file.exists():
        # Generate via synapse_voice.sh (will cache)
        script_path = repo_root / "scripts" / "synapse_voice.sh"
        if script_path.exists():
            try:
                # Non-blocking generation call
                subprocess.Popen(
                    ["/bin/bash", str(script_path), phrase, voice],
                    stdout=subprocess.DEVNULL,
                    stderr=subprocess.DEVNULL,
                )
            except Exception as e:
                logger.warning(f"[SYNAPSE-VOICE] TTS generation failed: {e}")
        return
    
    # Cache hit — play immediately (async)
    if async_play:
        try:
            subprocess.Popen(
                ["afplay", str(cache_file)],
                stdout=subprocess.DEVNULL,
                stderr=subprocess.DEVNULL,
            )
        except Exception as e:
            logger.warning(f"[SYNAPSE-VOICE] Audio playback failed: {e}")
    else:
        # Sync fallback (rarely used)
        subprocess.run(["afplay", str(cache_file)], capture_output=True)

def _auto_notify(self, event_type: str, task: Dict[str, Any]) -> None:
    """
    MARKER_VOICE.5: Enhanced auto-notify with TTS voice alerts
    """
    # ── Existing wake notification (SYNAPSE-WAKE-CMD) ─────────────────
    wake_script = PROJECT_ROOT / "scripts" / "synapse_wake.sh"
    if wake_script.exists():
        try:
            role = task.get("assigned_to", "unknown")
            subprocess.Popen(
                ["/bin/bash", str(wake_script), role],
                stdout=subprocess.DEVNULL,
                stderr=subprocess.DEVNULL,
            )
        except Exception as e:
            logger.warning(f"[SYNAPSE-WAKE] Wake notification failed: {e}")
    
    # ── NEW: Voice notification (SYNAPSE-VOICE) ─────────────────────
    # Trigger on specific events only to avoid noise
    voice_events = {"task_completed", "task_verified", "task_failed_qa"}
    if event_type in voice_events:
        self._synapse_voice(
            event_type=event_type,
            task_id=task.get("id"),
            agent=task.get("assigned_to"),
            language="ru",  # Russian as requested
            voice="Dylan",   # Clear male voice, handles Russian well
            async_play=True, # NEVER blocks
        )

    ping_file = PROJECT_ROOT / "data" / "Ping.aiff"

        try:
            subprocess.Popen(
                ["afplay", str(ping_file)],

7. Key Design Decisions
Table
Aspect	Decision	Rationale
TTS Backend	Local qwen-tts package	Offline capability, no API costs, low latency 
Model Size	0.6B (default)	Fast inference; upgrade to 1.7B for quality if GPU available 
Voice	Dylan (default)	Male voice optimized for Russian/English bilingual 
Caching	MD5(phrase+voice) → WAV	Avoids regeneration, instant playback on cache hit
Async	subprocess.Popen	Never blocks task completion pipeline
Language	Russian default	Per user preference; English fallback available
