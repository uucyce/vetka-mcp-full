"""
MARKER_198.DEBRIEF.FIX3: Worktree synchronization utility.

Ensures agent worktrees are up-to-date with main before task execution.
Prevents 'file exists in main but not in worktree' drift errors.
"""

from __future__ import annotations

import logging
import subprocess
from pathlib import Path
from typing import Optional

logger = logging.getLogger(__name__)


class WorktreeDriftError(Exception):
    """Raised when worktree cannot be synced due to divergence."""
    pass


def detect_worktree_root(path: Path) -> Optional[Path]:
    """Detect if path is inside a git worktree, return the worktree root."""
    try:
        result = subprocess.run(
            ["git", "rev-parse", "--show-toplevel"],
            cwd=path,
            capture_output=True,
            text=True,
            check=True
        )
        wt_root = Path(result.stdout.strip())
        # Check if it's actually a worktree (not main repo)
        git_dir = subprocess.run(
            ["git", "rev-parse", "--git-dir"],
            cwd=path,
            capture_output=True,
            text=True,
            check=True
        )
        if ".claude/worktrees" in git_dir.stdout:
            return wt_root
    except (subprocess.CalledProcessError, FileNotFoundError):
        pass
    return None


def sync_worktree(worktree_path: Path, main_branch: str = "main") -> None:
    """
    Fast-forward worktree to include latest main commits.
    
    Strategy:
    1. Fetch origin/main
    2. Attempt fast-forward merge
    3. If FF fails, raise (manual intervention needed)
    """
    logger.info("[WorktreeSync] Syncing %s with origin/%s", worktree_path, main_branch)
    
    # Fetch
    subprocess.run(
        ["git", "fetch", "origin", main_branch],
        cwd=worktree_path,
        check=True,
        capture_output=True
    )
    
    # Try fast-forward
    result = subprocess.run(
        ["git", "merge", "--ff-only", f"origin/{main_branch}"],
        cwd=worktree_path,
        capture_output=True,
        text=True
    )
    
    if result.returncode != 0:
        raise WorktreeDriftError(
            f"Worktree {worktree_path} has diverged from {main_branch}. "
            f"Manual rebase required. Error: {result.stderr}"
        )
    
    logger.info("[WorktreeSync] Sync complete: %s", result.stdout.strip() or "Already up to date")


def ensure_file_available(file_path: str, fallback_ref: str = "main") -> None:
    """
    Emergency checkout: if file missing, try to checkout from main.
    Use sparingly — indicates sync failure.
    """
    if Path(file_path).exists():
        return
    
    logger.warning("[WorktreeSync] File missing: %s, attempting checkout from %s", file_path, fallback_ref)
    
    result = subprocess.run(
        ["git", "checkout", fallback_ref, "--", file_path],
        capture_output=True,
        text=True
    )
    
    if result.returncode != 0:
        raise WorktreeDriftError(
            f"Cannot checkout {file_path} from {fallback_ref}: {result.stderr}"
        )
