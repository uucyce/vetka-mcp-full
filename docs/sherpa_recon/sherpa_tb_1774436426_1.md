from enum import Enum
from typing import Optional, List, Dict, Any
from dataclasses import dataclass, field
from uuid import UUID, uuid4


class NodeType(Enum):
    """Core node types in CUT scene graph"""
    # Structural
    PROJECT = "project"
    TIMELINE = "timeline"
    
    # Script Spine
    SCENE_CHUNK = "scene_chunk"
    
    # Lore Graph (W7 Addition)
    LORE_CHARACTER = "lore_character"
    LORE_LOCATION = "lore_location"
    LORE_PROP = "lore_prop"           # object
    LORE_THEME = "lore_theme"
    LORE_MOTIF = "lore_motif"
    
    # Media Graph
    MEDIA = "media"
    SOURCE_FILE = "source_file"
    CLIP_USAGE = "clip_usage"
    
    # Analysis & Meta
    ANALYSIS = "analysis"
    MARKER = "marker"
    FEEDBACK_EVENT = "feedback_event"


class EdgeType(Enum):
    """Edge types defining relationships between nodes"""
    # Structural

    # Narrative (Script Spine)

LORE_NODE_TYPES = {
    NodeType.LORE_CHARACTER: "character",
    NodeType.LORE_LOCATION: "location",
    NodeType.LORE_PROP: "object",
    NodeType.LORE_THEME: "theme",
    NodeType.LORE_MOTIF: "motif",
}


def is_lore_node(node_type: NodeType) -> bool:
    """Check if node type is a LoreNode variant"""
    return node_type in LORE_NODE_TYPES


def get_lore_type(node_type: NodeType) -> Optional[str]:
    """Get string lore type from NodeType enum"""
    return LORE_NODE_TYPES.get(node_type)


# Edge validation rules for Lore relationships
LORE_EDGE_RULES = {
    EdgeType.MENTIONS: {
        "from": ["scene_chunk"],
        "to": ["lore_character", "lore_location", "lore_prop", "lore_theme", "lore_motif"],
    },
    EdgeType.RELATES_TO: {
        "from": ["lore_character", "lore_location", "lore_prop", "lore_theme", "lore_motif"],
        "to": ["lore_character", "lore_location", "lore_prop", "lore_theme", "lore_motif"],
    },
    EdgeType.APPEARS_IN: {
        "from": ["lore_character", "lore_prop"],
        "to": ["scene_chunk"],
    },
    EdgeType.LOCATED_AT: {
        "from": ["lore_character", "lore_prop"],
        "to": ["lore_location"],
    },
}


def validate_lore_edge(edge_type: EdgeType, from_type: NodeType, to_type: NodeType) -> bool:
    """Validate if a lore edge is allowed between node types"""
    rules = LORE_EDGE_RULES.get(edge_type)
    if not rules:
        return True  # No specific rules, allow by default
    
    from_allowed = rules["from"]
    to_allowed = rules["to"]
    
    from_type_str = from_type.value
    to_type_str = to_type.value
    
    return from_type_str in from_allowed and to_type_str in to_allowed

export const useLoreNodes = () => {
  const [loreNodes, setLoreNodes] = useState<LoreNodeData[]>([]);
  const [loreEdges, setLoreEdges] = useState<LoreEdge[]>([]);
  
  const addLoreNode = useCallback((node: Omit<LoreNodeData, 'lore_id'>) => {
    const newNode: LoreNodeData = {
      ...node,
      lore_id: crypto.randomUUID(),
    };
    setLoreNodes(prev => [...prev, newNode]);
    return newNode;
  }, []);
  
  const addLoreEdge = useCallback((edge: Omit<LoreEdge, 'edge_id'>) => {
    const newEdge: LoreEdge = {
      ...edge,
      edge_id: crypto.randomUUID(),
    };
    setLoreEdges(prev => [...prev, newEdge]);
    return newEdge;
  }, []);
  
  const getNodeRelationships = useCallback((nodeId: string) => {
    return loreEdges.filter(
      e => e.source_id === nodeId || e.target_id === nodeId
    );
  }, [loreEdges]);
  
  return {
    loreNodes,
    loreEdges,
    addLoreNode,
    addLoreEdge,
    getNodeRelationships,
    setLoreNodes,
    setLoreEdges,
  };
};
