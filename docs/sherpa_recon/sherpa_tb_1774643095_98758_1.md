def extract_layer_from_depth_band(
    depth_map: np.ndarray,
    depth_band: Tuple[float, float],
    bbox: Optional[Tuple[int, int, int, int]] = None,
    semantic_mask: Optional[np.ndarray] = None,
) -> LayerExtractionResult:
    """
    Extract layer using depth band + optional hints.
    Bbox is hint-only, not mask authority.
    """
    # Build depth-based candidate
    depth_candidate = (depth_map >= depth_band[0]) & (depth_map <= depth_band[1])
    
    # Apply bbox as SEARCH REGION, not mask
    if bbox is not None:
        x1, y1, x2, y2 = bbox
        # Dilate bbox by 15% for search margin
        h, w = depth_map.shape
        margin_x = int((x2 - x1) * 0.15)
        margin_y = int((y2 - y1) * 0.15)
        search_region = np.zeros_like(depth_candidate, dtype=bool)
        search_region[
            max(0, y1 - margin_y):min(h, y2 + margin_y),
            max(0, x1 - margin_x):min(w, x2 + margin_x)
        ] = True
        candidate_mask = depth_candidate & search_region
        is_provisional = semantic_mask is None
        mask_method = "depth_band+bbox_hint" if is_provisional else "semantic+depth"
    else:
        candidate_mask = depth_candidate
        is_provisional = semantic_mask is not None
        mask_method = "depth_band_only"
    
    # If semantic mask available, use it as authority
    if semantic_mask is not None:
        # Refine: semantic mask constrained by depth band
        final_mask = semantic_mask & depth_candidate
        is_provisional = False
        mask_method = "semantic_authority+depth_constraint"
    else:
        final_mask = candidate_mask
        # Apply morphological cleanup for provisional masks
        final_mask = morphological_cleanup(final_mask, iterations=2)
        is_provisional = True
    
    # Quality scoring
    quality_score = calculate_mask_quality(final_mask, depth_candidate, bbox)
    
    return LayerExtractionResult(
        mask=final_mask,
        depth_band=depth_band,
        is_provisional=is_provisional,
        mask_method=mask_method,
        quality_score=quality_score
    )

def extract_background_layer(
    depth_map: np.ndarray,
    foreground_masks: List[np.ndarray],
    depth_far_threshold: float = 0.85,  # Top 15% depth values
) -> LayerExtractionResult:
    """
    Extract background using depth-based far plane, NOT complement.
    Complement only fills gaps where depth indicates far plane.
    """
    # Primary: depth-based far plane
    far_plane_mask = depth_map >= depth_far_threshold
    
    # Secondary: complement ONLY for gap filling in far depth regions
    all_foreground = np.any(foreground_masks, axis=0) if foreground_masks else np.zeros_like(depth_map, dtype=bool)
    gap_region = ~all_foreground
    
    # Only fill gaps that are also in far depth plane
    gap_fill = gap_region & far_plane_mask
    
    final_mask = far_plane_mask | gap_fill
    
    # Mark as provisional if significant gap fill used
    gap_ratio = np.sum(gap_fill) / (np.sum(final_mask) + 1e-6)
    is_provisional = gap_ratio > 0.3  # >30% gap fill = provisional
    
    return LayerExtractionResult(
        mask=final_mask,
        depth_band=(depth_far_threshold, 1.0),
        is_provisional=is_provisional,
        mask_method="depth_far_plane+gap_fill" if gap_ratio > 0 else "depth_far_plane",
        quality_score=1.0 - gap_ratio
    )
