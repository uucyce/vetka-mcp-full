// New component structure
type InspectorSection = 'transform' | 'motion' | 'matte' | 'groups';

const Inspector: React.FC = () => {
  const [expanded, setExpanded] = useState<Set<InspectorSection>>(new Set(['transform']));
  
  // Only show relevant sections based on stageTool
  const visibleSections = useMemo(() => {
    switch (stageTool) {
      case 'brush': return ['transform', 'matte'];
      case 'group': return ['transform', 'groups'];
      case 'matte': return ['transform', 'matte'];
      default: return ['transform', 'motion'];
    }
  }, [stageTool]);
  
  return (
    <div className="inspector">
      {visibleSections.map(section => (
        <CollapsibleSection 
          key={section}
          title={sectionLabels[section]}
          expanded={expanded.has(section)}
          onToggle={() => toggleSection(section)}
        >
          {sectionContent[section]}
        </CollapsibleSection>
      ))}
    </div>
  );
};
