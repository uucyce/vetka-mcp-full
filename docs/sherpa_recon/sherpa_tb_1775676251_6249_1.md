MAX_ROLE_MEMORY_TOKENS = 500  # hard cap for haiku roles, architecture requirement

def _cap_role_memory(role_memory: Dict[str, Any], max_tokens: int = MAX_ROLE_MEMORY_TOKENS) -> Dict[str, Any]:
    """Truncate role_memory content to max_tokens (rough estimate)."""
    if not role_memory or "content" not in role_memory:
        return role_memory
    
    content = role_memory.get("content", "")
    # Rough token estimate: 1 token ≈ 4 chars for English
    estimated_tokens = len(content) // 4
    if estimated_tokens <= max_tokens:
        return role_memory
    
    # Truncate to max_tokens * 4 chars
    truncated = content[:max_tokens * 4]
    role_memory["content"] = truncated + "\n… [truncated due to token budget]"
    role_memory["truncated"] = True
    logger.info(f"[TokenBudget] role_memory truncated from ~{estimated_tokens} to {max_tokens} tokens")
    return role_memory
