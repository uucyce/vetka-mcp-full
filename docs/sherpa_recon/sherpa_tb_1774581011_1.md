        if not current_file.exists():
            return True

import hashlib
import shutil
from pathlib import Path
from datetime import datetime, timezone
from typing import Optional

def _file_provenance(file_path: Path) -> Optional[dict]:
    """Return SHA256, mtime, size for a file. Returns None if file missing."""
    if not file_path.is_file():
        return None
    
    sha256 = hashlib.sha256()
    with open(file_path, 'rb') as f:
        for chunk in iter(lambda: f.read(65536), b''):
            sha256.update(chunk)
    
    stat = file_path.stat()
    return {
        "path": str(file_path.absolute()),
        "hash": sha256.hexdigest(),
        "mtime": datetime.fromtimestamp(stat.st_mtime, tz=timezone.utc).isoformat(),
        "size_bytes": stat.st_size
    }

def _snapshot_critical_files(export_root: Path, output_dir: Path) -> Optional[Path]:
    """Copy critical JSON files to snapshot directory. Returns snapshot path or None."""
    critical = [
        "plate_layout.json", 
        "plate_export_manifest.json", 
        "plate_stack.json", 
        "plate_export_job_state.json"
    ]
    
    snapshot_dir = output_dir / "_provenance_snapshot"
    snapshot_dir.mkdir(parents=True, exist_ok=True)
    
    copied_any = False
    for filename in critical:
        src = export_root / filename
        if src.exists():
            shutil.copy2(src, snapshot_dir / filename)
            copied_any = True
    
    return snapshot_dir if copied_any else None

def _capture_provenance(export_root: Path, output_dir: Path) -> dict:
    """Capture immutable provenance for export root files."""
    provenance = {
        "export_root": str(export_root.absolute()),
        "captured_at": datetime.now(timezone.utc).isoformat(),
        "files": {},
        "snapshot_dir": None
    }
    
    # Capture each critical file
    for filename in ["plate_layout.json", "plate_export_manifest.json", 
                     "plate_stack.json", "plate_export_job_state.json"]:
        file_path = export_root / filename
        provenance_info = _file_provenance(file_path)
        if provenance_info:
            provenance["files"][filename] = provenance_info
    
    # Add warning if layout exists but manifest missing
    if "plate_layout.json" in provenance["files"] and "plate_export_manifest.json" not in provenance["files"]:
        provenance["warning"] = "manifest missing at render time (partial export state)"
    
    # Create snapshot
    snapshot_path = _snapshot_critical_files(export_root, output_dir)
    if snapshot_path:
        provenance["snapshot_dir"] = str(snapshot_path.relative_to(output_dir))
    
    return provenance
