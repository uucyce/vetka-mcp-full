import hashlib
import shutil
from pathlib import Path
from datetime import datetime, timezone

def file_provenance(file_path: Path) -> dict | None:
    """Return SHA256, mtime (ISO), and absolute path for a file."""
    if not file_path.is_file():
        return None
    sha256 = hashlib.sha256()
    with open(file_path, 'rb') as f:
        for chunk in iter(lambda: f.read(65536), b''):
            sha256.update(chunk)
    mtime = datetime.fromtimestamp(file_path.stat().st_mtime, tz=timezone.utc).isoformat()
    return {
        "path": str(file_path.absolute()),
        "hash": sha256.hexdigest(),
        "mtime": mtime
    }

def snapshot_file(src: Path, dst_dir: Path) -> Path | None:
    """Copy file to snapshot directory, preserving name. Returns dest path or None."""
    if not src.is_file():
        return None
    dst_dir.mkdir(parents=True, exist_ok=True)
    dst = dst_dir / src.name
    shutil.copy2(src, dst)  # preserves metadata
    return dst
