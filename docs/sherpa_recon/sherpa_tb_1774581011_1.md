2. Approach

The current report records mutable file paths without any content verification. When export root files are overwritten (as happened with hover-politsia on 2026-03-20), the render report becomes misleading — it points to different content than what was actually rendered.

Solution: Three-layer hardening:

Minimum (SHA256 + mtime): Record immutable identifiers for each source file

Better (snapshot): Copy actual files used into render output

Best (content-addressed naming): Use hashes in snapshot filenames

def add_provenance_to_report(export_root: Path, output_dir: Path) -> dict:
    """
    Capture immutable provenance for layout and manifest files.
    
    Returns provenance dict that gets merged into report.json.
    """
    provenance = {
        "export_root": str(export_root.absolute()),
        "captured_at": datetime.now(timezone.utc).isoformat(),
        "files": {},
        "snapshot_dir": None
    }
    
    # Critical files that define export state
    critical_files = [
        "plate_layout.json",
        "plate_export_manifest.json",
        "plate_stack.json",  # optional but useful
        "plate_export_job_state.json"  # optional
    ]
    
    for filename in critical_files:
        file_path = export_root / filename
        if file_path.exists():
            provenance["files"][filename] = file_provenance(file_path)
    
    # Optional: snapshot entire critical set
    snapshot_root = output_dir / "_provenance_snapshot"
    for filename in critical_files:
        src = export_root / filename
        if src.exists():
            shutil.copy2(src, snapshot_root / filename)
    
    if any(snapshot_root.iterdir()):
        provenance["snapshot_dir"] = str(snapshot_root.relative_to(output_dir))
    
    return provenance

def file_provenance(file_path: Path) -> dict:
    """Return SHA256, mtime (ISO), and size for a file."""
    sha256 = hashlib.sha256()
    with open(file_path, 'rb') as f:
        for chunk in iter(lambda: f.read(65536), b''):
            sha256.update(chunk)
    
    stat = file_path.stat()
    return {
        "path": str(file_path.absolute()),
        "hash": sha256.hexdigest(),
        "mtime": datetime.fromtimestamp(stat.st_mtime, tz=timezone.utc).isoformat(),
        "size_bytes": stat.st_size
    }

import hashlib
import shutil
from pathlib import Path
from datetime import datetime, timezone

def _file_provenance(file_path: Path) -> dict | None:
    """Return SHA256, mtime, size for a file. Returns None if file missing."""
    if not file_path.is_file():
        return None
    
    sha256 = hashlib.sha256()
    with open(file_path, 'rb') as f:
        for chunk in iter(lambda: f.read(65536), b''):
            sha256.update(chunk)
    
    stat = file_path.stat()
    return {
        "path": str(file_path.absolute()),
        "hash": sha256.hexdigest(),
        "mtime": datetime.fromtimestamp(stat.st_mtime, tz=timezone.utc).isoformat(),
        "size_bytes": stat.st_size
    }

def _snapshot_critical_files(export_root: Path, output_dir: Path) -> Path | None:
    """Copy critical JSON files to snapshot directory. Returns snapshot path or None."""
    critical = ["plate_layout.json", "plate_export_manifest.json", 
                "plate_stack.json", "plate_export_job_state.json"]
    
    snapshot_dir = output_dir / "_provenance_snapshot"
    snapshot_dir.mkdir(parents=True, exist_ok=True)
    
    copied_any = False
    for filename in critical:
        src = export_root / filename
        if src.exists():
            shutil.copy2(src, snapshot_dir / filename)
            copied_any = True
    
    return snapshot_dir if copied_any else None

def _capture_provenance(export_root: Path, output_dir: Path) -> dict:
    """Capture immutable provenance for export root files."""
    provenance = {
        "export_root": str(export_root.absolute()),
        "captured_at": datetime.now(timezone.utc).isoformat(),
        "files": {},
        "snapshot_dir": None
    }
    
    # Capture each critical file
    for filename in ["plate_layout.json", "plate_export_manifest.json", 
                     "plate_stack.json", "plate_export_job_state.json"]:
        file_path = export_root / filename
        if file_path.exists():
            provenance["files"][filename] = _file_provenance(file_path)
    
    # Create snapshot
    snapshot_path = _snapshot_critical_files(export_root, output_dir)
    if snapshot_path:
        provenance["snapshot_dir"] = str(snapshot_path.relative_to(output_dir))
    
    return provenance
4. Risks and Edge Cases
Risk	Mitigation
Large snapshot size	Only snapshot JSON metadata files (few KB), not PNG assets (tens of MB)
Export root missing files	Graceful degradation: log warning but continue render
mtime changes after render	Capture at render start, not after
Cross-platform path issues	Use pathlib.Path and relative paths for snapshot_dir
Backward compatibility	Old reports without provenance field still readable; new field is additive
Snapshot directory clutter	Each render gets its own snapshot; can be cleaned by normal output rotation
