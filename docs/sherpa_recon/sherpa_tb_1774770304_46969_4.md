const [localFile, setLocalFile] = useState<{ url: string; name: string } | null>(null);

// Add handler function
const handleLocalFileImport = (e: React.ChangeEvent<HTMLInputElement>) => {
  const file = e.target.files?.[0];
  if (!file || !file.type.startsWith('image/')) return;

  // Revoke previous blob URL to prevent memory leak
  if (localFile?.url) {
    URL.revokeObjectURL(localFile.url);
  }

  const objectUrl = URL.createObjectURL(file);
  setLocalFile({ url: objectUrl, name: file.name });
  
  // Set as active source - triggers existing raster hydration flow
  setSampleId('__local__');
  setSourceRasterReady(false);
  
  // Hydrate the source raster from the local file
  const img = new Image();
  img.onload = () => {
    setSourceSize({ width: img.naturalWidth, height: img.naturalHeight });
    setSourceRasterReady(true);
  };
  img.src = objectUrl;
};

// Cleanup on unmount
useEffect(() => {
  return () => {
    if (localFile?.url) {
      URL.revokeObjectURL(localFile.url);
    }
  };
}, [localFile]);
