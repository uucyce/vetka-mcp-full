const pollJob = async (jobId: string, onComplete: (result: any) => void, onError: (err: Error) => void) => {
  const interval = setInterval(async () => {
    try {
      const res = await fetch(`${API_BASE}/api/cut/import/job/${jobId}`);
      const data = await res.json();
      
      if (data.state === 'completed') {
        clearInterval(interval);
        // CRITICAL: Check success field even on 'completed' state
        if (data.success === false) {
          onError(new Error(data.message || `Job ${jobId} failed with success=false`));
        } else {
          onComplete(data.result);
        }
      } else if (data.state === 'error') {
        clearInterval(interval);
        onError(new Error(data.message || `Job ${jobId} failed`));
      }
    } catch (err) {
      clearInterval(interval);
      onError(err instanceof Error ? err : new Error('Polling error'));
    }
  }, 1000);
};
