import subprocess

def count_commits(branch: str, base: str = 'main') -> int:
    """
    Count commits between base branch and target branch.
    
    Args:
        branch: The branch to count commits on
        base: The base branch to compare against (default: 'main')
    
    Returns:
        Number of commits ahead of base
    """
    try:
        result = subprocess.run(
            ['git', 'rev-list', '--count', f'{base}..{branch}'],
            capture_output=True,
            text=True,
            check=True,
            cwd=Path(__file__).parent.parent.parent  # Project root
        )
        return int(result.stdout.strip())
    except subprocess.CalledProcessError as e:
        raise RuntimeError(f"Failed to count commits: {e.stderr}") from e


def require_integration_tests(branch: str) -> None:
    """
    Require full integration test suite to pass before large merge.
    
    Raises:
        ValidationError: If integration tests fail or are missing
    """
    # Check for integration test marker
    test_result = subprocess.run(
        ['python', '-m', 'pytest', 'tests/integration/', '-v', '--tb=short'],
        capture_output=True,
        text=True,
        cwd=Path(__file__).parent.parent.parent
    )
    
    if test_result.returncode != 0:
        raise ValidationError(
            f"Integration tests failed for branch '{branch}':\n{test_result.stdout}\n{test_result.stderr}"
        )
    
    # Check for specific large-merge test markers
    marker_check = subprocess.run(
        ['grep', '-r', 'LARGE_MERGE', 'tests/'],
        capture_output=True,
        text=True
    )
    if marker_check.returncode != 0:
        raise ValidationError(
            f"No LARGE_MERGE test markers found for branch '{branch}'. "
            "Large merges require dedicated integration tests."
        )


def require_no_force_push(force: bool = False) -> None:
    """
    Prevent force pushes for large merges.
    
    Raises:
        ValidationError: If force=True is passed for large merge
    """
    if force:
        raise ValidationError(
            "Force push is not allowed for large merges (>20 commits). "
            "Resolve conflicts manually and retry without force=True."
        )


def validate_hook_syntax(branch: str) -> None:
    """
    Validate generate_claude_md.py hooks for syntax errors.
    
    Raises:
        ValidationError: If hook syntax errors detected
    """
    project_root = Path(__file__).parent.parent.parent
    hook_file = project_root / 'scripts' / 'generate_claude_md.py'
    
    if not hook_file.exists():
        raise ValidationError(f"Hook file not found: {hook_file}")
    
    # Syntax check using Python's compile
    try:
        with open(hook_file, 'r') as f:
            compile(f.read(), str(hook_file), 'exec')
    except SyntaxError as e:
        raise ValidationError(f"Hook syntax error in {hook_file}: {e}") from e
    
    # Check for common hook patterns in branch
    result = subprocess.run(
        ['git', 'diff', f'main..{branch}', '--name-only'],
        capture_output=True,
        text=True
    )
    
    changed_files = result.stdout.strip().split('\n')
    hook_related = [f for f in changed_files if 'hook' in f.lower() or 'claude_md' in f.lower()]
    
    if hook_related:
        # Run hook in dry-run mode if available
        dry_run = subprocess.run(
            ['python', str(hook_file), '--dry-run'],
            capture_output=True,
            text=True
        )
        if dry_run.returncode != 0:
            raise ValidationError(f"Hook dry-run failed: {dry_run.stderr}")


def validate_enum_registrations(branch: str) -> None:
    """
    Validate MCP strategy enum registrations are complete.
    
    Raises:
        ValidationError: If enum registrations are missing or inconsistent
    """
    project_root = Path(__file__).parent.parent.parent
    
    # Check MCP strategy files
    mcp_files = list(project_root.glob('src/mcp/**/*.py'))
    
    for mcp_file in mcp_files:
        content = mcp_file.read_text()
        
        # Check for enum definitions without registrations
        enum_pattern = r'class\\s+(\\w+)(?:Enum|Strategy).*?:'
        enums = re.findall(enum_pattern, content)
        
        for enum_name in enums:
            # Each enum should have a corresponding registration
            registration_pattern = rf'register.*{enum_name}|{enum_name}.*register'
            if not re.search(registration_pattern, content, re.IGNORECASE):
                # Check if it's in a separate registry file
                registry_files = list(project_root.glob('src/**/registry*.py'))
                found_in_registry = False
                
                for reg_file in registry_files:
                    reg_content = reg_file.read_text()
                    if enum_name in reg_content:
                        found_in_registry = True
                        break
                
                if not found_in_registry:
                    raise ValidationError(
                        f"Enum '{enum_name}' in {mcp_file} has no registration. "
                        "All MCP strategy enums must be registered."
                    )


def validate_variable_changes(branch: str) -> None:
    """
    Flag dangerous variable deletions like _qa_skipped.
    
    Raises:
        ValidationError: If critical variables are deleted
    """
    result = subprocess.run(
        ['git', 'diff', f'main..{branch}', '-U0'],
        capture_output=True,
        text=True
    )
    
    diff = result.stdout
    
    # Critical variables that should not be deleted without review
    critical_vars = [
        '_qa_skipped',
        '_validation_bypassed',
        'FORCE_MERGE',
        'SKIP_HOOKS',
        'bypass_checks'
    ]
    
    for var in critical_vars:
        # Check for deletion pattern (lines starting with -)
        deletion_pattern = rf'^-\\s*.*\\b{re.escape(var)}\\b'
        if re.search(deletion_pattern, diff, re.MULTILINE):
            raise ValidationError(
                f"Critical variable '{var}' was deleted in branch '{branch}'. "
                "This requires manual security review before merge."
            )
    
    # Check for suspicious pattern: variable renamed but logic preserved
    for var in critical_vars:
        # Check if variable was renamed (deleted in one place, added in another)
        deleted = bool(re.search(rf'^-\\s*.*\\b{re.escape(var)}\\b', diff, re.MULTILINE))
        added = bool(re.search(rf'^\\+\\s*.*\\b{re.escape(var)}\\b', diff, re.MULTILINE))
        
        if deleted and not added:
            raise ValidationError(
                f"Variable '{var}' was removed without replacement. "
                "Verify this is intentional and safe."
            )

    """Raised when pre-flight validation fails."""
    pass

def merge_request(
    self,
    task_id: str,
    branch_name: Optional[str] = None,
    force: bool = False,
    skip_validation: bool = False
) -> Dict[str, Any]:
    """
    Request merge of a branch to main with pre-flight validation.
    
    Args:
        task_id: The task identifier
        branch_name: Branch to merge (defaults to task's branch)
        force: Allow force merge (not allowed for large merges)
        skip_validation: Skip validation checks (emergency only)
    
    Returns:
        Merge result with status and details
    
    Raises:
        ValidationError: If pre-flight checks fail
        TaskNotFoundError: If task doesn't exist
    """
    task = self.get_task(task_id)
    if not task:
        raise TaskNotFoundError(f"Task {task_id} not found")
    
    branch = branch_name or task.get('branch', f"task/{task_id}")
    
    # ── Pre-flight Validation ───────────────────────────────────────
    if not skip_validation:
        commit_count = count_commits(branch, 'main')
        
        # Large merge threshold: 20 commits
        if commit_count > 20:
            self._logger.info(f"Large merge detected: {commit_count} commits")
            
            # Check 1: Integration tests must pass
            try:
                require_integration_tests(branch)
            except ValidationError as e:
                return {
                    'success': False,
                    'error': 'INTEGRATION_TESTS_FAILED',
                    'message': str(e),
                    'commits': commit_count
                }
            
            # Check 2: No force push allowed
            try:
                require_no_force_push(force)
            except ValidationError as e:
                return {
                    'success': False,
                    'error': 'FORCE_PUSH_NOT_ALLOWED',
                    'message': str(e),
                    'commits': commit_count
                }
            
            # Check 3: Hook validation
            try:
                validate_hook_syntax(branch)
            except ValidationError as e:
                return {
                    'success': False,
                    'error': 'HOOK_VALIDATION_FAILED',
                    'message': str(e),
                    'commits': commit_count
                }
            
            # Check 4: Enum registration validation
            try:
                validate_enum_registrations(branch)
            except ValidationError as e:
                return {
                    'success': False,
                    'error': 'ENUM_VALIDATION_FAILED',
                    'message': str(e),
                    'commits': commit_count
                }
            
            # Check 5: Variable change validation
            try:
                validate_variable_changes(branch)
            except ValidationError as e:
                return {
                    'success': False,
                    'error': 'VARIABLE_VALIDATION_FAILED',
                    'message': str(e),
                    'commits': commit_count
                }
        
        # Extra-large merge threshold: 50 commits
        if commit_count > 50:
            self._logger.warning(f"Extra-large merge: {commit_count} commits - requiring additional review")
            
            # Require explicit approval marker in task
            if not task.get('extra', {}).get('large_merge_approved'):
                return {
                    'success': False,
                    'error': 'LARGE_MERGE_APPROVAL_REQUIRED',
                    'message': f'Merge of {commit_count} commits requires explicit approval. '
                              'Set large_merge_approved=true in task extra field.',
                    'commits': commit_count
                }
    
    # ── Execute Merge ────────────────────────────────────────────────
    try:
        merge_result = self._execute_merge(branch, force=force)
        
        # Post-merge validation
        if commit_count > 20:
            post_merge_errors = self._run_post_merge_validation(branch)
            if post_merge_errors:
                # Rollback on post-merge failure
                self._rollback_merge(branch)
                return {
                    'success': False,
                    'error': 'POST_MERGE_VALIDATION_FAILED',
                    'message': post_merge_errors,
                    'commits': commit_count
                }
        
        # Update task status
        task['status'] = 'merged'
        task['completed_at'] = datetime.now().isoformat()
        task['commit_hash'] = merge_result.get('merge_commit')
        self._save_task(task)
        
        return {
            'success': True,
            'merge_commit': merge_result.get('merge_commit'),
            'commits_merged': commit_count,
            'validation_passed': not skip_validation
        }
        
    except Exception as e:
        return {
            'success': False,
            'error': 'MERGE_FAILED',
            'message': str(e),
            'commits': commit_count if 'commit_count' in locals() else 'unknown'
        }


def _run_post_merge_validation(self, branch: str) -> Optional[str]:
    """
    Run post-merge validation checks.
    
    Returns:
        Error message if validation fails, None otherwise
    """
    errors = []
    
    # Validate enum registrations again post-merge
    try:
        validate_enum_registrations('main')
    except ValidationError as e:
        errors.append(f"Enum validation: {e}")
    
    # Validate hook syntax post-merge
    try:
        validate_hook_syntax('main')
    except ValidationError as e:
        errors.append(f"Hook validation: {e}")
    
    # Validate variable changes post-merge
    try:
        validate_variable_changes('main')
    except ValidationError as e:
        errors.append(f"Variable validation: {e}")
    
    return '\\n'.join(errors) if errors else None


def _rollback_merge(self, branch: str) -> None:
    """Rollback a failed merge."""
    subprocess.run(
        ['git', 'reset', '--hard', 'HEAD~1'],
        check=True,
        capture_output=True
    )

def promote_to_main(
    self,
    task_id: str,
    branch: Optional[str] = None,
    force: bool = False,
    skip_validation: bool = False
) -> Dict[str, Any]:
    """
    Promote task branch to main with large-merge validation.
    
    This is the canonical entry point for merging work to main.
    Includes Wave 3 regression prevention via pre-flight checks.
    
    Args:
        task_id: Task identifier
        branch: Source branch (defaults to task branch)
        force: Force promotion (blocked for large merges)
        skip_validation: Skip all validation (emergency only)
    
    Returns:
        Promotion result
    """
    task = self.get_task(task_id)
    if not task:
        raise TaskNotFoundError(f"Task {task_id} not found")
    
    source_branch = branch or task.get('branch', f"task/{task_id}")
    
    # ── Large Merge Pre-flight ───────────────────────────────────────
    if not skip_validation:
        commit_count = count_commits(source_branch, 'main')
        
        if commit_count > 20:
            self._logger.info(f"PROMOTE_GUARD: Large merge ({commit_count} commits) for {task_id}")
            
            # Requirement 1: Integration tests
            try:
                require_integration_tests(source_branch)
            except ValidationError as e:
                return {
                    'success': False,
                    'error': 'PROMOTE_GUARD_INTEGRATION_FAILED',
                    'message': str(e),
                    'task_id': task_id,
                    'commits': commit_count,
                    'guard': 'pre-flight'
                }
            
            # Requirement 2: No force push
            try:
                require_no_force_push(force)
            except ValidationError as e:
                return {
                    'success': False,
                    'error': 'PROMOTE_GUARD_FORCE_BLOCKED',
                    'message': str(e),
                    'task_id': task_id,
                    'commits': commit_count,
                    'guard': 'pre-flight'
                }
            
            # Requirement 3: Hook validation
            try:
                validate_hook_syntax(source_branch)
            except ValidationError as e:
                return {
                    'success': False,
                    'error': 'PROMOTE_GUARD_HOOK_FAILED',
                    'message': str(e),
                    'task_id': task_id,
                    'commits': commit_count,
                    'guard': 'pre-flight'
                }
    
    # ── Execute Promotion ────────────────────────────────────────────
    result = self.merge_request(
        task_id=task_id,
        branch_name=source_branch,
        force=force,
        skip_validation=skip_validation
    )
    
    if not result['success']:
        return result
    
    # ── Post-merge Validation ────────────────────────────────────────
    if not skip_validation:
        post_errors = []
        
        try:
            validate_enum_registrations('main')
        except ValidationError as e:
            post_errors.append(f"Enum: {e}")
        
        try:
            validate_hook_syntax('main')
        except ValidationError as e:
            post_errors.append(f"Hook: {e}")
        
        try:
            validate_variable_changes('main')
        except ValidationError as e:
            post_errors.append(f"Variable: {e}")
        
        if post_errors:
            return {
                'success': False,
                'error': 'PROMOTE_GUARD_POST_VALIDATION_FAILED',
                'message': '; '.join(post_errors),
                'task_id': task_id,
                'guard': 'post-merge'
            }
    
    # Update task to done_main
    task['status'] = 'done_main'
    task['promoted_at'] = datetime.now().isoformat()
    task['promote_commit'] = result.get('merge_commit')
    self._save_task(task)
    
    return {
        'success': True,
        'task_id': task_id,
        'merge_commit': result.get('merge_commit'),
        'commits': result.get('commits_merged'),
        'validation': 'full' if not skip_validation else 'skipped'
    }

## Phase 201: Large Merge Pre-flight Validation

**Date:** 2026-04-09  
**Task:** IMPL_large_merge_preflight_2026-04-09  
**Status:** IMPLEMENTED

### Purpose

Prevent Wave 3 regression where 56-commit merge (`harness-eta`) passed through 
`merge_request()` without validation. Large merges (>20 commits) now require 
mandatory pre-flight checks.

### Thresholds

| Commit Count | Classification | Requirements |
|--------------|----------------|--------------|
| ≤ 20 | Standard merge | Normal validation |
| 21-50 | Large merge | Integration tests, no force push, hook validation, enum validation, variable validation |
| > 50 | Extra-large merge | All large merge requirements + explicit approval flag |

### Pre-flight Checks

#### 1. Integration Tests (`require_integration_tests`)
- Full integration test suite must pass
- Must contain `LARGE_MERGE` test markers
- Run via: `pytest tests/integration/ -v`

#### 2. Force Push Prevention (`require_no_force_push`)
- `force=True` parameter rejected for large merges
- Forces manual conflict resolution

#### 3. Hook Syntax Validation (`validate_hook_syntax`)
- Validates `generate_claude_md.py` compiles
- Dry-run hook execution
- Checks for syntax errors in changed hook files

#### 4. Enum Registration Validation (`validate_enum_registrations`)
- All MCP strategy enums must be registered
- Prevents orphaned enum definitions
- Checks both inline and registry file registrations

#### 5. Variable Change Validation (`validate_variable_changes`)
- Flags deletion of critical variables:
  - `_qa_skipped`
  - `_validation_bypassed`
  - `FORCE_MERGE`
  - `SKIP_HOOKS`
  - `bypass_checks`

### Post-merge Validation

After merge to main, re-run:
- `validate_enum_registrations('main')`
- `validate_hook_syntax('main')`
- `validate_variable_changes('main')`

Failures trigger automatic rollback via `git reset --hard HEAD~1`.
