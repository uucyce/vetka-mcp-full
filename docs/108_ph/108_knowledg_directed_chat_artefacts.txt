// Knowledge Time Force (твоя идея: время = knowledge)
function timeForce(alpha) {
  const nowNode = nodes.find(n => n.isNow);  // "Current time" node
  return nodes.map(node => ({
    x: nowNode.x + (node.time - nowTime) * 0.1 * alpha,  // Attract recent
    y: nowNode.y
  }));
}

// Folder Gravity (папки = wells)
function folderForce(alpha) {
  return nodes.map(node => {
    const parent = folders[node.parentId];  // Из metadata
    if (parent) {
      const dx = parent.x - node.x;
      const dy = parent.y - node.y;
      const dist = Math.sqrt(dx*dx + dy*dy);
      return {
        x: dx * (50 / dist) * alpha,  // Strong pull to folder
        y: dy * (50 / dist) * alpha
      };
    }
    return {x:0, y:0};
  });
}
