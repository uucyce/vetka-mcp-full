def build\_tag\_hierarchy(tags, file\_embeddings):  
    """  
    Build hierarchical tree of tags (like folder structure)  
      
    Algorithm:  
    1\. Find root tag (most central / general)  
    2\. For each other tag, find semantic parent  
    3\. Result: tree structure with parent-child relationships  
    """  
    import numpy as np  
    from sklearn.metrics.pairwise import cosine\_similarity  
      
    if not tags:  
        return \[\]  
      
    \# Compute tag centroids (average of file embeddings)  
    tag\_centroids \= {}  
    for tag in tags:  
        tag\_file\_embeddings \= \[file\_embeddings\[f.id\] for f in tag.files if f.id in file\_embeddings\]  
        if tag\_file\_embeddings:  
            tag\_centroids\[tag.id\] \= np.mean(tag\_file\_embeddings, axis=0)  
      
    \# Find root: tag closest to global centroid  
    all\_embeddings \= list(tag\_centroids.values())  
    global\_centroid \= np.mean(all\_embeddings, axis=0)  
      
    root\_tag \= None  
    min\_dist \= float('inf')  
    for tag in tags:  
        if tag.id in tag\_centroids:  
            dist \= 1 \- cosine\_similarity(\[tag\_centroids\[tag.id\]\], \[global\_centroid\])\[0\]\[0\]  
            if dist \< min\_dist:  
                min\_dist \= dist  
                root\_tag \= tag  
      
    \# Assign parents based on similarity (excluding self and descendants)  
    for tag in tags:  
        if tag \== root\_tag:  
            tag.parent\_tag \= None  
            tag.depth \= 0  
        else:  
            \# Find most similar tag that's not a descendant  
            best\_parent \= root\_tag  
            best\_sim \= 0  
              
            for candidate in tags:  
                if candidate \== tag:  
                    continue  
                if tag.id in tag\_centroids and candidate.id in tag\_centroids:  
                    sim \= cosine\_similarity(  
                        \[tag\_centroids\[tag.id\]\],   
                        \[tag\_centroids\[candidate.id\]\]  
                    )\[0\]\[0\]  
                      
                    \# Prefer candidates with lower depth (more general)  
                    depth\_penalty \= getattr(candidate, 'depth', 0\) \* 0.1  
                    adjusted\_sim \= sim \- depth\_penalty  
                      
                    if adjusted\_sim \> best\_sim:  
                        best\_sim \= adjusted\_sim  
                        best\_parent \= candidate  
              
            tag.parent\_tag \= best\_parent.id  
            tag.depth \= getattr(best\_parent, 'depth', 0\) \+ 1  
      
    return tags

def calculate\_hierarchical\_tag\_positions(tags):  
    """  
    Position tags using SAME logic as Directory Mode\!  
      
    Y \= depth \* LAYER\_HEIGHT  (like directory depth)  
    X \= parent\_x \+ angular\_offset  (fan distribution from parent)  
    """  
    TAG\_Y\_BASE \= 100  
    TAG\_LAYER\_HEIGHT \= 200  \# Vertical spacing between tag levels  
    TAG\_SPREAD \= 300  \# Horizontal spread among siblings  
      
    positions \= {}  
      
    \# Sort by depth to process parents first  
    sorted\_tags \= sorted(tags, key=lambda t: getattr(t, 'depth', 0))  
      
    for tag in sorted\_tags:  
        depth \= getattr(tag, 'depth', 0\)  
        parent\_id \= getattr(tag, 'parent\_tag', None)  
          
        \# Y \= depth-based (like directory mode\!)  
        y \= TAG\_Y\_BASE \+ depth \* TAG\_LAYER\_HEIGHT  
          
        \# X \= parent\_x \+ angular offset among siblings  
        if parent\_id and parent\_id in positions:  
            parent\_pos \= positions\[parent\_id\]  
              
            \# Find siblings (same parent)  
            siblings \= \[t for t in tags if getattr(t, 'parent\_tag', None) \== parent\_id\]  
            sibling\_index \= siblings.index(tag) if tag in siblings else 0  
            num\_siblings \= len(siblings)  
              
            \# Angular spread from parent  
            if num\_siblings \> 1:  
                normalized \= (sibling\_index / (num\_siblings \- 1)) \- 0.5  
                x\_offset \= normalized \* TAG\_SPREAD  
            else:  
                x\_offset \= 0  
              
            x \= parent\_pos\['x'\] \+ x\_offset  
        else:  
            \# Root tag at center  
            x \= 0  
          
        positions\[tag.id\] \= {  
            'x': x,  
            'y': y,  
            'z': 0,  
            'depth': depth,  
            'is\_tag': True  
        }  
      
    return positions

### **Step 3: Position Files Under Their Tags (Small Fan)**

def calculate\_file\_positions\_under\_tags(tags, tag\_positions, file\_embeddings):  
    """  
    Files fan out BELOW their parent tag  
      
    Like leaves on a branch \- small spread at the end  
    """  
    FILE\_SPREAD \= 50  \# Small fan under tag  
    FILE\_STEP\_Y \= 40  \# Vertical step between files  
      
    positions \= {}  
      
    for tag in tags:  
        if tag.id not in tag\_positions:  
            continue  
              
        tag\_pos \= tag\_positions\[tag.id\]  
        tag\_files \= sorted(tag.files, key=lambda f: f.knowledge\_level)  
        num\_files \= len(tag\_files)  
          
        for i, file in enumerate(tag\_files):  
            \# Y \= below tag, stacking down  
            y \= tag\_pos\['y'\] \+ (i \+ 1\) \* FILE\_STEP\_Y  
              
            \# X \= small fan around tag center  
            if num\_files \> 1:  
                normalized \= (i / (num\_files \- 1)) \- 0.5  
                x\_offset \= normalized \* FILE\_SPREAD  
            else:  
                x\_offset \= 0  
              
            x \= tag\_pos\['x'\] \+ x\_offset  
              
            positions\[file.id\] \= {  
                'x': x,  
                'y': y,  
                'z': 0,  
                'parent\_tag': tag.id,  
                'knowledge\_level': file.knowledge\_level  
            }  
      
    return positions

### **Step 4: Adaptive Gradient (Similarity-Based Spread)**

def compute\_adaptive\_spread(files, embeddings, base\_spread=100):  
    """  
    Adaptive gradient:  
    \- High similarity within group → tight spread  
    \- Low similarity → wide spread  
    """  
    if len(files) \< 2:  
        return base\_spread  
      
    \# Compute average pairwise similarity  
    file\_embs \= \[embeddings.get(f.id) for f in files if f.id in embeddings\]  
    if len(file\_embs) \< 2:  
        return base\_spread  
      
    from sklearn.metrics.pairwise import cosine\_similarity  
    import numpy as np  
      
    sim\_matrix \= cosine\_similarity(file\_embs)  
    \# Exclude diagonal (self-similarity)  
    np.fill\_diagonal(sim\_matrix, 0\)  
    avg\_sim \= sim\_matrix.sum() / (len(file\_embs) \* (len(file\_embs) \- 1))  
      
    \# High similarity (0.9+) → spread \= 0.3 \* base  
    \# Low similarity (0.5) → spread \= 1.0 \* base  
    spread\_factor \= 1.0 \- (avg\_sim \- 0.5) \* 1.4  
    spread\_factor \= max(0.3, min(1.0, spread\_factor))  
      
    return base\_spread \* spread\_factor
