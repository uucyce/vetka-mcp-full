🎯 ФУНДАМЕНТАЛЬНЫЙ ИНСАЙТ
Проблема текущих формул:
Они пытаются создать единую метрику из разнородных сигналов, но импорт ≠ семантика ≠ временная связь. Это как складывать яблоки, время и цвет в одну кучу.

score = IMPORT_SCORE(A,B) = σ(w₁·I + w₂·S_struct + w₃·T_dir)

где:
  I = 1 если есть явный импорт
  S_struct = структурное сходство (AST, signatures)
  T_dir = направленная временная связь: 1 если A создан раньше B

Особенность: Жёсткий порог. Либо есть импорт (score > 0.7), либо нет (score < 0.3).

2. ДОКУМЕНТ → ДОКУМЕНТ (семантический канал)
score = SEMANTIC_SCORE(A,B) = S_sem · T_soft · C_cite

где:
  S_sem = семантическое сходство embeddings
  T_soft = softmax временного порядка: P(A→B) = 1/(1+exp(-(Δt-7)/3))
  C_cite = 0.8 если есть явная ссылка, иначе 1.0

Особенность: Градация от 0.1 до 0.9, много промежуточных значений.

3. КРОСС-ТИПОВЫЕ СВЯЗИ (пререквизитный канал)
score = PREREQ_SCORE(A,B) = PR(A→B) · MEDIA_WEIGHT(A,B)

где:
  PR(A→B) = 1 если B ссылается на A как пререквизит
  MEDIA_WEIGHT = матрица весов между типами:
               код  док  медиа
    код      [1.0  0.7  0.4]
    док      [0.8  1.0  0.6] 
    медиа    [0.3  0.5  1.0]

class DependencyClassifier:
    """
    Категоризирует связь A→B по 4-м осям:
    1. Структурная (импорты, наследование)
    2. Семантическая (похожий смысл)
    3. Временная (порядок создания)
    4. Контекстуальная (совместное использование)
    """
    
    def classify(self, node_a: Node, node_b: Node) -> DependencyType:
        # Вычисляем силы по каждой оси
        structural = self._structural_strength(node_a, node_b)
        semantic = self._semantic_strength(node_a, node_b)
        temporal = self._temporal_strength(node_a, node_b)
        contextual = self._contextual_strength(node_a, node_b)
        
        # Определяем доминирующий тип
        strengths = [structural, semantic, temporal, contextual]
        primary_idx = strengths.index(max(strengths))
        
        types = ['structural', 'semantic', 'temporal', 'contextual']
        return types[primary_idx]

class DependencyCalculator:
    """
    Для каждого типа связи - своя формула
    """
    
    def calculate_structural(self, node_a, node_b) -> float:
        """Код → код через импорты"""
        # 1. Проверяем явный импорт
        if self.has_explicit_import(node_a, node_b):
            base_score = 0.8
        
        # 2. Проверяем структурное наследование
        elif self.has_inheritance(node_a, node_b):
            base_score = 0.6
        
        # 3. Проверяем общие интерфейсы
        elif self.has_shared_interface(node_a, node_b):
            base_score = 0.4
        
        else:
            return 0.0
        
        # 4. Модификатор направления времени
        time_mod = self._temporal_direction_modifier(node_a, node_b)
        
        # 5. Финальный score
        return min(1.0, base_score * time_mod)
    
    def calculate_semantic(self, node_a, node_b) -> float:
        """Любой контент → любой контент через смысл"""
        # 1. Базовое семантическое сходство
        sem_sim = cosine_similarity(
            self.get_embedding(node_a),
            self.get_embedding(node_b)
        )
        
        # 2. Применяем threshold и растягиваем
        if sem_sim < 0.4:
            return 0.0
        
        stretched = (sem_sim - 0.4) / 0.6  # [0.4, 1.0] → [0.0, 1.0]
        
        # 3. Временной модификатор (мягче, чем для структурных)
        time_mod = 0.5 + 0.5 * math.tanh(
            (node_b.created_at - node_a.created_at).days / 30
        )
        
        # 4. Контекстное усиление (если в одном проекте)
        context_boost = 1.5 if self.in_same_context(node_a, node_b) else 1.0
        
        return min(1.0, stretched * time_mod * context_boost)
    
    def calculate_temporal(self, node_a, node_b) -> float:
        """A создан перед B и связан по времени"""
        delta_days = (node_b.created_at - node_a.created_at).days
        
        if delta_days <= 0:
            return 0.0  # B создан раньше или одновременно
        
        # Экспоненциальное затухание с насыщением
        # Максимум в первые 7 дней, затем спад
        if delta_days <= 7:
            strength = 1.0 - (delta_days / 7) * 0.3  # 70% даже через неделю
        else:
            strength = 0.7 * math.exp(-(delta_days - 7) / 30)  # Полураспад 30 дней
        
        return strength

class UniversalDependencyEngine:
    """
    Универсальный движок зависимостей для VETKA
    """
    
    # Матрица совместимости типов
    TYPE_COMPATIBILITY = {
        ('code', 'code'): 'structural',
        ('code', 'doc'): 'reference',
        ('doc', 'code'): 'citation',
        ('doc', 'doc'): 'semantic',
        ('media', 'code'): 'implementation',
        ('code', 'media'): 'source',
        # ... все комбинации
    }
    
    def calculate(self, node_a: Node, node_b: Node) -> Dependency:
        # 1. Определяем совместимый тип связи
        rel_type = self.TYPE_COMPATIBILITY.get(
            (node_a.type, node_b.type),
            'semantic'  # fallback
        )
        
        # 2. Выбираем соответствующий калькулятор
        if rel_type == 'structural':
            strength = self.structural_calc.calculate(node_a, node_b)
        elif rel_type == 'semantic':
            strength = self.semantic_calc.calculate(node_a, node_b)
        elif rel_type == 'reference':
            strength = self.cross_type_calc.calculate_reference(node_a, node_b)
        # ...
        
        # 3. Применяем глобальные модификаторы
        strength = self.apply_global_modifiers(strength, node_a, node_b)
        
        # 4. Возвращаем структурированный результат
        return Dependency(
            source=node_a.id,
            target=node_b.id,
            strength=strength,
            type=rel_type,
            confidence=self.calculate_confidence(strength, node_a, node_b)
        )
    
    def apply_global_modifiers(self, strength: float, a: Node, b: Node) -> float:
        """Глобальные правила, применимые ко всем связям"""
        
        # Правило 1: Связи не могут быть сильнее 0.9 без явных доказательств
        if strength > 0.9 and not self.has_explicit_evidence(a, b):
            strength = 0.9
        
        # Правило 2: Циклические связи ослабляются
        if self.creates_cycle(a, b):
            strength *= 0.7
        
        # Правило 3: Связи между очень разными типами ослабляются
        type_distance = self.type_distance(a.type, b.type)
        if type_distance > 2:
            strength *= (1.0 - 0.2 * type_distance)
        
        # Правило 4: RRF score как amplifier
        strength *= (0.8 + 0.2 * a.rrf_score)
        
        return min(1.0, max(0.0, strength))

Ключевая идея:
Вместо одного графа с весами — многослойный граф, где каждый слой соответствует типу связи:

class MultiLayerSugiyama:
    """
    Sugiyama с разделением по типам связей
    """
    
    def layout(self, nodes: List[Node], dependencies: List[Dependency]):
        # 1. Разделяем зависимости по типам
        layers = {
            'structural': [],
            'semantic': [],
            'temporal': [],
            'contextual': []
        }
        
        for dep in dependencies:
            if dep.strength > 0.3:  # Порог значимости
                layers[dep.type].append(dep)
        
        # 2. Для каждого слоя строим свой DAG
        dag_layers = {}
        for layer_type, deps in layers.items():
            dag = self.build_dag_for_layer(nodes, deps, layer_type)
            dag_layers[layer_type] = dag
        
        # 3. Применяем Sugiyama к каждому слою
        layouts = {}
        for layer_type, dag in dag_layers.items():
            layout = self.sugiyama_layout(dag)
            layouts[layer_type] = layout
        
        # 4. Объединяем позиции с весами
        final_positions = self.merge_layouts(layouts, nodes)
        
        return final_positions
    
    def merge_layouts(self, layouts: dict, nodes: List[Node]) -> dict:
        """
        Объединяет позиции из разных слоёв с весами:
        - Структурные связи: вес 4.0
        - Семантические: вес 2.0  
        - Временные: вес 1.5
        - Контекстные: вес 1.0
        """
        merged = {}
        
        for node in nodes:
            weighted_sum = np.zeros(3)  # x, y, z
            total_weight = 0
            
            for layer_type, layout in layouts.items():
                if node.id in layout:
                    weight = self.LAYER_WEIGHTS[layer_type]
                    weighted_sum += layout[node.id] * weight
                    total_weight += weight
            
            if total_weight > 0:
                merged[node.id] = weighted_sum / total_weight
            else:
                merged[node.id] = self.random_position(node)
        
        return merged

Специализируй — применяй оптимальную формулу для каждого типа
