171 def _format_search_results_for_stream(results: List[Dict]) -> str:
172     """
173     Format hybrid search results for system prompt injection in streaming.
174     Phase 114.8: Converts nested search dicts into readable markdown.
175
176     Args:
177         results: List of search result dicts from hybrid_search.search()
178
179     Returns:
180         Formatted markdown string for system prompt
181     """
182     if not results:
183         return ""
184
185     formatted = "## Pre-fetched Search Results\n\n"
186     for i, result in enumerate(results[:5], 1):
187         path = result.get("path", result.get("file_path", "unknown"))
187         score = result.get("score", 0)
188         content = result.get("content", "")[:500]
189         source = result.get("source", "unknown")
190
191         formatted += f"### Result {i} ({source}, score: {score:.2f})\n"
192         formatted += f"**Path:** `{path}`\n"
193         formatted += f"**Preview:**\n```\n{content}\n```\n\n"
194
195     return formatted
196
