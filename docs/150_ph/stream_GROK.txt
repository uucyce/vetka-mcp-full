### 3. Backend — Mycelium Pipeline (агентный движок)

В `agent_pipeline.py` добавляем **structured streaming**:

```python
class AgentPipeline:
    async def execute(self, task: str, ...):
        async for event in self._stream_execution(task, ...):
            await self.ws_broadcaster.broadcast({
                "type": "agent_stream",
                "data": event
            })

    async def _stream_execution(self, task: str, ...):
        # Шаг 1: Начало
        yield {
            "type": "reasoning",
            "content": "Начинаю анализ задачи...",
            "status": "executing"
        }

        # Шаг 2: Chain-of-Thought
        reasoning = await self._generate_reasoning(task)
        yield {
            "type": "reasoning",
            "content": reasoning,
            "status": "done"
        }

        # Шаг 3: Tool Call Proposal
        tool_call = await self._propose_tool_call(reasoning)
        if tool_call:
            yield {
                "type": "tool_call_proposed",
                "tool_name": tool_call["name"],
                "tool_args": tool_call["args"],
                "status": "pending"
            }

            # Ждём ответа пользователя (approve/edit/reject)
            decision = await self._wait_for_user_decision(tool_call["id"])
            
            if decision["action"] == "reject":
                yield {"type": "tool_call_rejected", "reason": decision["reason"]}
                return

            elif decision["action"] == "edit":
                tool_call["args"] = decision["edited_args"]
                yield {
                    "type": "tool_call_edited",
                    "tool_args": tool_call["args"]
                }

            # Выполняем инструмент
            yield {"type": "tool_call_executing", "tool_name": tool_call["name"]}
            result = await self._execute_tool(tool_call)
            yield {
                "type": "tool_call_result",
                "tool_result": result,
                "status": "done"
            }

        # Шаг 4: Финальный ответ
        final = await self._generate_final_answer(...)
        yield {
            "type": "final_answer",
            "content": final,
            "status": "done"
        }
```

### 4. Backend — WebSocket для управления (approve/edit/reject)

В `mycelium_ws_server.py` (или где у вас WS):

```python
@router.websocket("/ws/agent-control")
async def agent_control_ws(websocket: WebSocket):
    await websocket.accept()
    pending_decisions: Dict[str, asyncio.Future] = {}

    try:
        while True:
            data = await websocket.receive_json()
            action = data["action"]
            tool_call_id = data["tool_call_id"]

            if action in ("approve", "reject", "edit"):
                if tool_call_id in pending_decisions:
                    future = pending_decisions.pop(tool_call_id)
                    future.set_result(data)
    finally:
        await websocket.close()
```

Метод ожидания решения:

```python
async def _wait_for_user_decision(self, tool_call_id: str):
    loop = asyncio.get_running_loop()
    future = loop.create_future()
    pending_decisions[tool_call_id] = future  # глобально или в сессии

    try:
        return await asyncio.wait_for(future, timeout=300)  # 5 мин
    except asyncio.TimeoutError:
        return {"action": "reject", "reason": "timeout"}
```

```tsx
// AgentStreamPanel.tsx
import { useEffect, useState } from 'react';
import { useMyceliumSocket } from '../../hooks/useMyceliumSocket';

type StreamEvent = {
  type: string;
  content?: string;
  tool_name?: string;
  tool_args?: Record<string, any>;
  status?: string;
  // ...
};

export default function AgentStreamPanel() {
  const { socket } = useMyceliumSocket();
  const [events, setEvents] = useState<StreamEvent[]>([]);
  const [pendingToolCalls, setPendingToolCalls] = useState<Record<string, StreamEvent>>({});

  useEffect(() => {
    if (!socket) return;

    socket.on('agent_stream', (event: StreamEvent) => {
      setEvents(prev => [...prev, event]);

      if (event.type === 'tool_call_proposed') {
        setPendingToolCalls(prev => ({ ...prev, [event.event_id]: event }));
      } else if (event.type === 'tool_call_result' || event.type === 'tool_call_rejected') {
        setPendingToolCalls(prev => {
          const next = { ...prev };
          delete next[event.event_id];
          return next;
        });
      }
    });

    return () => socket.off('agent_stream');
  }, [socket]);

  const handleDecision = (toolCallId: string, action: 'approve' | 'reject' | 'edit', editedArgs?: any) => {
    socket.emit('agent_control', {
      action,
      tool_call_id: toolCallId,
      edited_args: editedArgs
    });
  };

  return (
    <div className="agent-stream">
      {events.map((e, i) => (
        <div key={i} className={`event ${e.type}`}>
          {e.type === 'reasoning' && <div className="thinking">{e.content}</div>}
          {e.type === 'tool_call_proposed' && (
            <div className="tool-call-card">
              <h4>Tool: {e.tool_name}</h4>
              <pre>{JSON.stringify(e.tool_args, null, 2)}</pre>
              <div className="actions">
                <button onClick={() => handleDecision(e.event_id, 'approve')}>Approve</button>
                <button onClick={() => handleDecision(e.event_id, 'reject')}>Reject</button>
                {/* Edit modal можно добавить */}
              </div>
            </div>
          )}
          {e.type === 'tool_call_result' && (
            <div className="tool-result">Result: {JSON.stringify(e.tool_result)}</div>
          )}
        </div>
      ))}

      {Object.values(pendingToolCalls).length > 0 && (
        <div className="waiting">Waiting for approval...</div>
      )}
    </div>
  );
}
```

```python
class AgentPipeline:
    # ... ваш существующий __init__ ...

    # ===================================================================
    # НОВЫЙ: Structured Streaming Generator
    # ===================================================================

    async def _stream_event(self, event_type: str, data: Dict[str, Any]):
        """Единая точка отправки всех событий стрима"""
        event = {
            "event_id": f"evt_{int(time.time() * 1000000)}",
            "timestamp": time.time(),
            "agent_id": "dragon",                    # можно сделать multi-agent
            "model": getattr(self, "_last_used_model", "unknown"),

            "content": reasoning,
            "status": "done"

        tool_call = await self._propose_tool_call(reasoning, phase_type)
```

### 1. Backend — WebSocket для управления (approve/edit/reject)

Добавьте этот код в ваш существующий WebSocket файл (например `src/mcp/mycelium_ws_server.py` или отдельный `src/mcp/agent_control_ws.py`).

```python
"""
Agent Control WebSocket — approve/edit/reject tool calls in real-time
MARKER_2026_AGENT_CONTROL_WS
"""

from fastapi import APIRouter, WebSocket, WebSocketDisconnect
from typing import Dict, Any, Callable
import asyncio
import uuid
import json
from src.utils.logger import get_logger

logger = get_logger(__name__)

router = APIRouter()

# Глобальный словарь ожидающих решений (tool_call_id → future)
pending_decisions: Dict[str, asyncio.Future] = {}
pending_locks = asyncio.Lock()

@router.websocket("/ws/agent-control")
async def agent_control_endpoint(websocket: WebSocket):
    await websocket.accept()
    client_id = str(uuid.uuid4())[:8]
    logger.info(f"Agent Control WS connected: {client_id}")

    try:
        while True:
            raw_data = await websocket.receive_text()
            try:
                data = json.loads(raw_data)
            except json.JSONDecodeError:
                await websocket.send_json({"error": "Invalid JSON"})
                continue

            action = data.get("action")
            tool_call_id = data.get("tool_call_id")

            if not tool_call_id or action not in ("approve", "reject", "edit"):
                await websocket.send_json({"error": "Missing tool_call_id or invalid action"})
                continue

            async with pending_locks:
                if tool_call_id not in pending_decisions:
                    await websocket.send_json({"error": "No pending decision for this tool_call_id"})
                    continue

                future = pending_decisions.pop(tool_call_id)

            # Формируем решение
            decision = {
                "action": action,
                "tool_call_id": tool_call_id,
                "edited_args": data.get("edited_args") if action == "edit" else None,
                "reason": data.get("reason") if action == "reject" else None
            }

            # Разрешаем future
            if not future.done():
                future.set_result(decision)

            await websocket.send_json({
                "success": True,
                "message": f"Decision {action} accepted for {tool_call_id}"
            })

    except WebSocketDisconnect:
        logger.info(f"Agent Control WS disconnected: {client_id}")
    except Exception as e:
        logger.error(f"Agent Control WS error: {e}")
        await websocket.close(code=1011)
```

### 2. Интеграция в AgentPipeline (ждём решение пользователя)

Добавьте этот метод в класс `AgentPipeline`:

```python
async def _wait_for_user_decision(self, tool_call_id: str, timeout_sec: int = 300) -> Dict:
    """
    Ждём решения пользователя по tool call (approve/edit/reject)
    Возвращает словарь с action и edited_args (если edit)
    """
    loop = asyncio.get_running_loop()
    future = loop.create_future()

    async with pending_locks:
        pending_decisions[tool_call_id] = future

    try:
        decision = await asyncio.wait_for(future, timeout=timeout_sec)
        return decision

    except asyncio.TimeoutError:
        async with pending_locks:
            pending_decisions.pop(tool_call_id, None)
        logger.warning(f"Timeout waiting for decision on tool_call {tool_call_id}")
        return {"action": "reject", "reason": "User timeout (5 minutes)"}

    except Exception as e:
        async with pending_locks:
            pending_decisions.pop(tool_call_id, None)
        logger.error(f"Error in user decision wait: {e}")
        return {"action": "reject", "reason": f"Internal error: {str(e)}"}
```

Используйте его в `_stream_execution`:

```python
# Внутри генератора, после "tool_call_proposed"

if decision["action"] == "reject":
    yield await self._stream_event("tool_call_rejected", {
        "tool_name": tool_call["name"],
        "reason": decision.get("reason", "Rejected by user")
    })
    return

elif decision["action"] == "edit":
    tool_call["args"] = decision["edited_args"]
    yield await self._stream_event("tool_call_edited", {
        "tool_name": tool_call["name"],
        "tool_args": tool_call["args"]
    })
```
