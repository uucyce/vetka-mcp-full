async def run_enhanced_workflow(feature, workflow_id, socketio):
    """
    Execute workflow using Phase 5.4 Enhanced Agents
    
    1. PM Agent: Sequential Thinking decomposition
    2. Dev Agent: Composio action library + code generation
    3. Traditional QA: Testing & validation
    """
    try:
        print(f"\n{'='*70}")
        print(f"🚀 PHASE 5.4 WORKFLOW: {workflow_id}")
        print(f"📝 Feature: {feature[:100]}...")
        print(f"{'='*70}")
        
        start_time = time.time()
        
        # Step 1: PM Agent - Sequential Thinking Decomposition
        print(f"\n📋 Step 1: PM Agent (Sequential Thinking)...")
        socketio.emit('workflow_step', {
            'workflow_id': workflow_id,
            'step': 'pm_analysis',
            'status': 'in_progress',
            'timestamp': time.time()
        })
        
        pm_agent = get_pm_agent_enhanced()
        try:
            pm_analysis = await pm_agent.generate_plan({
                'task': feature,
                'complexity': 'MEDIUM'  # Default, will be classified later
            })
            
            socketio.emit('workflow_step', {
                'workflow_id': workflow_id,
                'step': 'pm_analysis',
                'status': 'complete',
                'data': pm_analysis,
                'timestamp': time.time()
            })
            print(f"✅ PM Analysis complete")
        except Exception as e:
            print(f"⚠️  PM Analysis error: {str(e)[:80]}")
            pm_analysis = None
        
        # Step 2: Dev Agent - Code Generation with Composio
        print(f"\n💻 Step 2: Dev Agent (Composio + Code Generation)...")
        socketio.emit('workflow_step', {
            'workflow_id': workflow_id,
            'step': 'code_generation',
            'status': 'in_progress',
            'timestamp': time.time()
        })
        
        dev_agent = get_dev_agent_enhanced()
        try:
            code, action_usages = await dev_agent.generate_code(
                task=feature,
                plan=str(pm_analysis) if pm_analysis else feature
            )
            
            # Validate code
            is_valid, issues = dev_agent.validate_generated_code(code)
            
            socketio.emit('workflow_step', {
                'workflow_id': workflow_id,
                'step': 'code_generation',
                'status': 'complete',
                'data': {
                    'code': code[:500] + '...' if len(code) > 500 else code,
                    'actions_used': [u.action_id for u in action_usages],
                    'valid': is_valid,
                    'issues': issues
                },
                'timestamp': time.time()
            })
            print(f"✅ Code generation complete ({len(action_usages)} actions)")
        except Exception as e:
            print(f"⚠️  Code generation error: {str(e)[:80]}")
            code = None
            action_usages = []
        
        # Step 3: Generate Tests
        print(f"\n🧪 Step 3: Generating Tests...")
        socketio.emit('workflow_step', {
            'workflow_id': workflow_id,
            'step': 'test_generation',
            'status': 'in_progress',
            'timestamp': time.time()
        })
        
        try:
            if code:
                tests = await dev_agent.generate_tests(code, action_usages)
                socketio.emit('workflow_step', {
                    'workflow_id': workflow_id,
                    'step': 'test_generation',
                    'status': 'complete',
                    'data': {
                        'tests': tests[:300] + '...' if len(tests) > 300 else tests,
                        'count': len(tests.split('def test_'))
                    },
                    'timestamp': time.time()
                })
                print(f"✅ Tests generated")
        except Exception as e:
            print(f"⚠️  Test generation error: {str(e)[:80]}")
        
        # Calculate duration
        duration = time.time() - start_time
        
        # Emit completion
        socketio.emit('workflow_complete', {
            'workflow_id': workflow_id,
            'phase54': True,
            'result': {
                'status': 'success',
                'pm_analysis': bool(pm_analysis),
                'code_generated': bool(code),
                'actions_used': len(action_usages),
                'duration': duration,
                'timestamp': time.time()
            }
        })
        
        print(f"\n✅ WORKFLOW COMPLETE: {workflow_id}")
        print(f"   Duration: {duration:.2f}s")
        print(f"   Actions used: {len(action_usages)}")
        print(f"{'='*70}\n")
        
    except Exception as e:
        error_msg = str(e)
        print(f"❌ WORKFLOW ERROR: {error_msg}")
        import traceback
        traceback.print_exc()
        
        socketio.emit('workflow_error', {
            'workflow_id': workflow_id,
            'error': error_msg,
            'phase54_error': True,
            'timestamp': time.time()
        })


def run_workflow_async(feature, workflow_id):
    """Execute workflow in background thread"""
    try:
        # Use enhanced agents if available
        if PHASE54_AVAILABLE:
            print(f"🚀 Using Phase 5.4 Enhanced Agents")
            loop = asyncio.new_event_loop()
            asyncio.set_event_loop(loop)
            loop.run_until_complete(run_enhanced_workflow(feature, workflow_id, socketio))
            loop.close()
        else:
            # Fallback to traditional orchestrator
            print(f"⚠️  Using traditional orchestrator (Phase 5.4 not available)")
            orchestrator = get_orchestrator()
            result = orchestrator.execute_full_workflow_streaming(feature, workflow_id)
            
            socketio.emit('workflow_complete', {
                'workflow_id': workflow_id,
                'result': result,
                'timestamp': time.time()
            })
        
    except Exception as e:
        error_msg = str(e)
        print(f"❌ Background thread error: {error_msg}")
        import traceback
        traceback.print_exc()
        
        socketio.emit('workflow_error', {
            'workflow_id': workflow_id,
            'error': error_msg,
            'timestamp': time.time()
        })


@socketio.on('start_workflow')
def handle_workflow(data):
    """
    Start agent workflow with real-time streaming
    
    Expected data:
    {
        'feature': 'Feature description',
        'workflow_id': 'optional-custom-id'
    }
    """
    feature = data.get('feature', '')
    workflow_id = data.get('workflow_id', str(uuid.uuid4())[:8])
    
    if not feature.strip():
        emit('workflow_error', {
            'workflow_id': workflow_id,
            'error': 'Feature description is empty',
            'timestamp': time.time()
        })
        return
    
    print(f"\n{'='*70}")
    print(f"📋 WORKFLOW QUEUED: {workflow_id}")
    print(f"📝 Feature: {feature[:100]}...")
    print(f"{'='*70}")
    
    emit('workflow_started', {
        'workflow_id': workflow_id,
        'feature': feature,
        'phase54': PHASE54_AVAILABLE,
        'timestamp': time.time()
    })
    
    # Run workflow in background thread
    thread = threading.Thread(
        target=run_workflow_async,
        args=(feature, workflow_id),
        daemon=True
    )
    thread.start()
    
    print(f"✅ Background thread started for {workflow_id}")
