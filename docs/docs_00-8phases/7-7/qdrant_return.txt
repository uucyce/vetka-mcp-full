@app.route("/api/eval/feedback/submit", methods=["POST"])
def submit_eval_feedback():
    """Submit user feedback on evaluation"""
    try:
        data = request.json or {}
        eval_id = data.get('evaluation_id', '')
        task = data.get('task', '')
        output = data.get('output', '')
        rating = data.get('rating', '')
        correction = data.get('correction', '')
        score = data.get('score', None)
        
        if not eval_id:
            return jsonify({'error': 'evaluation_id is required'}), 400
        
        memory = get_memory_manager()
        success = memory.save_feedback(
            evaluation_id=eval_id,
            task=task,
            output=output,
            rating=rating,
            correction=correction,
            score=score
        )
        
        # Also store in Feedback Loop v2 if available
        feedback_loop_success = False
        if FEEDBACK_LOOP_V2_AVAILABLE and feedback_loop:
            try:
                feedback_loop_success = feedback_loop.submit_feedback(
                    eval_id=eval_id,
                    task=task,
                    output=output,
                    rating=rating,
                    score=score or 0.0,
                    correction=correction
                )
                if feedback_loop_success:
                    print(f"✅ Feedback saved to FeedbackLoopV2: {eval_id}")
                else:
                    print(f"⚠️  FeedbackLoopV2 returned False: {eval_id}")
            except Exception as e:
                print(f"⚠️  FeedbackLoopV2 error (fallback): {e}")
        
        if success:
            print(f"📝 Feedback saved: {eval_id} → {rating}")
            return jsonify({
                'status': 'success',
                'message': f'Feedback saved to learning system ({rating})',
                'evaluation_id': eval_id,
                'learning_context': 'This will improve future similar tasks',
                'weaviate_saved': True,
                'feedback_loop_saved': feedback_loop_success
            }), 200
        else:
            print(f"❌ Failed to save feedback: {eval_id}")
            return jsonify({
                'status': 'error',
                'message': 'Failed to save feedback',
                'evaluation_id': eval_id
            }), 500
            
    except Exception as e:
        print(f"❌ Feedback endpoint error: {str(e)}")
        return jsonify({'error': str(e)}), 500

def run_workflow_async(feature, workflow_id):
    """Execute workflow in background thread"""
    with app.app_context():
        thread_name = f"WORKFLOW-{workflow_id}"
        threading.current_thread().name = thread_name
        print(f"[THREAD] Started workflow thread: {thread_name}")
        
        try:
            print(f"\n{'='*70}")
            print(f"🚀 WORKFLOW BACKGROUND EXECUTION: {workflow_id}")
            print(f"📝 Feature: {feature[:100]}...")
            print(f"{'='*70}")
            
            # Record workflow start in metrics
            if METRICS_AVAILABLE and metrics_engine:
                metrics_engine.record_workflow_start(workflow_id, feature)
            
            # Execute workflow with streaming
            orchestrator = get_orchestrator()
            result = orchestrator.execute_full_workflow_streaming(feature, workflow_id)
            
            # Record workflow completion in metrics
            if METRICS_AVAILABLE and metrics_engine:
                metrics_engine.record_workflow_complete(
                    workflow_id,
                    success=result.get('success', False)
                )
            
            # Emit completion to room subscribers only
            socketio.emit('workflow_complete', {
                'workflow_id': workflow_id,
                'result': result,
                'timestamp': time.time()
            }, room=workflow_id)
            
            print(f"✅ WORKFLOW COMPLETE: {workflow_id}")
            print(f"   Duration: {result.get('duration', 0):.2f}s")
            print(f"{'='*70}\n")
            
        except Exception as e:
            error_msg = str(e)
            print(f"❌ WORKFLOW ERROR in background thread: {error_msg}")
            import traceback
            traceback.print_exc()
            
            # Log error
            try:
                memory = get_memory_manager()
                memory.log_error(workflow_id, 'Orchestrator', error_msg)
            except:
                pass
            
            # Emit error to room subscribers only
            socketio.emit('workflow_error', {
                'workflow_id': workflow_id,
                'error': error_msg,
                'timestamp': time.time()
            }, room=workflow_id)


@socketio.on('start_workflow')
def handle_workflow(data):
    """Start agent workflow with real-time streaming"""
    feature = data.get('feature', '')
    workflow_id = data.get('workflow_id', str(uuid.uuid4())[:8])
    
    if not feature.strip():
        emit('workflow_error', {
            'workflow_id': workflow_id,
            'error': 'Feature description is empty',
            'timestamp': time.time()
        })
        return
    
    queue_size = executor._work_queue.qsize()
    if queue_size > 10:
        print(f"⚠️  Server busy: {queue_size} workflows in queue")
        emit('workflow_error', {
            'workflow_id': workflow_id,
            'error': f'Server busy: {queue_size} workflows queued',
            'timestamp': time.time()
        })
        return
    
    print(f"\n{'='*70}")
    print(f"📋 WORKFLOW QUEUED: {workflow_id}")
    print(f"📝 Feature: {feature[:100]}...")
    print(f"📊 Queue size: {queue_size + 1}/10")
    print(f"{'='*70}")
    
    join_room(workflow_id)
    
    emit('workflow_started', {
        'workflow_id': workflow_id,
        'feature': feature,
        'timestamp': time.time()
    }, room=workflow_id)
    
    executor.submit(run_workflow_async, feature, workflow_id)
    
    print(f"✅ Workflow submitted to executor: {workflow_id}")

                    result = autogen_ext.execute_autogen_workflow_with_eval(
                        feature_request=feature,
                        workflow_id=workflow_id
                    )
                    
                    socketio.emit('autogen_workflow_complete', {
                        'workflow_id': workflow_id,
                        'result': result,
                        'timestamp': time.time()
                    }, room=workflow_id)
                    
                    print(f"\n✅ AUTOGEN WORKFLOW COMPLETE: {workflow_id}")
                    print(f"   Score: {result.get('evaluation', {}).get('score', 0):.2f}")
                    print(f"{'='*70}\n")
                    
                except Exception as e:
                    error_msg = str(e)
                    print(f"\n❌ AUTOGEN WORKFLOW ERROR: {error_msg}")
                    socketio.emit('autogen_workflow_error', {
                        'workflow_id': workflow_id,
                        'error': error_msg,
                        'timestamp': time.time()
                    }, room=workflow_id)
        
        executor.submit(run_autogen)
        
        return jsonify({
            'workflow_id': workflow_id,
            'status': 'started',
            'message': 'Autogen workflow started (Phase 7)',
            'timestamp': time.time()
        }), 202
        
    except Exception as e:
        return jsonify({'error': str(e)}), 500
