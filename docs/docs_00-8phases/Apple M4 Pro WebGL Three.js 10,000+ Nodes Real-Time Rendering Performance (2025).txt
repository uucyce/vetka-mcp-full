geometry.setAttribute('color', new THREE.BufferAttribute(colors, 3));

const vertexShader = `
  attribute float size;
  varying vec3 vColor;
  void main() {
    vColor = color;
    vec4 mvPosition = modelViewMatrix * vec4(position, 1.0);
    gl_PointSize = size * (300.0 / -mvPosition.z); // LOD: scale size by distance
    gl_Position = projectionMatrix * mvPosition;

  varying vec3 vColor;
  void main() {
    gl_FragColor = vec4(vColor, 1.0);
  }

  vertexShader,
  fragmentShader,
  vertexColors: true,
});

- **LOD Techniques**:
  - **Distance-Based LOD**:

  - **Cluster LOD**:
    - Group distant nodes into clusters, rendering a single node with aggregated properties (e.g., average position/color).

  - **Spatial Partitioning**:
    - Use octrees or BVH to query visible nodes, culling non-visible ones.
